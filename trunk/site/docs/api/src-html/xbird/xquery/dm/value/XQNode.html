089        public XQNode getRoot() {
090            if(isRoot()) {
091                return this;
092            } else {
093                XQNode root = getDocumentNode();
094                if(root == null) { // document node may be null.
095                    XQNode p = this;
096                    while(true) {
097                        XQNode tmp = p.parent();
098                        if(tmp != null) {
099                            p = tmp;
100                        } else {
101                            break;
102                        }
103                    }
104                    assert (p != null);
105                    return p;
106                } else {
107                    return root;
108                }
109            }
110        }

127        public NodeType getType() {
128            byte kind = nodeKind();
129            final NodeType type;
130            switch(kind) {
131                case NodeKind.DOCUMENT:
132                    type = DocumentTest.ANY_DOCUMENT;
133                    break;
134                case NodeKind.ELEMENT:
135                    type = ElementTest.ANY_ELEMENT;
136                    break;
137                case NodeKind.ATTRIBUTE:
138                    type = AttributeTest.ANY_ATTRIBUTE;
139                    break;
140                case NodeKind.NAMESPACE:
141                    type = NamespaceTest.ANY_NAMESPACE;
142                    break;
143                case NodeKind.PROCESSING_INSTRUCTION:
144                    type = PITest.ANY_PI;
145                    break;
146                case NodeKind.COMMENT:
147                    type = NodeType.COMMENT;
148                    break;
149                case NodeKind.TEXT:
150                case NodeKind.CDATA:
151                    type = NodeType.TEXT;
152                    break;
153                case NodeKind.ANY:
154                    type = NodeType.ANYNODE;
155                    break;
156                default:
157                    throw new IllegalStateException("Illegal node kind was detected: " + kind);
158            }
159            return type;
160        }

213        @Override
214        public String toString() {
215            StringWriter sw = new StringWriter();
216            SAXWriter saxwriter = new SAXWriter(sw);
217            saxwriter.setXMLDeclaration(false);
218            SAXSerializer ser = new SAXSerializer(saxwriter, sw);
219            try {
220                ser.emit(this);
221            } catch (XQueryException e) {
222                return "failed at " + e.getStackTrace()[1] + ".\n" + e.getMessage();
223            }
224            return sw.toString();
225        }

230        @Override
231        public boolean equals(Object obj) {
232            if(this == obj) {
233                return true;
234            }
235            if(!(obj instanceof XQNode)) {
236                return false;
237            }
238            XQNode n = (XQNode) obj;
239            if(n instanceof DMNode) {
240                XQNode mydoc = getDocumentNode();
241                XQNode trgdoc = n.getDocumentNode();
242                if(mydoc != trgdoc) { // document root is not same
243                    XQNode myroot = getRoot();
244                    XQNode trgroot = n.getRoot();
245                    assert (myroot != null && trgroot != null);
246                    if(myroot != trgroot) { // root is not same
247                        return System.identityHashCode(myroot) == System.identityHashCode(trgroot);
248                    } else {
249                        long trgpos = n.getPosition();
250                        return _id == trgpos;
251                    }
252                } else {
253                    long trgpos = n.getPosition();
254                    return _id == trgpos;
255                }
256            } else {
257                assert (n.getDataModel() instanceof DocumentTableModel);
258                long mynodeid = getDocumentOrder();
259                long trgnodeid = n.getDocumentOrder();
260                return mynodeid == trgnodeid;
261            }
262        }

264        public int compareTo(Item node) {
265            if(!(node instanceof XQNode)) {
266                throw new XQRTException("err:XPTY0004", "XQNode#compareTo(Object) accepts only XQNode instance as an argument, but was.. "
267                        + node.getClass().getCanonicalName());
268            }
269            if(this == node) {
270                return 0;
271            }
272            XQNode n = (XQNode) node;
273            if(n instanceof DMNode) {
274                XQNode mydoc = getDocumentNode();
275                XQNode trgdoc = n.getDocumentNode();
276                if(mydoc != trgdoc) { // document root is not same
277                    XQNode myroot = getRoot();
278                    XQNode trgroot = n.getRoot();
279                    if(myroot != trgroot) { // root is not same
280                        return System.identityHashCode(myroot) > System.identityHashCode(trgroot) ? 1
281                                : -1;
282                    } else {
283                        long trgpos = n.getPosition();
284                        if(_id == trgpos) {
285                            return 0;
286                        }
287                        return _id > trgpos ? 1 : -1;
288                    }
289                } else {
290                    long trgpos = n.getPosition();
291                    if(_id == trgpos) {
292                        long mySerial = getSerialNumber();
293                        long trgSerial = n.getSerialNumber();
294                        return mySerial > trgSerial ? 1 : -1;
295                    }
296                    return _id > trgpos ? 1 : -1;
297                }
298            } else {
299                assert (n.getDataModel() instanceof DocumentTableModel);
300                long mynodeid = getDocumentOrder();
301                long trgnodeid = n.getDocumentOrder();
302                return mynodeid == trgnodeid ? 0 : (mynodeid > trgnodeid ? 1 : -1);
303            }
304        }

307            if(this instanceof DMNode) {
308                return (T) this;
309            }
310            final T proxy = XQueryDataModel.<T> createProxy(this);
311            return proxy;
312        }

352            if(fromIndex == 0 && toIndex < 1) {

364            if(_cloned++ == 0) {
365                return this;
366            }

370                throw new InternalError(e.getMessage());
371            }
