098        @Override
099        public <T extends AtomicValue> T castAs(AtomicType type, DynamicContext dynEnv)
100                throws XQueryException {
101            assert (type != null);
102            final int tid = type.getTypeId();
103            final AtomicValue atomv;
104            switch(tid) {
105                case TypeTable.UNTYPED_ATOMIC_TID:
106                    atomv = new UntypedAtomicValue(stringValue());
107                    break;
108                case TypeTable.STRING_TID:
109                    atomv = new XString(stringValue());
110                    break;
111                case TypeTable.DURATION_TID:
112                    atomv = this;
113                    break;
114                case TypeTable.YEAR_MONTH_DURATION_TID:
115                    atomv = new DurationValue(getYearMonthDuration(), YearMonthDurationType.YEARMONTH_DURATION);
116                    break;
117                case TypeTable.DAYTIME_DURATION_TID:
118                    atomv = new DurationValue(getDayTimeDuration(), DayTimeDurationType.DAYTIME_DURATION);
119                    break;
120                default:
121                    throw new TypeError("Could not cast '" + _type + "' as '" + type + "'");
122            }
123            return (T) atomv;
124        }

127            try {
128                return DatatypeFactory.newInstance().newDurationYearMonth(value.getSign() >= 0, value.getYears(), value.getMonths());
129            } catch (DatatypeConfigurationException e) {
130                throw new IllegalStateException(e);

138                throw new IllegalStateException(e);

152        public DurationValue multiply(double factor) {
153            final int sign = value.getSign();
154            if(sign == 0) {
155                return this;
156            }
157            final Duration d;
158            final int tid = _type.getTypeId();
159            if(tid == TypeTable.YEAR_MONTH_DURATION_TID) {
160                assert (sign != 0) : sign;
161                boolean isNegativeFactor = factor < 0;
162                boolean resIsPositive = (sign == -1) ? isNegativeFactor : !isNegativeFactor;
163                int y = value.getYears(), m = value.getMonths();
164                int months = y * 12 + m;
165                int absRes = (int) (months * Math.abs(factor));
166                int resYear = absRes / 12;
167                int resMonth = absRes % 12;
168                try {
169                    d = DatatypeFactory.newInstance().newDurationYearMonth(resIsPositive, resYear, resMonth);
170                } catch (DatatypeConfigurationException e) {
171                    throw new IllegalStateException(e);
172                }
173            } else {
174                d = value.multiply(BigDecimal.valueOf(factor));
175            }
176            return new DurationValue(d, getDurationType());
177        }

179        public int getYears() {
180            int years = value.getYears();
181            int months = Math.abs(getMonths());
182            if(months >= 12) {
183                int yadd = months / 12;
184                years += yadd;
185            }
186            int sign = value.getSign();
187            return sign == -1 ? -years : years;
188        }
189    
190        public int getMonths() {
191            int months = value.getMonths();
192            int days = Math.abs(getDays());
193            if(days > 31) {
194                int madd = days / 31;
195                months += madd;
196            }
197            int sign = value.getSign();
198            return sign == -1 ? -months : months;
199        }
200    
201        public int totalMonths() {
202            int years = value.getYears();
203            int months = (years * 12) + value.getMonths();
204            int sign = value.getSign();
205            return sign == -1 ? -months : months;
206        }

208        public int getDays() {
209            int days = value.getDays();
210            int hours = Math.abs(getHours());
211            if(hours > 23) {
212                int dadd = hours / 24;
213                days += dadd;
214            }
215            int sign = value.getSign();
216            return sign == -1 ? -days : days;
217        }
218    
219        public int getHours() {
220            int hours = value.getHours();
221            int minutes = Math.abs(getMinutes());
222            if(minutes > 60) {
223                int hadd = minutes / 60;
224                hours += hadd;
225            }
226            int sign = value.getSign();
227            return sign == -1 ? -hours : hours;
228        }
229    
230        public int getMinutes() {
231            int minutes = value.getMinutes();
232            int seconds = Math.abs(getSeconds());
233            if(seconds > 60) {
234                int madd = seconds / 60;
235                minutes += madd;
236            }
237            int sign = value.getSign();
238            return sign == -1 ? -minutes : minutes;
239        }
240    
241        public int getSeconds() {
242            int seconds = value.getSeconds();
243            int sign = value.getSign();
244            return sign == -1 ? -seconds : seconds;
245        }

247        private Number getSecondsField() {
248            return value.getField(DatatypeConstants.SECONDS);
249        }
250    
251        private double totalSeconds() {
252            final int days = value.getDays();
253            int hours = value.getHours();
254            if(days != 0) {
255                hours += days * 24;
256            }
257            int minutes = value.getMinutes();
258            if(hours != 0) {
259                minutes += hours * 60;
260            }
261            final Number secField = getSecondsField();
262            final double seconds;
263            if(secField == null) {
264                seconds = minutes * 60;
265            } else {
266                if(minutes == 0) {
267                    seconds = secField.doubleValue();
268                } else {
269                    seconds = secField.doubleValue() + (minutes * 60);
270                }
271            }
272            final int sign = value.getSign();
273            return (sign == -1) ? -seconds : seconds;
274        }

276        private boolean isYearMonthSet() {
277            return value.isSet(DatatypeConstants.YEARS) | value.isSet(DatatypeConstants.MONTHS);
278        }

281            return value.isSet(DatatypeConstants.DAYS) | value.isSet(DatatypeConstants.HOURS)
282                    | value.isSet(DatatypeConstants.MINUTES) | value.isSet(DatatypeConstants.SECONDS);
283        }

290        @Override
291        public boolean equals(Object o) {
292            if(o == this) {
293                return true;
294            }
295            if(o instanceof DurationValue) {
296                final DurationValue dv = (DurationValue) o;
297                if(isYearMonthSet() && !dv.isYearMonthSet()) {
298                    return totalMonths() == 0;
299                } else if(isDayTimeSet() && !dv.isDayTimeSet()) {
300                    return totalSeconds() == 0;
301                }
302                final int diffm = totalMonths() - dv.totalMonths();
303                if(diffm != 0) {
304                    return false;
305                }
306                final double diffs = totalSeconds() - dv.totalSeconds();
307                if(diffs != 0) {
308                    return false;
309                }
310                return true;
311            }
312            return false;
313        }

315        @Override
316        public int compareTo(Item trg) {
317            if(!(trg instanceof DurationValue)) {
318                throw new XQRTException("err:XPTY0004", "Imcomparable "
319                        + trg.getClass().getSimpleName() + " with DurationValue");
320            }
321            final DurationValue dv = (DurationValue) trg;
322            final int diffm = totalMonths() - dv.totalMonths();
323            if(diffm != 0) {
324                return diffm < 0 ? -1 : 1;
325            }
326            final double diffs = totalSeconds() - dv.totalSeconds();
327            if(diffs != 0) {
328                return diffs < 0 ? -1 : 1;
329            }
330            return 0;
331        }

333        @Override
334        public String toString() {
335            if(_canonical == null) {
336                final StringBuilder buf = new StringBuilder(128);
337                final int sign = value.getSign();
338                boolean yset = value.isSet(DatatypeConstants.YEARS);
339                boolean mset = value.isSet(DatatypeConstants.MONTHS);
340                boolean isset = yset | mset | value.isSet(DatatypeConstants.DAYS)
341                        | value.isSet(DatatypeConstants.HOURS) | value.isSet(DatatypeConstants.MINUTES)
342                        | value.isSet(DatatypeConstants.SECONDS);
343                if(sign == 0 && !isset) {
344                    this._canonical = "PT0S";
345                    return "PT0S";
346                }
347                if(sign == -1) {
348                    buf.append('-');
349                }
350                buf.append('P');
351                int years = value.getYears();
352                int months = value.getMonths();
353                int days = value.getDays();
354                int hours = value.getHours();
355                int minutes = value.getMinutes();
356                int seconds = value.getSeconds();
357                if(seconds > 60) {
358                    int madd = seconds / 60;
359                    minutes += madd;
360                    seconds %= 60;
361                }
362                if(minutes > 60) {
363                    int hadd = minutes / 60;
364                    hours += hadd;
365                    minutes %= 60;
366                }
367                if(hours > 23) {
368                    int dadd = hours / 24;
369                    if(dadd != 0) {
370                        days += dadd;
371                    }
372                    hours %= 24;
373                }
374                if(days > 31 && (mset | yset)) {
375                    int madd = days / 31;
376                    if(madd != 0) {
377                        months += madd;
378                    }
379                    days = days % 31;
380                }
381                if(months >= 12) {
382                    int yadd = months / 12;
383                    if(yadd != 0) {
384                        years += yadd;
385                    }
386                    months = months % 12;
387                }
388                if(years != 0) {
389                    buf.append(years);
390                    buf.append('Y');
391                }
392                final int tid = _type.getTypeId();
393                if(months != 0 || (years == 0 && tid == TypeTable.YEAR_MONTH_DURATION_TID)) {
394                    buf.append(months);
395                    buf.append('M');
396                }
397                if(tid != TypeTable.YEAR_MONTH_DURATION_TID) {
398                    if(days != 0) {
399                        buf.append(days);
400                        buf.append('D');
401                    }
402                    buf.append('T');
403                    if(hours != 0) {
404                        buf.append(hours);
405                        buf.append('H');
406                    }
407                    if(minutes != 0) {
408                        buf.append(minutes);
409                        buf.append('M');
410                    }
411                    if(seconds != 0) {
412                        Number sec = value.getField(DatatypeConstants.SECONDS);
413                        String secStr = sec.toString();
414                        String trimed = removeTrilingZeros(secStr);
415                        buf.append(trimed);
416                        buf.append('S');
417                    } else if(hours == 0 && minutes == 0) {
418                        if(days != 0) {
419                            buf.deleteCharAt(buf.length() - 1); // remove 'T'
420                        } else {
421                            buf.append("0S");
422                        }
423                    }
424                }
425                final String res = buf.toString();
426                this._canonical = res;
427                return res;
428            }
429            return _canonical;
430        }
431    
432        private static String removeTrilingZeros(final String src) {
433            if(src.indexOf('.') == -1) {
434                return src;
435            } else {
436                // remove trailing zeros
437                int last = src.length();
438                while(--last > 0) {
439                    if(src.charAt(last) != '0') {
440                        break;
441                    }
442                }
443                if(src.charAt(last) == '.') { // for the case 'xxx.000'
444                    --last;
445                }
446                return src.substring(0, last + 1);
447            }
448        }
