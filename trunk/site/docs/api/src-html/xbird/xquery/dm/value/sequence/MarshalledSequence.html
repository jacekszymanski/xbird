061    public final class MarshalledSequence extends AbstractSequence<Item> implements Externalizable {
062        private static final long serialVersionUID = 296919228793372877L;
063        private static final int BUFFERING_BLOCK_SIZE = 16384;
064        private static final boolean ENV_SER_PIPED = Boolean.parseBoolean(Settings.get("xbird.remote.ser.piped", "true"));
065        private static final boolean DEBUG_DESER_SPEED = System.getProperty("xbird.debug.deser_speed") != null;

072        private boolean _piped = ENV_SER_PIPED;

076        public MarshalledSequence(Sequence<? extends Item> seq, DynamicContext dynEnv) {
077            super(dynEnv);
078            if(seq == null) {
079                throw new IllegalArgumentException();
080            }
081            if(DEBUG_DESER_SPEED) {
082                this._entity = new ValueSequence(seq.materialize(), dynEnv);
083            } else {
084                this._entity = (Sequence<Item>) seq;
085            }
086            this._type = seq.getType();
087        }

109        public void writeExternal(ObjectOutput out) throws IOException {
110            out.writeObject(_type);
111            final boolean reaccessable = _reaccessable;
112            final boolean redirectable = _redirectable;
113            out.writeBoolean(redirectable);
114            if(redirectable) {
115                out.writeObject(_entity);
116            } else {
117                if(_entity instanceof IncrDecodedSequnece) {
118                    out.writeBoolean(true);
119                    IncrDecodedSequnece incr = ((IncrDecodedSequnece) _entity);
120                    incr._reaccessable = reaccessable;
121                    incr.writeExternal(out);
122                } else {
123                    out.writeBoolean(false);
124                    if(_piped) {
125                        out.writeBoolean(true);
126                        pipedOut(out, _entity, _remotePaging);
127                    } else {
128                        out.writeBoolean(false);
129                        bulkOut(out, _entity, _remotePaging);
130                    }
131                }
132            }
133            if(!reaccessable) {
134                this._entity = null;
135            }
136        }
137    
138        public void readExternal(ObjectInput in) throws IOException, ClassNotFoundException {
139            final Type type = (Type) in.readObject();
140            this._type = type;
141            final boolean redirectable = in.readBoolean();
142            if(redirectable) {
143                this._redirectable = true;
144                this._entity = (Sequence<Item>) in.readObject();
145            } else {
146                this._redirectable = false; // just for readability
147                final boolean isDecodedSeq = in.readBoolean();
148                if(isDecodedSeq) {
149                    IncrDecodedSequnece entity = IncrDecodedSequnece.readFrom(in);
150                    entity._type = type;
151                    this._entity = entity;
152                } else {
153                    final boolean piped = in.readBoolean();
154                    final XQEventDecoder decoder;
155                    if(piped) {
156                        this._piped = true;
157                        decoder = pipedIn(in, _reaccessable);
158                    } else {
159                        this._piped = false;
160                        decoder = bulkIn(in); // This is required for nested Object serialization/deserialization                    
161                    }
162                    if(DEBUG_DESER_SPEED) {
163                        try {
164                            this._entity = decoder.decode();
165                        } catch (XQueryException e) {
166                            throw new IllegalStateException("failed decoding", e);
167                        }
168                    } else {
169                        final IncrDecodedSequnece entity = new IncrDecodedSequnece(decoder, type);
170                        entity._piped = piped;
171                        this._entity = entity;
172                    }
173                }
174            }
175        }

183        public boolean next(IFocus<Item> focus) throws XQueryException {
184            final boolean hasNext = _entity.next(focus);
185            if(!hasNext) {
186                //focus.close();    // may already be closed
187                if(!_reaccessable) {
188                    this._entity = null;
189                }
190                return false;
191            }
192            return true;
193        }

199        private static final class IncrDecodedSequnece extends AbstractSequence<Item>
200                implements Externalizable {
201            private static final long serialVersionUID = 5692780446373650820L;
202            private static final float DECODE_UNIT_GROWTH = 1.4f;
203    
204            private final ArrayList<Item> _decodedItems = new ArrayList<Item>(256);
205            private final XDMTreeBuilder _treeBuilder;
206    
207            private/* final */XQEventDecoder _decoder; // may be null if decoding is already finished
208            private/* final */Type _type; // not null
209    
210            private transient boolean _decodeFinished = false;
211            private transient boolean _piped = false;
212            private transient boolean _reaccessable = true;
213    
214            private transient int _decodeUnit = 32;
215    
216            public IncrDecodedSequnece(XQEventDecoder decoder, Type type) {
217                super(DynamicContext.DUMMY);
218                this._treeBuilder = new XDMTreeBuilder();
219                this._decoder = decoder;
220                this._type = type;
221            }
222    
223            public IncrDecodedSequnece() {
224                super(DynamicContext.DUMMY);
225                this._treeBuilder = new XDMTreeBuilder();
226            }
227    
228            public void readExternal(ObjectInput in) throws IOException, ClassNotFoundException {
229                //this._type = (Type) in.readObject();
230                final ArrayList<Item> decodedItems = _decodedItems;
231                final int numDecoded = in.readInt();
232                for(int i = 0; i < numDecoded; i++) {
233                    Item item = (Item) in.readObject();
234                    decodedItems.add(item);
235                }
236                final boolean decodeFinished = in.readBoolean();
237                if(decodeFinished) {
238                    this._decodeFinished = true;
239                } else {
240                    this._decodeFinished = false;
241                    final boolean piped = in.readBoolean();
242                    if(piped) {
243                        this._piped = true;
244                        // avoid readStreamHeader()
245                        ObjectInputStream ois = (ObjectInputStream) in; //TODO REVIEWME too hacky
246                        this._decoder = pipedIn(new AppendedObjectInputStream(ois), _reaccessable);
247                    } else {
248                        this._piped = false;
249                        this._decoder = bulkIn(in);
250                    }
251                }
252            }
253    
254            public void writeExternal(ObjectOutput out) throws IOException {
255                //out.writeObject(_type);
256                final ArrayList<Item> decodedItems = _decodedItems;
257                final int numDecoded = decodedItems.size();
258                out.writeInt(numDecoded);
259                for(int i = 0; i < numDecoded; i++) {
260                    Item item = decodedItems.get(i);
261                    out.writeObject(item);
262                }
263                final boolean decodeFinished = _decodeFinished;
264                if(decodeFinished) {
265                    out.writeBoolean(true);
266                } else {
267                    out.writeBoolean(false);
268                    if(_piped && out instanceof ObjectOutputStream) {
269                        out.writeBoolean(true);
270                        if(_reaccessable) {
271                            incrPipedOutReaccessable((ObjectOutputStream) out);
272                        } else {
273                            incrPipedOut((ObjectOutputStream) out);
274                        }
275                    } else {// rarely comes here
276                        if(LOG.isDebugEnabled()) {
277                            LOG.debug("Unusually piped in/out streams in BULK mode.");
278                        }
279                        out.writeBoolean(false);
280                        incrBulkOut(out);
281                    }
282                }
283            }
284    
285            private void incrPipedOut(final ObjectOutputStream out) throws IOException {
286                assert (!_reaccessable);
287                final XQEventDecoder decoder = _decoder;
288                decoder.redirectTo(out);
289                decoder.close();
290            }
291    
292            /**
293             * @link http://java.sun.com/javase/technologies/core/basic/serializationFAQ.jsp#appendSerialStream
294             */
295            private void incrPipedOutReaccessable(final ObjectOutputStream out) throws IOException {
296                assert (_reaccessable);
297                final FastMultiByteArrayOutputStream bufOut = new FastMultiByteArrayOutputStream(BUFFERING_BLOCK_SIZE);
298                final TeeOutputStream tee = new TeeOutputStream(out, bufOut);
299                final ObjectOutputStream objectOut = new AppendingObjectOutputStream(tee);
300                final XQEventDecoder decoder = _decoder;
301                decoder.redirectTo(objectOut);
302                decoder.close();
303                objectOut.flush();
304    
305                // Because input of decoder fully read, this section required for re-object serialization, etc.
306                final byte[][] buf = bufOut.toMultiByteArray();
307                final int bufTotalSize = bufOut.size();
308                bufOut.clear();
309                final FastMultiByteArrayInputStream inputBuf = new FastMultiByteArrayInputStream(buf, BUFFERING_BLOCK_SIZE, bufTotalSize);
310                if(!_reaccessable) {
311                    inputBuf.setCleanable(true);
312                }
313                final ObjectInputStream objInput = new AppendedObjectInputStream(inputBuf);
314                this._decoder = new XQEventDecoder(objInput); // replace Old Decoder with fresh Decoder
315            }
316    
317            private void incrBulkOut(final ObjectOutput out) throws IOException {
318                final FastByteArrayOutputStream bufOut = new FastByteArrayOutputStream(16384);
319                final ObjectOutputStream objectOut = new ObjectOutputStream(bufOut);
320                final XQEventDecoder decoder = _decoder;
321                decoder.redirectTo(objectOut);
322                decoder.close();
323                objectOut.flush();
324                final byte[] buf = bufOut.toByteArray();
325                bufOut.clear();
326    
327                // Because input of decoder fully read, this section required for re-object serialization, etc.
328                if(_reaccessable) {
329                    final FastByteArrayInputStream inputBuf = new FastByteArrayInputStream(buf);
330                    final ObjectInputStream objectInput = new ObjectInputStream(inputBuf);
331                    this._decoder = new XQEventDecoder(objectInput); // replace Old Decoder with fresh Decoder
332                }
333    
334                final int buflen = buf.length;
335                if(LOG.isDebugEnabled()) {
336                    LOG.debug("encoded sequence size: " + buflen);
337                }
338                out.writeInt(buflen);
339                out.write(buf);
340            }

342            public boolean next(IFocus<Item> focus) throws XQueryException {
343                if(focus.reachedEnd()) {
344                    return false;
345                }
346    
347                final ArrayList<Item> decodedItems = _decodedItems;
348                final int decodedCount = decodedItems.size();
349                final int curPos = focus.getContextPosition();
350                if(curPos < decodedCount) {
351                    Item it = decodedItems.get(curPos);
352                    if(it == null) {
353                        throw new IllegalStateException();
354                    }
355                    focus.setContextItem(it);
356                    return true;
357                }
358    
359                if(_decodeFinished) {
360                    focus.setReachedEnd(true);
361                    focus.closeQuietly();
362                    return false;
363                }
364                final XDMTreeBuilder treeBuilder = _treeBuilder;
365                final int last = _decodeUnit - 1;
366                for(int i = 0; i < _decodeUnit; i++) {
367                    final Item item;
368                    try {
369                        item = _decoder.decodeItem(treeBuilder);
370                    } catch (IOException e) {
371                        throw new XQueryException("failed decoding an Item", e);
372                    } finally {
373                        treeBuilder.reset();
374                    }
375                    if(item != null) {
376                        decodedItems.add(item);
377                        if(i == 0) {
378                            focus.setContextItem(item);
379                        } else if(i == last) {
380                            _decodeUnit *= DECODE_UNIT_GROWTH;
381                            return true;
382                        }
383                    } else {
384                        this._decodeFinished = true;
385                        _decoder.close();
386                        if(i > 0) {
387                            return true;
388                        } else {
389                            break;
390                        }
391                    }
392                }
393                focus.setReachedEnd(true);
394                focus.closeQuietly();
395                return false;
396            }

399                return _type;
400            }

402            public static IncrDecodedSequnece readFrom(ObjectInput in) throws IOException,
403                    ClassNotFoundException {
404                final IncrDecodedSequnece seq = new IncrDecodedSequnece();
405                seq.readExternal(in);
406                return seq;
407            }

410        private static XQEventDecoder pipedIn(ObjectInput in, boolean reaccessable) throws IOException {
411            final FastMultiByteArrayOutputStream bufOut = new FastMultiByteArrayOutputStream(BUFFERING_BLOCK_SIZE);
412            final ObjectOutputStream objectOut = new ObjectOutputStream(bufOut);
413            final XQEventDecoder decoder = new XQEventDecoder(in);
414            decoder.redirectTo(objectOut);
415            objectOut.flush();
416    
417            final byte[][] buf = bufOut.toMultiByteArray();
418            final int totalBufSize = bufOut.size();
419            bufOut.clear();
420    
421            // Because input of decoder fully read, this section required for re-object serialization, etc.
422            final FastMultiByteArrayInputStream inputBuf = new FastMultiByteArrayInputStream(buf, BUFFERING_BLOCK_SIZE, totalBufSize);
423            if(!reaccessable) {
424                inputBuf.setCleanable(true);
425            }
426            final ObjectInputStream objectInput = new ObjectInputStream(inputBuf);
427            final XQEventDecoder newDecoder = new XQEventDecoder(objectInput); // replace Old Decoder with fresh Decoder
428            return newDecoder;
429        }

431        private static void pipedOut(ObjectOutput out, Sequence<Item> entity, boolean remotePaing)
432                throws IOException {
433            final XQEventEncoder encoder = new XQEventEncoder(out);
434            if(remotePaing) {
435                encoder.setRemotePaging(true);
436            }
437            try {
438                encoder.emit(entity);
439            } catch (XQueryException xqe) {
440                throw new IllegalStateException("failed encoding", xqe);
441            } catch (Throwable e) {
442                LOG.fatal(PrintUtils.prettyPrintStackTrace(e));
443                throw new IllegalStateException("failed encoding", e);
444            }
445        }
446    
447        private static XQEventDecoder bulkIn(final ObjectInput input) throws IOException {
448            final int inputLen = input.readInt();
449            final byte[] buf = new byte[inputLen];
450            input.readFully(buf);
451            if(LOG.isDebugEnabled()) {
452                LOG.debug("decoding sequence size: " + inputLen);
453            }
454            FastByteArrayInputStream inputBuf = new FastByteArrayInputStream(buf);
455            ObjectInputStream objInput = new ObjectInputStream(inputBuf);
456            XQEventDecoder decoder = new XQEventDecoder(objInput);
457            return decoder;
458        }
459    
460        private static void bulkOut(ObjectOutput out, Sequence<Item> entity, boolean remotePaing)
461                throws IOException {
462            final FastByteArrayOutputStream bufOut = new FastByteArrayOutputStream(16384);
463            final ObjectOutputStream objOut = new ObjectOutputStream(bufOut);
464            final XQEventEncoder encoder = new XQEventEncoder(objOut);
465            if(remotePaing) {
466                encoder.setRemotePaging(true);
467            }
468            try {
469                encoder.emit(entity);
470            } catch (XQueryException xqe) {
471                throw new IllegalStateException("failed encoding", xqe);
472            } catch (Throwable e) {
473                LOG.fatal(e);
474                throw new IllegalStateException("failed encoding", e);
475            } finally {
476                objOut.flush();
477            }
478            final byte[] buf = bufOut.toByteArray();
479            bufOut.clear();
480            final int buflen = buf.length;
481            if(LOG.isDebugEnabled()) {
482                LOG.debug("encoded sequence size: " + buflen);
483            }
484            out.writeInt(buflen);
485            out.write(buf);
486        }
