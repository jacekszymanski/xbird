021    package xbird.xquery.dm.value.sequence;
022    
023    import java.util.HashSet;
024    import java.util.Set;
025    
026    import xbird.xquery.DynamicError;
027    import xbird.xquery.XQRTException;
028    import xbird.xquery.dm.value.node.DMAttribute;
029    import xbird.xquery.meta.DynamicContext;
030    import xbird.xquery.misc.QNameTable.QualifiedName;
031    
032    /**
033     * 
034     * <DIV lang="en"></DIV>
035     * <DIV lang="ja"></DIV>
036     * 
037     * @author Makoto YUI (yuin405+xbird@gmail.com)
038     */
039    public final class AttrNodeSequence<T extends DMAttribute> extends NodeSequence<T> {
040        private static final long serialVersionUID = 1L;
041        
042        private final Set<QualifiedName> attrSet = new HashSet<QualifiedName>(6);
043    
044        public AttrNodeSequence(DynamicContext dynEnv) {
045            super(dynEnv);
046        }
047    
048        /**
049         * @throws DynamicError 
050         * @link http://www.w3.org/TR/xquery/#ERRXQDY0025
051         */
052        @Override
053        public void addItem(T attr) {
054            QualifiedName qname = attr.nodeName();
055            boolean uniq = attrSet.add(qname);
056            if(!uniq) {
057                throw new XQRTException("err:XQDY0025", "Duplicate attribute: " + qname);
058            }
059            super.addItem(attr);
060        }
061    
062    }
