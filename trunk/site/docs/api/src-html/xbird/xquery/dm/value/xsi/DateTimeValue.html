046    public class DateTimeValue extends AtomicValue {
047        private static final long serialVersionUID = 2749362266774252238L;
048        public static final int ID = 7;

050        protected/* final */XMLGregorianCalendar value;
051        private transient String _canonical = null;

053        public DateTimeValue() {
054            super();
055        }

057        public DateTimeValue(String literal, DateTimeBaseType type) {
058            super(literal, type);
059            this.value = XsDatatypeFactory.createXMLGregorianCalendar(literal);
060        }

062        public DateTimeValue(XMLGregorianCalendar value, DateTimeBaseType type) {
063            super(value.toXMLFormat(), type);
064            this.value = value;
065        }

067        public XMLGregorianCalendar getValue() {
068            return value;
069        }
070    
071        public DateTimeBaseType getDateTimeType() {
072            return (DateTimeBaseType) _type;
073        }
074    
075        public XMLGregorianCalendar toJavaObject() throws XQueryException {
076            return value;
077        }

079        @Override
080        public int compareTo(Item trg) {
081            if(!(trg instanceof DateTimeValue)) {
082                throw new XQRTException("err:XPTY0004", "Imcomparable "
083                        + trg.getClass().getSimpleName() + " with DateTimeValue");
084            }
085            XMLGregorianCalendar trgUri = ((DateTimeValue) trg).value;
086            return value.compare(trgUri);
087        }

089        @SuppressWarnings("unchecked")
090        @Override
091        public <T extends AtomicValue> T castAs(AtomicType trgType, DynamicContext dynEnv)
092                throws XQueryException {
093            final int ttid = trgType.getTypeId();
094            if(ttid == TypeTable.NOTATION_TID || ttid == TypeTable.ANY_ATOM_TID) {
095                throw new StaticError("err:XPST0080", "Illegal target type: " + trgType);
096            }
097            final AtomicType type = _type;
098            if(type == trgType) {
099                return (T) this;
100            }
101            final boolean mayCastable = TypeTable.isMayCastable(type, trgType);
102            if(mayCastable) {
103                if(trgType instanceof DateTimeBaseType) {
104                    final DateTimeBaseType dtType = (DateTimeBaseType) trgType;
105                    return (T) dtType.createInstance(value);
106                } else {
107                    final String sv = stringValue();
108                    final String literal = trgType.processWhitespace(sv);
109                    try {
110                        return (T) trgType.createInstance(literal, type, dynEnv);
111                    } catch (RuntimeException e) {
112                        throw new DynamicError("err:FORG0001", "cast to '" + trgType + "' failed: "
113                                + sv, e);
114                    }
115                }
116            }
117            throw new DynamicError("err:XPTY0004", "failed to cast '" + _type + "' as '" + trgType
118                    + "': " + _lexicalValue);
119        }

126        @Override
127        public String toString() {
128            if(_canonical == null) {
129                final DateTimeBaseType targetType = getDateTimeType();
130                final boolean settime = targetType.isTimeSet();
131                if(settime) {
132                    final int hour = value.getHour();
133                    if(hour == 24 && value.getMinute() == 0 && value.getSecond() == 0) {
134                        // 2004-03-31T24:00:00" => "2004-04-01T00:00:00"
135                        value.add(XsDatatypeFactory.getDatatypeFactory().newDuration(true, 0, 0, 0, 0, 0, 1));
136                        value.setSecond(0);
137                    }
138                }
139                final StringBuilder buf = new StringBuilder(128);
140                if(targetType.isDateSet()) {
141                    final boolean yset = targetType.isYearSet();
142                    final boolean mset = targetType.isMonthSet();
143                    final boolean dset = targetType.isDaySet();
144                    if(yset) {
145                        final int year = value.getYear();
146                        if(year != DatatypeConstants.FIELD_UNDEFINED) {
147                            if(year == 0) {// There is no year 0, and '0000' is not a valid lexical representation
148                                appendPaddedZeros(1, 4, buf);
149                            } else if(year < 0) {
150                                buf.append('-');
151                                appendPaddedZeros(-year, 4, buf);
152                            } else {
153                                appendPaddedZeros(year, 4, buf);
154                            }
155                        } else {
156                            buf.append('-');
157                        }
158                    } else if(mset || dset) {
159                        buf.append('-');
160                    }
161                    if(mset) {
162                        buf.append('-');
163                        int month = value.getMonth();
164                        if(month != DatatypeConstants.FIELD_UNDEFINED) {
165                            appendPaddedZeros(month, 2, buf);
166                        }
167                    } else if(dset) {
168                        buf.append('-');
169                    }
170                    if(dset) {
171                        buf.append('-');
172                        int day = value.getDay();
173                        if(day != DatatypeConstants.FIELD_UNDEFINED) {
174                            appendPaddedZeros(day, 2, buf);
175                        }
176                    }
177                    if(settime) {
178                        buf.append('T');
179                    }
180                }
181                if(settime) {
182                    final int hour = value.getHour();
183                    if(hour == DatatypeConstants.FIELD_UNDEFINED) {
184                        buf.append("00:00:00");
185                    } else {
186                        appendPaddedZeros(hour, 2, buf);
187                        buf.append(':');
188                        final int minute = value.getMinute();
189                        assert (minute != DatatypeConstants.FIELD_UNDEFINED);
190                        appendPaddedZeros(minute, 2, buf);
191                        buf.append(':');
192                        final int second = value.getSecond();
193                        assert (second != DatatypeConstants.FIELD_UNDEFINED);
194                        appendPaddedZeros(second, 2, buf);
195                        int millisec = value.getMillisecond();
196                        if(millisec != DatatypeConstants.FIELD_UNDEFINED && millisec != 0) {
197                            buf.append('.');
198                            String ms = Integer.toString(millisec);
199                            // remove trailing zeros
200                            int last = ms.length();
201                            while(--last > 0) {
202                                if(ms.charAt(last) != '0') {
203                                    break;
204                                }
205                            }
206                            String trimed = ms.substring(0, last + 1);
207                            buf.append(trimed);
208                        }
209                    }
210                }
211                int tz = value.getTimezone();
212                if(tz != DatatypeConstants.FIELD_UNDEFINED) {
213                    if(tz == 0) {
214                        buf.append('Z');
215                    } else if(tz != DatatypeConstants.FIELD_UNDEFINED) {
216                        if(tz < 0) {
217                            buf.append('-');
218                            tz *= -1;
219                        } else {
220                            buf.append('+');
221                        }
222                        appendPaddedZeros(tz / 60, 2, buf);
223                        buf.append(':');
224                        appendPaddedZeros(tz % 60, 2, buf);
225                    }
226                }
227                String res = buf.toString();
228                this._canonical = res;
229                return res;
230            }
231            return _canonical;
232        }
233    
234        private static void appendPaddedZeros(final int value, final int digits, final StringBuilder buf) {
235            final String s = Integer.toString(value);
236            for(int i = s.length(); i < digits; i++) {//padding
237                buf.append('0');
238            }
239            buf.append(s);
240        }

242        public void readExternal(ObjectInput in) throws IOException, ClassNotFoundException {
243            this._lexicalValue = ObjectUtils.readString(in);
244            this._type = AtomicType.readAtomicType(in);
245            this.value = (XMLGregorianCalendar) in.readObject();
246        }
247    
248        public void writeExternal(ObjectOutput out) throws IOException {
249            ObjectUtils.writeString(out, _lexicalValue);
250            _type.writeExternal(out);
251            out.writeObject(value); //TODO XMLGregorianCalendar might not be Serializable
252        }

255        public int getIdentifier() {
256            return ID;
257        }
