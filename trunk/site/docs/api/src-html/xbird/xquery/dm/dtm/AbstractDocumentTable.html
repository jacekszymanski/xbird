052        /** @see NodeKind */
053        private static final int KIND_BITS = 3; /* .....111 */
054        private static final int KIND_MASK = (1 << KIND_BITS) - 1;
055        private static final int FLAG_FIRST_ENTRY = 1 << 3; /* ....1000 */
056        private static final int FLAG_LAST_ENTRY = 1 << 4; /* ...1|0000 */// FIXME currently not used
057    
058        /*  for element */
059    
060        private static final int FLAG_HAS_CHILD = 1 << 5; /* ..10|0000 */
061    
062        private static final int ATTR_COUNT_SHIFT = 6;
063        private static final int ATTR_COUNT_BITS = 10;
064        /* 0000|0000|0000|0000|1111|1111|1100|0000 */
065        private static final int ATTR_COUNT_MASK = ((1 << ATTR_COUNT_BITS) - 1) << ATTR_COUNT_SHIFT;
066    
067        private static final int NAMESPACE_COUNT_BITS = 8;
068        private static final int NAMESPACE_COUNT_SHIFT = ATTR_COUNT_SHIFT + ATTR_COUNT_BITS;
069        /* 0000|0000|1111|1111|0000|0000|0000|0000 */
070        private static final int NAMESPACE_COUNT_MASK = ((1 << NAMESPACE_COUNT_BITS) - 1) << NAMESPACE_COUNT_SHIFT;

098        public AbstractDocumentTable(final DbCollection coll, final String docName, final PropertyMap docProps) {
099            if(docProps == null) {
100                this._blockPtr = 0L;
101            } else {
102                final String ptr = docProps.getProperty(KEY_BLOCK_PTR + docName);
103                this._blockPtr = (ptr == null) ? 0L : Long.parseLong(ptr);
104            }
105            this._nameTable = coll.getSymbols().getQnameTable();
106            try {
107                this._strChunk = coll.getStringChunk();
108            } catch (IOException e) {
109                throw new IllegalStateException("failed loading string chunk of the collection: "
110                        + coll.getCollectionName(), e);
111            }
112        }

130        public final long openNode(final byte ev) {
131            final long newnode = _blockPtr;
132            if(_prevNode != -1L) {
133                setData(_prevNode + NEXTSIB_OFFSET, newnode);
134            }
135            if(ev == NodeKind.DOCUMENT) {
136                setFlag(newnode, ev | FLAG_FIRST_ENTRY | FLAG_LAST_ENTRY);
137            } else {
138                setFlag(newnode, ev);
139            }
140            fastSetData(newnode + PARENT_OFFSET, _curNode);
141            this._curNode = newnode;
142            this._prevNode = -1L;
143            _blockPtr += BLOCKS_PER_NODE;
144            return newnode;
145        }
146    
147        public void closeNode() {
148            this._prevNode = _curNode;
149            final long parentCandicate = parent(_curNode);
150            if(parentCandicate != 0L) {
151                if(!hasChildAt(parentCandicate)) {
152                    fastSetFlag(parentCandicate, FLAG_HAS_CHILD);
153                    fastSetFlag(_curNode, FLAG_FIRST_ENTRY);
154                }
155            } else {// document node has child            
156                fastSetFlag(0, FLAG_HAS_CHILD);
157            }
158            this._curNode = parentCandicate;
159        }

161        public final long putAttribute(final byte ev, final long parent, final int index, final int attsSize) {
162            final long newnode = _blockPtr;
163            setData(newnode, index == 0 /* first */? (ev | FLAG_FIRST_ENTRY) : ev);
164            if(index == (attsSize - 1)) { // last
165                fastSetFlag(newnode, FLAG_LAST_ENTRY);
166                // increment counter
167                final int cntFlag;
168                if(ev == NodeKind.ATTRIBUTE) {
169                    cntFlag = (attsSize << ATTR_COUNT_SHIFT) & ATTR_COUNT_MASK;
170                } else if(ev == NodeKind.NAMESPACE) {
171                    cntFlag = (attsSize << NAMESPACE_COUNT_SHIFT) & NAMESPACE_COUNT_MASK;
172                } else {
173                    throw new IllegalStateException("Illegal node kind: " + NodeKind.resolveName(ev));
174                }
175                fastSetFlag(parent, cntFlag);
176            }
177            fastSetData(newnode + PARENT_OFFSET, parent);
178            _blockPtr += BLOCKS_PER_NODE;
179            return newnode;
180        }

182        public final byte getNodeKindAt(final long at) {
183            return (byte) (dataAt(at) & KIND_MASK);
184        }
185    
186        public final long firstChild(final long curnode) {
187            final long code = dataAt(curnode);
188            if(!hasChild(code)) {
189                return -1L;
190            }
191            final int nscnt = getNamespaceCount(code);
192            final int attcnt = getAttributeCount(code);
193            final long addr = curnode + ((nscnt + attcnt) * BLOCKS_PER_NODE) + BLOCKS_PER_NODE;
194            return addr;
195        }
196    
197        public final long lastChild(final long curnode) {
198            long lastchild = firstChild(curnode);
199            if(lastchild == -1L) {
200                return -1L;
201            }
202            while(true) {
203                final long sb = nextSibling(lastchild);
204                if(sb == 0L) {
205                    return lastchild;
206                } else {
207                    lastchild = sb;
208                }
209            }
210        }
211    
212        public final long nextSibling(final long curnode) {
213            return dataAt(curnode + NEXTSIB_OFFSET);
214        }
215    
216        public final long parent(final long curnode) {
217            return dataAt(curnode + PARENT_OFFSET);
218        }
219    
220        public final long previousSibling(final long curnode) {
221            if(isFirstAt(curnode)) {
222                return -1L;
223            }
224            final long parent = parent(curnode);
225            if(parent == 0L) {
226                return -1L;
227            }
228            for(long prevAddr = curnode - BLOCKS_PER_NODE; prevAddr > 0; prevAddr -= BLOCKS_PER_NODE) {
229                final long prevParent = parent(prevAddr);
230                if(prevParent == parent) {
231                    return prevAddr;
232                }
233            }
234            return -1L;
235        }

237        public final long getAttribute(final long elemid, final int index) {
238            final long code = dataAt(elemid);
239            final int attcnt = getAttributeCount(code);
240            if((attcnt > 0) && (index < attcnt)) {
241                final int nscnt = getNamespaceCount(code);
242                final long firstAttrAddr = elemid + (nscnt * BLOCKS_PER_NODE);
243                final long addr = firstAttrAddr + (BLOCKS_PER_NODE * (index + 1));
244                return addr;
245            }
246            return -1L;
247        }

265        public final long getNamespaceDecl(final long elemid, final int index) {
266            final long code = dataAt(elemid);
267            final byte nodeKind = nodeKind(code);
268            if((nodeKind != NodeKind.ELEMENT) && (nodeKind != NodeKind.DOCUMENT)) {
269                throw new IllegalStateException("Illegal node kind: " + NodeKind.resolveName(nodeKind));
270            }
271            final int nscnt = getNamespaceCount(code);
272            if(index < nscnt) {
273                final long addr = elemid + (BLOCKS_PER_NODE * (index + 1));
274                return addr;
275            }
276            return -1L;
277        }

304        public final String stringValue(final long curnode) {
305            final byte nodeKind = getNodeKindAt(curnode);
306            switch(nodeKind) {
307                case NodeKind.ELEMENT:
308                case NodeKind.DOCUMENT:
309                    // concat descendant text node values
310                    final long firstChild = firstChild(curnode);
311                    if(firstChild == -1L) {
312                        return ""; // has no descendant text nodes.
313                    } else {
314                        final StringBuilder buf = new StringBuilder(384);
315                        final long nextsib = nextSibling(curnode);
316                        if(nextsib == 0L) {
317                            final long til = _blockPtr;
318                            for(long n = firstChild; n < til; n += BLOCKS_PER_NODE) {
319                                final long parent = parent(n);
320                                if(parent < curnode) {
321                                    break;
322                                }
323                                final byte nk = getNodeKindAt(n);
324                                if(nk == NodeKind.TEXT) {
325                                    getText(n, buf);
326                                }
327                            }
328                        } else {
329                            for(long n = firstChild; n < nextsib; n += BLOCKS_PER_NODE) {
330                                final byte nk = getNodeKindAt(n);
331                                if(nk == NodeKind.TEXT) {
332                                    getText(n, buf);
333                                }
334                            }
335                        }
336                        return buf.toString(); // TODO cache
337                    }
338                case NodeKind.ATTRIBUTE:
339                case NodeKind.NAMESPACE:
340                case NodeKind.PROCESSING_INSTRUCTION:
341                case NodeKind.COMMENT:
342                case NodeKind.TEXT:
343                case NodeKind.CDATA:
344                    return getText(curnode);
345                default:
346                    throw new IllegalStateException("Illegal node kind: " + nodeKind);
347            }
348        }

398        private static final byte nodeKind(final long code) {
399            return (byte) (code & KIND_MASK);
400        }
401    
402        private static final boolean hasChild(final long code) {
403            return (code & FLAG_HAS_CHILD) != 0;
404        }
405    
406        private final boolean hasChildAt(final long addr) {
407            return (dataAt(addr) & FLAG_HAS_CHILD) != 0L;
408        }
409    
410        private final boolean isFirstAt(final long addr) {
411            return isFirst(dataAt(addr));
412        }
413    
414        private static final boolean isFirst(final long code) {
415            return (code & FLAG_FIRST_ENTRY) != 0;
416        }
417    
418        private static final int getAttributeCount(final long code) {
419            return (int) ((ATTR_COUNT_MASK & code) >> ATTR_COUNT_SHIFT);
420        }
421    
422        private static final int getNamespaceCount(final long code) {
423            return (int) ((NAMESPACE_COUNT_MASK & code) >> NAMESPACE_COUNT_SHIFT);
424        }

438        @Override
439        public String toString() {
440            final DocumentTableModel dtm = new DocumentTableModel(this);
441            final FastMultiByteArrayOutputStream baos = new FastMultiByteArrayOutputStream();
442            final SAXWriter writer = new SAXWriter(baos);
443            final SAXSerializer ser = new SAXSerializer(writer);
444            try {
445                dtm.export(0, ser);
446            } catch (XQueryException e) {
447                PrintUtils.prettyPrintStackTrace(e, new PrintStream(baos));
448            }
449            return baos.toString();
450        }

452        public void close() throws IOException {
453            if(_refcount.getAndDecrement() == 0) {
454                _close();
455            }
456        }

458        protected final void _close() throws IOException {
459            this._nameTable = null;
460            if(_strChunk != null) {
461                _strChunk.close();
462                this._strChunk = null;
463            }
