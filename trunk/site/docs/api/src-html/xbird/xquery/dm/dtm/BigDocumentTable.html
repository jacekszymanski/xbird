064        private static final int PAGE_SHIFT = Integer.parseInt(Settings.get("xbird.database.pageshift", "9"));
065        /** Page size: 4K */
066        private static final int PAGE_LENGTH = 1 << PAGE_SHIFT;
067        private static final int PAGE_SIZE = PAGE_LENGTH * Primitives.LONG_BYTES;
068        private static final int PAGE_MASK = PAGE_LENGTH - 1;

074        private static final int DEFAULT_EXTENT_SIZE = PAGE_SIZE * DEFAULT_EXTENT_THRESHOLD;

108        protected final void enlarge(final int pages) {
109            final long[][] new_block = new long[pages][];
110            final int oldlen = _block.length;
111            System.arraycopy(_block, 0, new_block, 0, oldlen);
112            new_block[oldlen] = new long[PAGE_LENGTH];
113            this._block = new_block;
114        }

116        protected void setData(final long at, final long value) {
117            if(value > 0x7fffffffL) {
118                throw new IllegalArgumentException("Illegal value: " + value);
119            }
120            final long lp = at >> PAGE_SHIFT;
121            if(lp > 0x7fffffffL) {
122                throw new IllegalStateException("Illegal page number: " + lp);
123            }
124            final int page = (int) lp;
125            if(page >= _block.length) {
126                enlarge((int) (_block.length * ENLARGE_PAGES_FACTOR));
127            }
128            long[] block = _block[page];
129            if(block == null) {
130                block = new long[PAGE_LENGTH];
131                this._block[page] = block;
132            }
133            final int i = (int) (at & PAGE_MASK);
134            block[i] = value;
135        }

158            }

160        }

167        public void flush(final DbCollection coll, final String docName) throws IOException,
168                DbException {
169            PropertyMap properties = coll.getCollectionProperties();
170    
171            // #1 sync local symbol table with shared symbol table
172            if(BUILDIDX_AT_BULKLOAD) {
173                flushAuxiliaries(coll, docName, properties);
174            }
175            coll.flushSymbols();
176    
177            // #2 write properties
178            writeProperties(docName, properties);
179    
180            // #3 write str chunk
181            _strChunk.flush(coll, docName, properties);
182    
183            properties.save();
184    
185            // #4 write block in segment paged files
186            FixedSegments paged = getPaged(coll, docName);
187            pageOutSegments(paged, 0, true);
188            close();
189        }
190    
191        private final void writeProperties(final String docName, final PropertyMap properties)
192                throws IOException {
193            properties.setProperty(KEY_DTM_CLASS + docName, DTM_CLASS);
194            properties.setProperty(KEY_BLOCK_PTR + docName, String.valueOf(_blockPtr));
195            int usedPages = (int) _blockPtr / PAGE_LENGTH;
196            if(_blockPtr % PAGE_LENGTH != 0) {
197                usedPages += 1;
198            }
199            properties.setProperty(KEY_USED_PAGES + docName, String.valueOf(usedPages));
200        }

202        private final void flushAuxiliaries(final DbCollection coll, final String docName, final PropertyMap docProps)
203                throws DbException {
204            final DocumentTableModel dtm = new DocumentTableModel(this);
205            final LabelingHandler handler = new LabelingHandler(this, coll, docName, docProps);
206            try {
207                dtm.export(0, handler);
208            } catch (XQueryException e) {
209                throw new IllegalStateException(e);
210            }
211            final QNameTable sharedQNames = coll.getSymbols().getQnameTable();
212            sharedQNames.load(_nameTable);
213            sharedQNames.setDirty(true);
214        }

216        public FixedSegments getPaged(final DbCollection coll, final String docName) {
217            final File baseDir = new File(coll.getAbsolutePath());
218            final File segFile = new File(baseDir, docName + DTM_SEGMENT_FILE_SUFFIX);
219            return new FixedSegments(segFile, PAGE_SIZE);
220        }

222        private final int pageOutSegments(final FixedSegments paged, final int beginPage, boolean toend)
223                throws IOException {
224            final long[][] blockRef = _block;
225            final FastMultiByteArrayOutputStream buf = new FastMultiByteArrayOutputStream(DEFAULT_EXTENT_SIZE);
226            final long lp = (_blockPtr >> PAGE_SHIFT) + 1;
227            if(lp > 0x7fffffffL) {
228                throw new IllegalStateException("Illegal page number: " + lp);
229            }
230            final int usedPages = (int) lp;
231            int cnt = 0;
232            for(int i = beginPage; i <= usedPages; i += DEFAULT_EXTENT_THRESHOLD) {
233                final int pages;
234                if((i + DEFAULT_EXTENT_THRESHOLD) > usedPages) {
235                    if(!toend) {
236                        break;
237                    }
238                    pages = usedPages - i;
239                } else {
240                    pages = DEFAULT_EXTENT_THRESHOLD;
241                }
242                int addr = i;
243                final int limit = Math.min(i + pages, blockRef.length);
244                for(int p = i; p < limit; p++) {
245                    final long[] ref = blockRef[p];
246                    if(ref == null) {
247                        if(buf.size() > 0) {
248                            final byte[] b = buf.toByteArray();
249                            buf.reset();
250                            paged.bulkWrite(addr, b);
251                        }
252                    } else {
253                        if(buf.size() == 0) {
254                            addr = p;
255                        }
256                        buf.writeLongs(ref, 0, ref.length);
257                        blockRef[p] = null;
258                        ++cnt;
259                    }
260                }
261                if(buf.size() > 0) {
262                    final byte[] b = buf.toByteArray();
263                    buf.reset();
264                    paged.bulkWrite(addr, b);
265                }
266            }
267            buf.close();
268            paged.flush(false);
269            if(LOG.isInfoEnabled()) {
270                LOG.info("paged out docuemnt table seqment(" + beginPage + " - " + (beginPage + cnt)
271                        + ") to disk: " + paged.getFile().getAbsolutePath());
272            }
273            return cnt;
274        }

281        public static final class PersistentBigDocumentTable extends BigDocumentTable {
282            private static final long serialVersionUID = 5187802360153356037L;
283    
284            private static final int pageOutThreshold = 32;
285    
286            private final FixedSegments _paged;
287            /** 4K * 16 * 1024 = 64MB */
288            private final IntHash<long[]> _pageReadCache;
289            private final ObservableLongLRUMap<long[]> _dirtyBuffer;
290            private final PagingProfile _profile = new PagingProfile();
291    
292            private int lastPagedOutSegment = 0;
293            private int prevBlocks = -1;
294            private byte[] tmpBlock = null;
295    
296            private final ILock _lock = new AtomicBackoffLock();

298            protected PersistentBigDocumentTable(final DbCollection coll, final String docName, final PropertyMap docProps) {
299                super(coll, docName, docProps);
300                final FixedSegments paged = loadSegments(coll, docName);
301                this._paged = paged;
302                final IntHash<long[]> readCache = ENV_USE_2QCACHE ? new Int2QCache<long[]>(PAGE_CACHE_SIZE)
303                        : new IntLRUMap<long[]>(PAGE_CACHE_SIZE);
304                this._pageReadCache = readCache;
305                this._dirtyBuffer = new ObservableLongLRUMap<long[]>(256, 32, new BufferWriter(paged, readCache));
306            }

308            public PersistentBigDocumentTable(final DbCollection coll, final String docName) {
309                super(coll, docName);
310                final FixedSegments paged = loadSegments(coll, docName);
311                this._paged = paged;
312                final IntLRUMap<long[]> readCache = new IntLRUMap<long[]>(PAGE_CACHE_SIZE);
313                this._pageReadCache = readCache;
314                this._dirtyBuffer = new ObservableLongLRUMap<long[]>(256, 32, new BufferWriter(paged, readCache));
315            }

317            @Override
318            public void close() throws IOException {
319                if(_refcount.getAndDecrement() == 0) {
320                    _dirtyBuffer.clear();
321                    PrivilegedAccessor.unsafeSetField(this, PersistentBigDocumentTable.class, "_pageReadCache", null);
322                    _close();
323                    _paged.close();
324                }
325            }

327            @Override
328            protected void finalize() throws Throwable {
329                if(_dirtyBuffer != null) {
330                    _dirtyBuffer.clear();
331                }
332                _close();
333                if(_paged != null) {
334                    _paged.close();
335                }

338            private static FixedSegments loadSegments(final DbCollection coll, final String docName) {
339                final File segFile = new File(coll.getDirectory(), docName + DTM_SEGMENT_FILE_SUFFIX);
340                return new FixedSegments(segFile, PAGE_SIZE);

345                return _paged;

350                return _profile;

353            @Override
354            public void closeNode() {
355                final long lcp = _curNode >>> PAGE_SHIFT;
356                final int curpage = (int) lcp;
357                final int curSeg = (curpage / DEFAULT_EXTENT_THRESHOLD) - 1;
358                final int diff = curSeg - lastPagedOutSegment;
359                if(diff > pageOutThreshold) {
360                    final int beginPage = lastPagedOutSegment * DEFAULT_EXTENT_THRESHOLD;
361                    final int endPage = curSeg * DEFAULT_EXTENT_THRESHOLD - 1;
362                    try {
363                        pageOutSegments(_paged, beginPage, endPage);
364                    } catch (IOException e) {
365                        throw new IllegalStateException("paged out failed", e);
366                    }
367                    this.lastPagedOutSegment = curSeg;
368                }
369                super.closeNode(); // move up to parent
370            }

372            private void pageOutSegments(final FixedSegments paged, final int beginPage, final int endPage)
373                    throws IOException {
374                final long[][] blockRef = _block;
375                final FastMultiByteArrayOutputStream buf = new FastMultiByteArrayOutputStream(DEFAULT_EXTENT_SIZE);
376                for(int i = beginPage; i <= endPage; i += DEFAULT_EXTENT_THRESHOLD) {
377                    final int pages;
378                    if((i + DEFAULT_EXTENT_THRESHOLD - 1) > endPage) {
379                        break;
380                    } else {
381                        pages = DEFAULT_EXTENT_THRESHOLD;
382                    }
383                    int addr = i;
384                    final int limit = Math.min(i + pages, blockRef.length - 1);
385                    for(int p = i; p < limit; p++) {
386                        final long[] ref = blockRef[p];
387                        if(ref == null) {
388                            if(buf.size() > 0) {
389                                final byte[] b = buf.toByteArray();
390                                buf.reset();
391                                paged.bulkWrite(addr, b);
392                            }
393                        } else {
394                            if(buf.size() == 0) {
395                                addr = p;
396                            }
397                            buf.writeLongs(ref, 0, ref.length);
398                            blockRef[p] = null;
399                        }
400                    }
401                    if(buf.size() > 0) {
402                        final byte[] b = buf.toByteArray();
403                        buf.reset();
404                        paged.bulkWrite(i, b);
405                    }
406                }
407                buf.close();
408                paged.flush(false);
409                if(LOG.isInfoEnabled()) {
410                    LOG.info("paged out docuemnt table seqment(" + beginPage + " - " + endPage
411                            + ") to disk: " + paged.getFile().getAbsolutePath());
412                }
413            }

426            @Override
427            public long dataAt(final long at) {
428                if(at > _blockPtr) {// not initialized                
429                    return 0;
430                }
431                final long lp = at >> PAGE_SHIFT;
432                assert (lp <= 0x7fffffffL) : "Last page number is invalid: " + lp;
433                final int requiredPage = (int) lp;
434                final long[][] blocksRef = _block;
435                if(requiredPage >= blocksRef.length) {//prevent index of out error
436                    return 0;
437                }
438                final long[] page = blocksRef[requiredPage];
439                if(page == null) {// page in DTM segments
440                    if(at == _blockPtr) {
441                        return 0;
442                    }
443                    final long[] readBlock = pageInSegments(_paged, requiredPage);
444                    final int bi = (int) (at & PAGE_MASK);
445                    return readBlock[bi];
446                }
447                final int bi = (int) (at & PAGE_MASK);
448                return page[bi];
449            }

451            @Override
452            protected void setData(final long at, final long value) {
453                final long lp = at >> PAGE_SHIFT;
454                if(lp > 0x7fffffffL) {
455                    throw new IllegalStateException("Last page number is invalid: " + lp);
456                }
457                final int requiredPage = (int) lp;
458                final int blocklen = _block.length;
459                if((requiredPage < blocklen) && (_block[requiredPage] == null)) {
460                    final int segnum = requiredPage / DEFAULT_EXTENT_THRESHOLD;
461                    if(segnum < lastPagedOutSegment) {
462                        final long[] readBlock = pageIn(_paged, requiredPage);
463                        final int bi = (int) (at & PAGE_MASK);
464                        readBlock[bi] = value;
465                        _dirtyBuffer.put(requiredPage, readBlock);
466                        return;
467                    }
468                }
469                if(requiredPage >= blocklen) {
470                    enlarge((int) (blocklen * ENLARGE_PAGES_FACTOR));
471                }
472                long[] block = _block[requiredPage];
473                if(block == null) {
474                    block = new long[PAGE_LENGTH];
475                    _block[requiredPage] = block;
476                }
477                final int i = (int) (at & PAGE_MASK);
478                block[i] = value;
479            }

482            protected void fastSetData(final long at, final long value) {
483                this.setData(at, value);
484            }

486            // REVIEWME synchronized
487            private final long[] pageIn(final FixedSegments paged, final int requiredPage) {
488                final IntHash<long[]> cache = _pageReadCache;
489                long[] readBlock = cache.get(requiredPage);
490                if(readBlock != null) {
491                    return readBlock;
492                }
493                final byte[] v;
494                try {
495                    v = paged.read(requiredPage);
496                } catch (IOException e) {// should not come here
497                    throw new IllegalStateException("segment #" + requiredPage + " was not found.");
498                }
499                readBlock = Primitives.toLongs(v);
500                cache.put(requiredPage, readBlock);
501                return readBlock;
502            }
503    
504            private final long[] pageInSegments(final FixedSegments paged, final int requiredPage) {
505                long[] readBlock = getBlock(requiredPage);
506                if(readBlock != null) {
507                    return readBlock;
508                }
509                final long[][] blocksRef = _block;
510                final PagingProfile profile = _profile;
511                final int pageIdx = profile.getReadBackwards();
512                int fromPage = Math.max(requiredPage - pageIdx, 0);
513                int toPage = Math.min(requiredPage + profile.getReadForwards() + 1, blocksRef.length);
514                for(int i = fromPage; i < requiredPage; i++) {
515                    if(blocksRef[i] == null) {
516                        break;
517                    } else {
518                        fromPage++;
519                    }
520                }
521                for(int i = toPage - 1; i > requiredPage; i--) {
522                    if(blocksRef[i] == null) {
523                        break;
524                    } else {
525                        toPage--;
526                    }
527                }
528                final int numBlocks = toPage - fromPage;
529                profile.lastReadAccess(requiredPage, numBlocks);
530                if(LOG.isDebugEnabled()) {
531                    LOG.debug("page in " + numBlocks + " blocks [" + fromPage + ',' + toPage + ')');
532                }
533                final byte[] v;
534                if(numBlocks == prevBlocks) {
535                    v = tmpBlock;
536                    try {
537                        paged.bulkRead(fromPage, v);
538                    } catch (IOException e) {
539                        throw new IllegalStateException("failed reading " + numBlocks
540                                + " segments from #" + fromPage);
541                    }
542                } else {
543                    try {
544                        v = paged.bulkRead(fromPage, numBlocks);
545                    } catch (IOException e) {
546                        throw new IllegalStateException("failed reading " + numBlocks
547                                + " segments from #" + fromPage);
548                    }
549                    this.tmpBlock = v;
550                    this.prevBlocks = numBlocks;
551                }
552                for(int i = 0, off = 0; i < numBlocks; i++, off += PAGE_SIZE) {
553                    final long[] b = Primitives.toLongs(v, off, PAGE_SIZE);
554                    final int pi = fromPage + i;
555                    if(pi == requiredPage) {
556                        readBlock = b;
557                        putBlock(requiredPage, b);
558                    } else {
559                        putBlock(pi, b);
560                    }
561                }
562                if(readBlock == null) {
563                    throw new IllegalStateException("Page not found: " + requiredPage);
564                }
565                return readBlock;
566            }
567    
568            private long[] getBlock(final int page) {
569                final ILock lock = _lock;
570                lock.lock();
571                try {
572                    return _pageReadCache.get(page);
573                } finally {
574                    lock.unlock();
575                }
576            }
577    
578            private void putBlock(final int page, final long[] b) {
579                final ILock lock = _lock;
580                lock.lock();
581                try {
582                    _pageReadCache.putIfAbsent(page, b);
583                } finally {
584                    lock.unlock();
585                }
586            }

589            public void flush(final DbCollection coll, final String docName) throws IOException,
590                    DbException {
591                _dirtyBuffer.purgeAll();
592                super.flush(coll, docName);
593                _dirtyBuffer.purgeAll();
594            }

598        private static final class BufferWriter implements Cleaner<long[]> {
599    
600            private final FixedSegments paged;
601            private final IntHash<long[]> readCache;
602    
603            BufferWriter(final FixedSegments paged, final IntHash<long[]> readCache) {
604                super();
605                this.paged = paged;
606                this.readCache = readCache;
607            }
608    
609            public void cleanup(final long idx, final long[] value) {
610                final byte[] b = Primitives.toBytes(value);
611                try {
612                    paged.write(idx, b);
613                } catch (IOException e) {
614                    throw new IllegalStateException("failed to write #" + idx, e);
615                }
616                readCache.put((int) idx, value);
617            }
