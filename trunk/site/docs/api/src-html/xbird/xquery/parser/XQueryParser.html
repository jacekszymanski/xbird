267            assert (str != null) : str;
268            final int strlen = str.length();
269            final StringBuilder buf = new StringBuilder(strlen);
270            for(int i = 0; i < strlen; i++) {
271                char c = str.charAt(i);
272                if(c == '&' && (i + 1) < strlen) {
273                    final int end = str.indexOf(";", i + 1);
274                    if(end == -1) {
275                        buf.append(c);
276                    } else {
277                        String ref = str.substring(i, end + 1);
278                        String entity = parseCharRef(ref);
279                        buf.append(entity);
280                        i = end;
281                    }
282                } else {
283                    buf.append(c);
284                }
285            }
286            return buf.toString();

289        private static String parseCharRef(final String ref) {
290            assert (ref != null) : ref;
291            if(ref.startsWith("&#")) {
292                final int value;
293                if(ref.charAt(2) == 'x') {// hex digits
294                    String s = ref.substring(3, ref.length() - 1);
295                    value = Integer.parseInt(s, 16);
296                } else {
297                    String s = ref.substring(2, ref.length() - 1);
298                    value = Integer.parseInt(s);
299                }
300                return Character.toString((char) value);
301            } else if("&lt;".equals(ref)) {
302                return "<";
303            } else if("&gt;".equals(ref)) {
304                return ">";
305            } else if("&amp;".equals(ref)) {
306                return "&";
307            } else if("&quot;".equals(ref)) {
308                return "\"";
309            } else if("&apos;".equals(ref)) {
310                return "'";
311            } else {
312                throw new IllegalStateException("Illegal character: " + ref);
313            }
