256        public XQExpression visit(FLWRExpr expr, XQueryContext ctxt) throws XQueryException {
257            // #1 for loop
258            List<Binding> bindings = expr.getClauses();
259            int bingingLen = bindings.size();
260            for(int i = 0; i < bingingLen; i++) {
261                Binding b = bindings.get(i);
262                b.visit(this, ctxt);
263            }
264            // #2 return
265            XQExpression filteredRet = expr.getFilteredReturnExpr();
266            if(filteredRet != null) {
267                filteredRet.visit(this, ctxt);
268            } else {
269                // #2-1 where filtering
270                XQExpression where = expr.getWhereExpr();
271                if(where != null) {
272                    where.visit(this, ctxt);
273                }
274                // #2-2 return
275                XQExpression ret = expr.getReturnExpr();
276                assert (ret != null);
277                ret.visit(this, ctxt);
278            }
279            // #3 order by sorting
280            for(OrderSpec o : expr.getOrderSpecs()) {
281                o.visit(this, ctxt);
282            }
283            return expr;
284        }

420        public XQExpression visit(PromoteJoinExpression joinExpr, XQueryContext ctxt)
421                throws XQueryException {
422            final XQExpression srcExpr = joinExpr.getSrcExpr();
423            srcExpr.visit(this, ctxt);
424            final List<XQExpression> pkeyExprs = joinExpr.getPersistKeyExprs();
425            for(XQExpression pkeyExpr : pkeyExprs) {
426                pkeyExpr.visit(this, ctxt);
427            }
428            final List<XQExpression> skeyExprs = joinExpr.getSearchKeyExprs();
429            for(XQExpression skeyExpr : skeyExprs) {
430                skeyExpr.visit(this, ctxt);
431            }
432            final List<XQExpression> filterExprs = joinExpr.getFilterExprs();
433            for(XQExpression filter : filterExprs) {
434                filter.visit(this, ctxt);
435            }
436            return joinExpr;
437        }
