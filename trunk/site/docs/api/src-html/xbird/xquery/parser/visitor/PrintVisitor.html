107        public String getResult() {
108            // replace to simple expression.
109            final String toDel = "/descendant-or-self::node()/";
110            for(;;) {
111                int head = buf.indexOf(toDel);
112                if(head == -1) {
113                    break;
114                }
115                int rep_end = head + toDel.length();
116                buf.replace(head, rep_end, "//");
117            }
118            return buf.toString();
119        }

131        @Override
132        public XQExpression visit(AdditiveExpr op, XQueryContext ctxt) throws XQueryException {
133            op.getLeftOperand().visit(this, ctxt);
134            lineFeed();
135            if(op.isPlus()) {
136                buf.append('+');
137            } else {
138                buf.append('-');
139            }
140            lineFeed();
141            op.getRightOperand().visit(this, ctxt);
142            return op;
143        }

145        @Override
146        public XQExpression visit(AndExpr expr, XQueryContext ctxt) throws XQueryException {
147            List exprs = expr.getExpressions();
148            for(int i = 0; i < exprs.size(); i++) {
149                if(i != 0) {
150                    buf.append(" and");
151                    lineFeed();
152                }
153                XQExpression e = (XQExpression) exprs.get(i);
154                e.visit(this, ctxt);
155            }
156            return expr;
157        }

159        @Override
160        public XQExpression visit(AttributeConstructor constructor, XQueryContext ctxt)
161                throws XQueryException {
162            header("AttributeConstructor");
163            final QualifiedName name = constructor.getName();
164            if(name != null) {
165                attrFeed("name", name.toString());
166            } else {
167                attrFeed("name");
168                indentln();
169                XQExpression nameExpr = constructor.getNameExpr();
170                nameExpr.visit(this, ctxt);
171                indent--;
172            }
173    
174            attrFeed("value");
175            indent++;
176            final List values = constructor.getValueExprs();
177            for(int i = 0; i < values.size(); i++) {
178                XQExpression v = (XQExpression) values.get(i);
179                lineFeed();
180                v.visit(this, ctxt);
181            }
182            indent--;
183    
184            return constructor;
185        }

314        @Override
315        public XQExpression visit(ElementConstructor constructor, XQueryContext ctxt)
316                throws XQueryException {
317    
318            final String name = constructor.getElemName();
319            if(name != null) {
320                buf.append('<' + name + '>');
321            } else {
322                header("ElementConstructor");
323                attrFeed("name");
324                final XQExpression nameExpr = constructor.getNameExpr();
325                indentln();
326                nameExpr.visit(this, ctxt);
327                indent--;
328            }
329    
330            List atts = constructor.getAttributes();
331            final int attlen = atts.size();
332            if(attlen > 0) {
333                attrFeed("attributes");
334                indent++;
335                for(int i = 0; i < attlen; i++) {
336                    XQExpression a = (XQExpression) atts.get(i);
337                    lineFeed();
338                    a.visit(this, ctxt);
339                }
340                indent--;
341            }
342    
343            List values = constructor.getContents();
344            if(values.size() > 0) {
345                attrFeed("value");
346                indent++;
347                for(int i = 0; i < values.size(); i++) {
348                    XQExpression v = (XQExpression) values.get(i);
349                    if(!(v instanceof TextContent)) {
350                        lineFeed();
351                    }
352                    v.visit(this, ctxt);
353                }
354                indent--;
355            }
356    
357            if(name != null) {
358                lineFeed();
359                buf.append("</" + name + '>');
360            }
361    
362            return constructor;
363        }

422        @Override
423        public XQExpression visit(FLWRExpr expr, XQueryContext ctxt) throws XQueryException {
424            // for, let
425            List causes = expr.getClauses();
426            int prev_type = -1;
427            for(int i = 0; i < causes.size(); i++) {
428                Binding cause = (Binding) causes.get(i);
429                Variable var = cause.getVariable();
430                final int type = cause.getExpressionType();
431                if(i != 0) {
432                    if(prev_type == type) {
433                        buf.append(", ");
434                    }
435                    lineFeed();
436                }
437                switch(type) {
438                    case Binding.FOR_CLAUSE:
439                        if(prev_type != type) {
440                            buf.append("for $"); // fist appearence.
441                        } else {
442                            buf.append("    $");
443                        }
444                        String varname = var.getName();
445                        if(varname != null) {
446                            buf.append(varname);
447                            Variable posVar = ((ForClause) cause).getPositionVariable();
448                            if(posVar != null) {
449                                buf.append(" at $");
450                                buf.append(posVar.getName());
451                            }
452                        } else {
453                            assert (var instanceof BindingVariable || var instanceof PathVariable);
454                        }
455                        buf.append(" in ");
456                        //indentln();
457                        XQExpression inExpr = ((ForClause) cause).getInExpr();
458                        inExpr.visit(this, ctxt);
459                        //indent--;
460                        break;
461                    case Binding.LET_CLAUSE:
462                        if(prev_type != type) {
463                            buf.append("let $"); // fist appearence.
464                        } else {
465                            buf.append("    $");
466                        }
467                        buf.append(var.getName());
468                        buf.append(" := ");
469                        //indentln();
470                        var.getValue().visit(this, ctxt);
471                        //indent--;
472                        break;
473                    default:
474                        throw new IllegalStateException();
475                }
476                prev_type = type;
477            }
478    
479            XQExpression filteredRet = expr.getFilteredReturnExpr();
480    
481            // where
482            if(filteredRet == null) {
483                XQExpression filter = expr.getWhereExpr();
484                if(filter != null) {
485                    lineFeed();
486                    buf.append("where ");
487                    indentln();
488                    filter.visit(this, ctxt);
489                    indent--;
490                }
491            }
492    
493            // order by
494            List orders = expr.getOrderSpecs();
495            for(int i = 0; i < orders.size(); i++) {
496                if(i == 0) {
497                    lineFeed();
498                    if(expr.isStableOrdering()) {
499                        buf.append("stable ");
500                    }
501                    buf.append("order by");
502                } else {
503                    buf.append(',');
504                }
505                indentln();
506                OrderSpec spec = (OrderSpec) orders.get(i);
507                spec.getKeyExpr().visit(this, ctxt);
508                indent--;
509                if(spec.isDescending()) {
510                    buf.append(" descending");
511                }
512                if(spec.isEmptyGreatest()) {
513                    buf.append(" empty greatest");
514                }
515                URI collation = spec.getCollation();
516                if(collation != null) {
517                    buf.append(" collation ");
518                    buf.append(collation.toString());
519                }
520            }
521    
522            // return
523            lineFeed();
524            buf.append("return ");
525            indentln();
526            XQExpression ret = (filteredRet == null) ? expr.getReturnExpr() : filteredRet;
527            ret.visit(this, ctxt);
528            indent--;
529    
530            return expr;
531        }

533        @Override
534        public XQExpression visit(FunctionCall call, XQueryContext ctxt) throws XQueryException {
535            QualifiedName fname = call.getFuncName();
536            final String prefix = fname.getPrefix();
537            if(prefix != null && prefix.length() > 0) {
538                buf.append(prefix);
539                buf.append(':');
540            }
541            buf.append(fname.getLocalPart());
542            buf.append('(');
543            List params = call.getParams();
544            for(int i = 0; i < params.size(); i++) {
545                if(i != 0) {
546                    buf.append(", ");
547                }
548                XQExpression p = (XQExpression) params.get(i);
549                p.visit(this, ctxt);
550            }
551            buf.append(')');
552            return call;
553        }

684            final String content = constructor.getContent();

687                    attrFeed("target", target);
688                    attrFeed("content");

690                    constructor.getContentExpr().visit(this, ctxt);

692                } else {
693                    attrFeed(target, content);
694                }
695            } else {
696                attrFeed("target");
697                indentln();
698                constructor.getTargetExpr().visit(this, ctxt);
699                indent--;
700    
701                if(content == null) {
702                    attrFeed("content");
703                    indentln();
704                    constructor.getContentExpr().visit(this, ctxt);
705                    indent--;
706                } else {
707                    attrFeed("content", content);
708                }
709            }
710            return constructor;

727        @Override
728        public XQExpression visit(QuantifiedExpr expr, XQueryContext ctxt) throws XQueryException {
729            if(expr.isEvery()) {
730                buf.append("every ");
731            } else {
732                buf.append("some ");
733            }
734            BindingVariable binding = expr.getBinding();
735            assert (binding != null);
736            buf.append('$');
737            buf.append(binding.getName());
738    
739            buf.append(" in ");
740            indentln();
741            XQExpression inExpr = binding.getValue();
742            inExpr.visit(this, ctxt);
743            indent--;
744    
745            lineFeed();
746            buf.append("satisfies");
747            indentln();
748            expr.getCondExpr().visit(this, ctxt);
749            indent--;
750    
751            return expr;
752        }

758            indentln();

760                if(i != 0) {

763                }
764                XQExpression e = (XQExpression) exprs.get(i);
765                e.visit(this, ctxt);
766            }
767            indent--;
768            lineFeed();
769            buf.append(')');
770            return expr;

808        @Override
809        public XQExpression visit(TypeswitchExpr expr, XQueryContext ctxt) throws XQueryException {
810            buf.append("switch( ");
811            expr.getOperandExpr().visit(this, ctxt);
812            buf.append(")\n{");
813            indentln();
814            List cases = expr.getCaseClauses();
815            for(int i = 0; i < cases.size(); i++) {
816                if(i != 0) {
817                    lineFeed();
818                }
819                CaseClause c = (CaseClause) cases.get(i);
820                c.visit(this, ctxt);
821            }
822            lineFeed();
823            buf.append("default:");
824            XQExpression dc = expr.getDefaultClause();
825            if(dc != null) {
826                indentln();
827                dc.visit(this, ctxt);
828                indent--;
829            }
830            indent--;
831            lineFeed();
832            buf.append('}');
833            return expr;
834        }

846        @Override
847        public XQExpression visit(UserFunction function, XQueryContext ctxt) throws XQueryException {
848            buf.append("declare function ");
849            final QualifiedName funcName = function.getName();
850            final String prefix = funcName.getPrefix();
851            if(prefix != null && prefix.length() > 0) {
852                buf.append(prefix);
853                buf.append(':');
854            }
855            buf.append(funcName.getLocalPart());
856            buf.append('(');
857            List params = function.getParameters();
858            for(int i = 0; i < params.size(); i++) {
859                if(i != 0) {
860                    buf.append(", ");
861                }
862                VarRef p = (VarRef) params.get(i);
863                buf.append(p.toString());
864            }
865            buf.append(") ");
866            Type returnType = function.getReturnType();
867            if(returnType != null) {
868                buf.append("as ");
869                buf.append(returnType.toString());
870            }
871            buf.append(" {");
872            indentln();
873            function.getBodyExpression().visit(this, ctxt);
874            indent--;
875            lineFeed();
876            buf.append('}');
877    
878            return null;
879        }

896        @Override
897        public XQExpression visit(Variable variable, XQueryContext ctxt) throws XQueryException {
898            String name = variable.getName();
899            buf.append("$" + name);
900            Type type = variable.getType();
901            if(type != null) {
902                buf.append(" as ");
903                buf.append(type);
904            }
905            XQExpression value = variable.getValue();
906            buf.append(" := ");
907            if(value != null) { // may be null with funcall
908                value.visit(this, ctxt);
909            }
910            return variable;

913        @Override
914        public XQExpression visit(VarRef ref, XQueryContext ctxt) throws XQueryException {
915            Variable var = ref.getValue();
916            if(var instanceof PathVariable) {
917                visit((PathVariable) var, ctxt);
918            } else if(var instanceof ParametricVariable) {
919                XQExpression expr = ((ParametricVariable) var).getValue();
920                if(expr == null) {
921                    buf.append(var);
922                } else {
923                    visit(expr, ctxt);
924                }
925            } else if(var instanceof ThreadedVariable) {
926                ThreadedVariable threadVar = (ThreadedVariable) var;
927                buf.append(threadVar);
928                buf.append(" := ");
929                indentln();
930                XQExpression expr = threadVar.getValue();
931                assert (expr != null);
932                visit(expr, ctxt);
933                indent--;
934            } else if(var instanceof ShippedVariable) {
935                buf.append(var.getValue());
936            } else {
937                buf.append(var);
938            }
939            return ref;
940        }

956        @Override
957        public XQExpression visit(XQueryModule module, XQueryContext ctxt) throws XQueryException {
958            header("XQueryModule");
959    
960            attrFeed("version", module.getVersion());
961            attrFeed("encoding", module.getEncoding());
962            attrFeed("location", module.getLocation());
963            attrFeed("namespace", module.getNamespace());
964    
965            final Iterator<Variable> v_itor = module.getGlobalVariables().iterator();
966            if(v_itor.hasNext()) {
967                attrFeed("variable decls");
968                indent++;
969                do {
970                    lineFeed();
971                    Variable v = v_itor.next();
972                    v.visit(this, ctxt);
973                } while(v_itor.hasNext());
974                indent--;
975            }
976    
977            final Iterator<UserFunction> f_itor = module.getLocalFunctions().iterator();
978            if(f_itor.hasNext()) {
979                attrFeed("function decls");
980                indent++;
981                do {
982                    lineFeed();
983                    UserFunction f = f_itor.next();
984                    f.visit(this, ctxt);
985                } while(f_itor.hasNext());
986                indent--;
987            }
988    
989            final XQExpression body = module.getExpression();
990            if(body != null) {
991                attrFeed("expression body");
992                indentln();
993                body.visit(this, ctxt);
994                indent--;
995            }
996    
997            return null;
998        }

1000        private void attrFeed(String name) {
1001            lineFeed();
1002            buf.append(" - ").append(name).append(" : ");
1003        }
1004    
1005        private void attrFeed(String name, Object attr) {
1006            if(attr != null) {
1007                attrFeed(name);
1008                buf.append(attr.toString());
1009            }
1010        }
