021    package xbird.xquery.type.xs;
022    
023    import javax.xml.datatype.*;
024    
025    import xbird.xqj.XQJConstants;
026    import xbird.xquery.XQueryException;
027    import xbird.xquery.dm.value.AtomicValue;
028    import xbird.xquery.dm.value.xsi.GregorianDateTimeValue;
029    import xbird.xquery.meta.DynamicContext;
030    import xbird.xquery.misc.XsDatatypeFactory;
031    import xbird.xquery.type.*;
032    
033    /**
034     * 
035     * <DIV lang="en"></DIV>
036     * <DIV lang="ja"></DIV>
037     * 
038     * @author Makoto YUI (yuin405+xbird@gmail.com)
039     */
040    public final class GMonthDayType extends GregorianBaseType {
041        private static final long serialVersionUID = -3132418771499220432L;
042        public static final String SYMBOL = "xs:gMonthDay";
043    
044        public static final GMonthDayType GMONTHDAY = new GMonthDayType();
045    
046        public GMonthDayType() {
047            super(TypeTable.GMONTH_DAY_TID, SYMBOL);
048        }
049    
050        public AtomicValue createInstance(String literal, AtomicType srcType, DynamicContext dynEnv)
051                throws XQueryException {
052            return new GregorianDateTimeValue(literal, this);
053        }
054    
055        public GregorianDateTimeValue createInstance(XMLGregorianCalendar value) {
056            final DatatypeFactory factory = XsDatatypeFactory.getDatatypeFactory();
057            int month = value.getMonth();
058            int day = value.getDay();
059            int tz = value.getTimezone();
060            XMLGregorianCalendar cal = factory.newXMLGregorianCalendarDate(DatatypeConstants.FIELD_UNDEFINED, month, day, tz);
061            return new GregorianDateTimeValue(cal, GMONTHDAY);
062        }
063    
064        @Override
065        public boolean isYearSet() {
066            return false;
067        }
068    
069        @Override
070        protected boolean isSuperTypeOf(final AtomicType expected) {
071            return expected instanceof GMonthDayType;
072        }
073    
074        @Override
075        public int getXQJBaseType() {
076            return XQJConstants.XQBASETYPE_GMONTHDAY;
077        }
078    
079    }
