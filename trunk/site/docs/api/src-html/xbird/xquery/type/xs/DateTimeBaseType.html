021    package xbird.xquery.type.xs;
022    
023    import javax.xml.datatype.XMLGregorianCalendar;
024    
025    import xbird.xqj.XQJConstants;
026    import xbird.xquery.XQueryException;
027    import xbird.xquery.dm.value.AtomicValue;
028    import xbird.xquery.dm.value.xsi.DateTimeValue;
029    import xbird.xquery.meta.DynamicContext;
030    import xbird.xquery.misc.QNameTable;
031    import xbird.xquery.type.*;
032    
033    /**
034     * 
035     * <DIV lang="en"></DIV>
036     * <DIV lang="ja"></DIV>
037     * 
038     * @author Makoto YUI (yuin405+xbird@gmail.com)
039     * @link http://www.w3.org/TR/xquery-operators/#casting-to-datetimes
040     */
041    public abstract class DateTimeBaseType extends AtomicType {
042    
043        public static final DateTimeBaseType BASE_DATETIME = new DateTimeBaseType() {
044            private static final long serialVersionUID = -7138510290768398302L;
045    
046            public Class getJavaObjectType() {
047                return Object.class;
048            }
049    
050            public AtomicValue createInstance(String literal, AtomicType srcType, DynamicContext dynEnv)
051                    throws XQueryException {
052                throw new UnsupportedOperationException("DateTimeBaseType#createInstance is allowed");
053            }
054    
055            @Override
056            public DateTimeValue createInstance(XMLGregorianCalendar value) {
057                throw new UnsupportedOperationException("DateTimeBaseType#createInstance is allowed");
058            }
059    
060            @Override
061            public boolean accepts(Type expected) {
062                return expected instanceof DateTimeBaseType;
063            }
064    
065            @Override
066            public int getXQJBaseType() {
067                return XQJConstants.XQBASETYPE_ANYATOMICTYPE; // REVIEWME
068            }
069        };
070    
071        public DateTimeBaseType() {
072            super(TypeTable.DATETIME_BASE_TID, QNameTable.instantiate("", "dateTimeBase"), "dateTimeBase");
073        }
074    
075        protected DateTimeBaseType(final int typeId, final String type) {
076            super(typeId, type);
077        }
078    
079        public abstract DateTimeValue createInstance(XMLGregorianCalendar value);
080    
081        public boolean isDateSet() {
082            return isYearSet() || isMonthSet() || isDaySet();
083        }
084    
085        public boolean isYearSet() {
086            return true;
087        }
088    
089        public boolean isMonthSet() {
090            return true;
091        }
092    
093        public boolean isDaySet() {
094            return true;
095        }
096    
097        public boolean isTimeSet() {
098            return true;
099        }
100    
101        @Override
102        protected boolean isSuperTypeOf(final AtomicType expected) {
103            return expected instanceof DateTimeBaseType;
104        }
105    
106    }
