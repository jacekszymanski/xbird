021    package xbird.xquery.type.node;
022    
023    import xbird.xquery.dm.NodeKind;
024    import xbird.xquery.dm.labeling.RevPathCoder;
025    import xbird.xquery.misc.QNameTable.QualifiedName;
026    import xbird.xquery.optim.RewriteInfo;
027    import xbird.xquery.type.ItemType;
028    
029    /**
030     * 
031     * <DIV lang="en"></DIV>
032     * <DIV lang="ja"></DIV>
033     * 
034     * @author Makoto YUI (yuin405+xbird@gmail.com)
035     */
036    public class NodeType extends ItemType {
037        private static final long serialVersionUID = -50684513621274113L;
038    
039        public static final NodeType ANYNODE = new NodeType(NodeKind.ANY);
040        public static final NodeType TEXT = new NodeType(NodeKind.TEXT);
041        public static final NodeType COMMENT = new NodeType(NodeKind.COMMENT);
042        public static final NodeType ELEMENT = new NodeType(NodeKind.ELEMENT);
043        public static final NodeType ATTRIBUTE = new NodeType(NodeKind.ATTRIBUTE);
044    
045        private final byte nodeKind;
046    
047        protected NodeType(byte nodeKind) {
048            this.nodeKind = nodeKind;
049        }
050    
051        public byte getNodeKind() {
052            return nodeKind;
053        }
054    
055        public QualifiedName getNodeName() {
056            return null;
057        }
058    
059        public Class getJavaObjectType() {
060            return this.getClass();
061        }
062    
063        public boolean acceptNodeType(NodeType expected) {
064            final byte kind = expected.getNodeKind();
065            return acccept(kind);
066        }
067    
068        public boolean acccept(byte type) {
069            if(this.nodeKind == NodeKind.ANY) {
070                return true;
071            } else {
072                return this.nodeKind == type;
073            }
074        }
075    
076        @Override
077        public String toString() {
078            switch(nodeKind) {
079                case NodeKind.ANY:
080                    return "node()";
081                case NodeKind.TEXT:
082                    return "text()";
083                case NodeKind.COMMENT:
084                    return "comment()";
085                case NodeKind.ELEMENT:
086                case NodeKind.ATTRIBUTE:
087                    return "*";
088                default:
089                    throw new IllegalStateException("Illegal node kind was detected: "
090                            + NodeKind.resolveName(nodeKind));
091            }
092        }
093    
094        public int[] toQuery(RewriteInfo info, boolean isAttrStep) {
095            switch(nodeKind) {
096                case NodeKind.ANY:
097                    return isAttrStep ? new int[] { RevPathCoder.ATTRIBUTE, RevPathCoder.PERCENT }
098                            : new int[] { RevPathCoder.PERCENT };
099                case NodeKind.TEXT:
100                    return null; //"&T";
101                case NodeKind.COMMENT:
102                    return null; //"&C";
103                case NodeKind.ELEMENT:
104                    return isAttrStep ? null : new int[] { RevPathCoder.PERCENT };
105                case NodeKind.ATTRIBUTE:
106                    return isAttrStep ? new int[] { RevPathCoder.ATTRIBUTE, RevPathCoder.PERCENT }
107                            : null;
108                default:
109                    throw new IllegalStateException("Illegal node kind was detected: "
110                            + NodeKind.resolveName(nodeKind));
111            }
112        }
113    
114        public static String getNodeKindLiteral(NodeType nt) {
115            if(nt == null) {
116                return "*";
117            }
118            return NodeKind.resolveName(nt.getNodeKind());
119        }
120    
121        @Override
122        public boolean equals(Object obj) {
123            if(!(obj instanceof NodeType)) {
124                return false;
125            }
126            return this.nodeKind == ((NodeType) obj).nodeKind;
127        }
128    
129    }
