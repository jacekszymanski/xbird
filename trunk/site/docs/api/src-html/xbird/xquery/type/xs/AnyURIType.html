058    package xbird.xquery.type.xs;
059    
060    import java.net.URI;
061    
062    import xbird.xqj.XQJConstants;
063    import xbird.xquery.XQueryException;
064    import xbird.xquery.dm.value.xsi.AnyURIValue;
065    import xbird.xquery.meta.DynamicContext;
066    import xbird.xquery.type.*;
067    
068    /**
069     * 
070     * <DIV lang="en"></DIV>
071     * <DIV lang="ja"></DIV>
072     * 
073     * @author Makoto YUI (yuin405+xbird@gmail.com)
074     * @link http://www.w3.org/TR/xquery-operators/#casting-to-anyuri
075     * @link http://www.w3.org/TR/xmlschema-2/#built-in-datatypes
076     */
077    public final class AnyURIType extends AtomicType {
078        private static final long serialVersionUID = 4291451031828335983L;
079    
080        public static final String SYMBOL = "xs:anyURI";
081    
082        public static final AnyURIType ANYURI = new AnyURIType();
083    
084        public AnyURIType() {
085            super(TypeTable.ANY_URI_TID, SYMBOL);
086        }
087    
088        public Class getJavaObjectType() {
089            return URI.class;
090        }
091    
092        public AnyURIValue createInstance(String literal, AtomicType srcType, DynamicContext dynEnv)
093                throws XQueryException {
094            return AnyURIValue.valueOf(literal);
095        }
096    
097        @Override
098        protected boolean isSuperTypeOf(final AtomicType expected) {
099            return expected instanceof AnyURIType;
100        }
101    
102        @Override
103        public int getXQJBaseType() {
104            return XQJConstants.XQBASETYPE_ANYURI;
105        }
106    
107    }
