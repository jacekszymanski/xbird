068        private static final int BLOCK_SHIFT = 12;
069        private static final int DEFAULT_BLOCK_SIZE = 1 << BLOCK_SHIFT; // 2^12 = 4096 = 4KB
070        private static final long DEFAULT_BLOCK_SIZE_L = DEFAULT_BLOCK_SIZE;
071        private static final long BLOCK_MASK = DEFAULT_BLOCK_SIZE_L - 1L;

074        private static final int CHUNKED_THRESHOLD = DEFAULT_BLOCK_SIZE / 8; // 512 byte
075        private static final int DEFUALT_BIG_STRINGS_SIZE = (int) (DEFAULT_PAGES * ENLARGE_PAGES_FACTOR * 4);

111        public final long store(final char[] ch, final int start, final int length) {
112            final long raddr;
113            if(length >= CHUNKED_THRESHOLD) {
114                final byte[] b = Primitives.toBytes(ch, start, length);
115                raddr = storeStringChunk(b);
116            } else {
117                raddr = storeCharChunk(ch, start, length);
118            }
119            return raddr;
120        }

122        /**
123         * Store a String and return the stored address.
124         */
125        public final long store(final String s) {
126            final int strlen = s.length();
127            if(strlen < CHUNKED_THRESHOLD) {
128                s.getChars(0, strlen, tmpBuf, 0);
129                return storeCharChunk(tmpBuf, 0, strlen);
130            }
131            final byte[] b = StringUtils.getBytes(s);
132            return storeStringChunk(b);
133        }

135        protected final int storeStringChunk(final byte[] s) {
136            final byte[] b = compressor.compress(s);
137            return allocateStringChunk(b);
138        }

140        protected int allocateStringChunk(final byte[] s) {
141            final int index = _strPool.size();
142            _strPool.add(s);
143            return stringKey(index);
144        }

146        private static final int stringKey(final int index) {
147            assert (index >= 0) : index;
148            return (index << 1) + 1;
149        }
150    
151        protected final long storeCharChunk(final char[] ch, final int start, final int length) {
152            final int hcode = HashUtils.hashCode(ch, start, length);
153            final long haddr = _hashv2addr.get(hcode);
154            if(haddr != -1L) {
155                final char[] strInAddr = getChars(haddr);
156                assert (strInAddr != null);
157                if(ArrayUtils.equals(strInAddr, ch, start, length)) {
158                    return haddr;
159                } else {
160                    _hashv2addr.remove(hcode);
161                }
162            }
163            final long raddr = allocateCharChunk(ch, start, length);
164            _hashv2addr.put(hcode, raddr);
165            return raddr;
166        }
167    
168        protected long allocateCharChunk(final char[] src, final int start, final int length) {
169            assert (length < CHUNKED_THRESHOLD) : length;
170            final int reqlen = length + 1; // actual length is store in first char. 
171            final long lpage;
172            if(((_cpointer & BLOCK_MASK) + reqlen) > DEFAULT_BLOCK_SIZE_L) {
173                // spanning pages is not allowed, allocate in next chunk.
174                lpage = (_cpointer >> BLOCK_SHIFT) + 1;
175                _cpointer = lpage * DEFAULT_BLOCK_SIZE_L;
176            } else {
177                lpage = _cpointer >> BLOCK_SHIFT;
178            }
179            final int page = (int) lpage;
180            if(page >= _cchunks.length) {
181                enlarge((int) (_cchunks.length * ENLARGE_PAGES_FACTOR));
182            }
183            if(_cchunks[page] == null) {
184                _cchunks[page] = new char[DEFAULT_BLOCK_SIZE];
185            }
186            final long lblock = _cpointer & BLOCK_MASK;
187            final int block = (int) lblock;
188            assert (length <= Character.MAX_VALUE) : length;
189            _cchunks[page][block] = (char) length;
190            System.arraycopy(src, start, _cchunks[page], block + 1, length);
191            final long index = _cpointer;
192            _cpointer += reqlen; // move ahead pointer
193            return chunkKey(index);
194        }

196        private static final long chunkKey(final long index) {
197            return index << 1;
198        }

200        private static final long indexOf(final long addr) {
201            return addr >>> 1;
202        }

204        public void get(final long addr, final StringBuilder sb) {
205            final long ptr = indexOf(addr);
206            if((addr & BIG_STRING_MASK) != 0) { // is big string
207                assert (ptr <= 0x7fffffffL) : ptr;
208                final byte[] pooled = _strPool.get((int) ptr);
209                final char[] c = decompress(compressor, pooled);
210                sb.append(c);
211            } else {
212                assert (ptr <= _cpointer) : ptr;
213                final long lp = ptr >> BLOCK_SHIFT;
214                final int page = (int) lp;
215                if(lp != page) {
216                    throw new IllegalStateException("Illegal page number: " + lp);
217                }
218                final int block = (int) (ptr & BLOCK_MASK);
219                final char[] cc = _cchunks[page];
220                final int length = cc[block]; // the first char is the length of the string
221                sb.append(cc, block + 1, length);
222            }
223        }

227            final String ret;
228            if((addr & BIG_STRING_MASK) != 0) { // is big string

230                final byte[] b = _strPool.get((int) ptr);
231                final char[] c = decompress(compressor, b);
232                ret = new String(c);
233            } else {
234                assert (ptr <= _cpointer) : ptr;
235                final long lp = ptr >> BLOCK_SHIFT;
236                final int page = (int) lp;
237                if(lp != page) {
238                    throw new IllegalStateException("Illegal page number: " + lp);
239                }

241                final char[] cc = _cchunks[page];
242                final int length = cc[block]; // the first char is the length of the string
243                ret = new String(cc, block + 1, length);
244            }

248        protected char[] getChars(final long addr) {
249            final long ptr = indexOf(addr);
250            final char[] ch;
251            if((addr & BIG_STRING_MASK) != 0) { // is big string
252                assert (ptr <= 0x7fffffffL) : ptr;
253                final byte[] b = _strPool.get((int) ptr);
254                ch = decompress(compressor, b);
255            } else {
256                assert (ptr <= _cpointer) : ptr;
257                final long lp = ptr >> BLOCK_SHIFT;
258                final int page = (int) lp;
259                if(lp != page) {
260                    throw new IllegalStateException("Illegal page number: " + lp);
261                }
262                final int block = (int) (ptr & BLOCK_MASK);

264                final int length = cc[block]; // the first char is the length of the string
265                final int from = block + 1;
266                ch = ArrayUtils.copyOfRange(cc, from, from + length);
267            }
268            return ch;

293        public void flush(final DbCollection coll, final String docName, final PropertyMap docProps)
294                throws IOException {
295            final File chunkFile = getChunkFile(coll, docName);
296            assert (!chunkFile.exists()) : "file already exists: " + chunkFile.getAbsolutePath();
297            final int splen = _strPool.size();
298            final Segments paged = new VarSegments(chunkFile, DescriptorType.hash);
299            for(int i = 0; i < splen; i++) {// big string
300                final byte[] b = _strPool.get(i);
301                final int addr = stringKey(i);
302                paged.write(addr, b);
303            }
304            _strPool.clear();
305            final long lcclen = _cpointer >> BLOCK_SHIFT;
306            assert (lcclen <= Integer.MAX_VALUE) : lcclen;
307            final int cclen = Math.min((int) lcclen, _cchunks.length - 1);
308            for(int i = 0; i <= cclen; i++) {
309                final char[] c = _cchunks[i];
310                final byte[] b = compress(compressor, c);
311                final long addr = chunkKey(i * DEFAULT_BLOCK_SIZE_L);
312                paged.write(addr, b);
313                _cchunks[i] = null;
314            }
315            docProps.setProperty(KEY_STRPOOL_WRITTEN, String.valueOf(splen));
316            docProps.setProperty(KEY_CHUNK_WRITTEN, String.valueOf(cclen));
317            paged.flush(false);
318            close();
319            LOG.info("write string chunk file:" + chunkFile.getAbsolutePath());
320        }

322        private static final byte[] compress(final CompressionCodec compressor, final char[] c) {
323            final byte[] b = Primitives.toBytes(c);
324            return compressor.compress(b);
325        }
326    
327        private static final char[] decompress(final CompressionCodec compressor, final byte[] v) {
328            return compressor.decompressAsChars(v);
329        }

331        private static File getChunkFile(final DbCollection coll, final String docName) {
332            final File baseDir = new File(coll.getAbsolutePath());
333            assert (baseDir.exists() && baseDir.isDirectory());
334            final File chunkFile = new File(baseDir, docName + STRING_CHUNK_FILE_SUFFIX);
335            return chunkFile;
336        }

423            @Override
424            protected void enlarge(final int pages) {
425                ++enlarged;
426                if(enlarged >= paging_threshold) {
427                    try {
428                        pagedOut();
429                    } catch (IOException e) {
430                        throw new IllegalStateException("page out failed", e);
431                    }
432                    paging_threshold = Math.max(2, (int) (paging_threshold * 0.8));
433                    enlarged = 0;
434                }
435                super.enlarge(pages);
436            }

438            private void pagedOut() throws IOException {
439                // flush big strings
440                final int splen = _strPool.size();
441                for(int i = 0; i < splen; i++) {// big string
442                    final byte[] b = _strPool.get(i);
443                    final int addr = stringKey(spwritten + i);
444                    _paged.write(addr, b);
445                }
446                _strPool.clear();
447                this.spwritten += splen;
448                // flush char chunks
449                final long lcclen = _cpointer >> BLOCK_SHIFT;
450                assert (lcclen <= Integer.MAX_VALUE) : lcclen;
451                final int cclimit = (int) lcclen;
452                for(int i = lccwritten; i < cclimit; i++) {
453                    final char[] c = _cchunks[i];
454                    assert (c != null) : "cchunk[" + i + "] does not contain element";
455                    final byte[] b = compress(compressor, c);
456                    final long addr = chunkKey((ccwritten + i) * DEFAULT_BLOCK_SIZE_L);
457                    _paged.write(addr, b);
458                    _cchunks[i] = null; // prompt gc of heap object
459                }
460                if(LOG.isDebugEnabled()) {
461                    if(splen > 0) {
462                        LOG.debug("Paged out big string: " + spwritten + " - "
463                                + (spwritten + splen - 1));
464                    }
465                    if(cclimit > lccwritten) {
466                        LOG.debug("Paged out chunk chars: " + lccwritten + " - " + cclimit);
467                    }
468                }
469                this.lccwritten = cclimit;
470                _paged.flush(false);
471            }

473            private int outpages() {
474                final int splen = _strPool.size();
475                final long cpages = _cpointer >> BLOCK_SHIFT;
476                assert (cpages <= 0x7fffffffL) : cpages;
477                final int cclen = (int) cpages - lccwritten;

483                    throws IOException {
484                final int splen = _strPool.size();
485                for(int i = 0; i < splen; i++) {// big string
486                    final byte[] b = _strPool.get(i);
487                    final int addr = stringKey(spwritten + i);
488                    _paged.write(addr, b);
489                }
490                if(LOG.isDebugEnabled()) {
491                    if(splen > 0) {
492                        LOG.debug("Paged out big string: " + spwritten + " - "
493                                + (spwritten + splen - 1));
494                    }
495                }
496                _strPool.clear();
497                this.spwritten += splen;
498                final long lcclen = _cpointer >> BLOCK_SHIFT;
499                assert (lcclen <= Integer.MAX_VALUE) : lcclen;
500                final int cclimit = Math.min((int) lcclen, _cchunks.length - 1);

503                    if(c == null) {

508                    _paged.write(addr, b);

515                }

518                docProps.setProperty(KEY_CHUNK_WRITTEN, Integer.toString(ccwritten));

520                close();
521            }

541            private byte[] getStringInternal(final long addr, final long ptr) {
542                final int splen = _strPool.size();
543                final long spoffset = ptr - spwritten;
544                final byte[] b;
545                if((splen > 0) && (spoffset >= 0) && (spoffset <= splen)) {
546                    assert (spoffset <= 0x7fffffffL) : spoffset;
547                    b = _strPool.get((int) spoffset);
548                } else {
549                    assert (addr <= 0x7fffffffL) : addr;
550                    b = prepareBigString((int) addr);
551                    assert (b != null);
552                }
553                return b;
554            }

557            public String getString(final long addr) {
558                final long ptr = indexOf(addr);
559                final String ret;
560                if((addr & BIG_STRING_MASK) != 0) { // is big string
561                    final byte[] b = getStringInternal(addr, ptr);
562                    final char[] cc = decompress(compressor, b);
563                    ret = new String(cc);
564                } else {
565                    final char[] cc = getCharChunkInternal(ptr);
566                    final int block = (int) (ptr & BLOCK_MASK);
567                    final int length = cc[block]; // the first char is the length of the string
568                    ret = new String(cc, block + 1, length);
569                }

573            private char[] getCharChunkInternal(final long ptr) {
574                final long lp = ptr >> BLOCK_SHIFT;
575                if(lp > 0x7fffffffL) {
576                    throw new IllegalStateException("Illegal page number: " + lp);
577                }
578                final int requiredPage = (int) lp;
579                char[] cc;
580                if(requiredPage >= _cchunks.length) {
581                    cc = prepareCharChunk(requiredPage);
582                } else {
583                    cc = _cchunks[requiredPage];
584                    if(cc == null) {
585                        cc = prepareCharChunk(requiredPage);
586                    }
587                }
588                return cc;
589            }

602                    final int from = block + 1;

608            private byte[] prepareBigString(final int addr) {
609                assert ((addr & BIG_STRING_MASK) != 0L);
610                byte[] b = (byte[]) _cache.get(addr);
611                if(b == null) {// read from disk
612                    if(_tranfered) {
613                        b = readv(addr, false, _serContext);
614                    } else {
615                        b = read(addr);
616                    }
617                    assert (b != null) : "Illegal addr: " + addr;
618                    _cache.putIfAbsent(addr, b);
619                }
620                return b;
621            }

623            private char[] prepareCharChunk(final int page) {
624                final long addr = chunkKey(page * DEFAULT_BLOCK_SIZE_L);
625                char[] cc = (char[]) _cache.get(addr);
626                if(cc == null) {// read from disk
627                    final byte[] v;
628                    if(_tranfered) {
629                        v = readv(addr, true, _serContext);
630                    } else {
631                        v = read(addr);
632                    }
633                    assert (v != null) : "Illegal addr: " + addr + ", page: " + page + ", ccwritten:"
634                            + ccwritten + ", lccwritten:" + lccwritten;
635                    cc = decompress(compressor, v);
636                    _cache.putIfAbsent(addr, cc);
637                }
638                return cc;
639            }

649            private byte[] readv(final long addr, boolean charChunk, SerializationContext serContext) {
650                SortedSet<Long> tails = serContext.textBufferAddresses().tailSet(addr);
651                int nPages = Math.min(tails.size(), 16); // minimum 16 MB 
652                final long[] addrs;
653                if(nPages == 0) {
654                    addrs = new long[] { addr };
655                } else {
656                    addrs = new long[nPages];
657                    int n = 0;
658                    for(long l : tails) {
659                        addrs[n++] = l;
660                        if(n == nPages) {
661                            break;
662                        }
663                    }
664                }
665                final byte[][] tmp;
666                try {// read from disk
667                    tmp = _paged.readv(addrs);
668                } catch (IOException e) {
669                    throw new IllegalStateException("read from disk failed", e);
670                }
671                for(int i = 1; i < nPages; i++) {
672                    final long tmpAddr = addrs[i];
673                    if((tmpAddr & BIG_STRING_MASK) != 0L) {//is big string?
674                        _cache.putIfAbsent(tmpAddr, tmp[i]);
675                    } else {
676                        char[] cc = decompress(compressor, tmp[i]);
677                        _cache.putIfAbsent(tmpAddr, cc);
678                    }
679                }
680                return tmp[0];
681            }

683            @Override
684            public long getBufferAddress(long addr) {
685                final long ptr = indexOf(addr);
686                if((addr & BIG_STRING_MASK) != 0L) { // big string
687                    return ptr;
688                } else {
689                    long pageId = ptr >> BLOCK_SHIFT;
690                    return chunkKey(pageId * DEFAULT_BLOCK_SIZE_L);
691                }
692            }

701            private int page_out_threshold = 10000;
702    
703            @Override
704            protected long allocateCharChunk(final char[] src, final int start, final int length) {
705                if((length >= CHUNKED_THRESHOLD) || (length > Character.MAX_VALUE)) {
706                    throw new IllegalArgumentException("Illegal character size: " + length);
707                }
708                final int reqlen = length + 1; // actual length is store in first char. 
709                final int page;
710                if(((_cpointer & BLOCK_MASK) + reqlen) > DEFAULT_BLOCK_SIZE_L) {
711                    // spanning pages is not allowed, allocate in next chunk.
712                    final long lp = (_cpointer >> BLOCK_SHIFT) + 1L;
713                    assert (lp <= 0x7fffffffL) : "Assigned page size exceeds limit: " + lp;
714                    page = (int) lp;
715                    _cpointer = lp * DEFAULT_BLOCK_SIZE_L;
716                } else {
717                    final long lp = _cpointer >> BLOCK_SHIFT;
718                    assert (lp <= 0x7fffffffL) : "Assigned page size exceeds limit: " + lp;
719                    page = (int) lp;
720                }
721                assert (page >= 0) : "Illegal page '" + page + "' with cpointer '" + _cpointer + '\'';
722                if(page >= _cchunks.length) {
723                    enlarge((int) (_cchunks.length * ENLARGE_PAGES_FACTOR));
724                }
725                if(_cchunks[page] == null) {
726                    if(page < lccwritten) {
727                        _cchunks[page] = prepareCharChunk(page);
728                    } else {
729                        final int outs = outpages();
730                        if(outs > page_out_threshold) {
731                            try {
732                                pagedOut();
733                            } catch (IOException e) {
734                                throw new IllegalStateException("page out failed", e);
735                            }
736                            if(page_out_threshold > 5000) {
737                                page_out_threshold -= 100;
738                            }
739                        }
740                        _cchunks[page] = new char[DEFAULT_BLOCK_SIZE];
741                    }
742                }
743                final long lblock = _cpointer & BLOCK_MASK;
744                assert (lblock <= Integer.MAX_VALUE) : lblock;
745                final int block = (int) lblock;
746                _cchunks[page][block] = (char) length;
747                System.arraycopy(src, start, _cchunks[page], block + 1, length);
748                final long index = _cpointer;
749                _cpointer += reqlen; // move ahead pointer
750                return chunkKey((ccwritten * DEFAULT_BLOCK_SIZE_L) + index);
751            }
