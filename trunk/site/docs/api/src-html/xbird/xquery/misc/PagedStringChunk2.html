085        static {
086            ReferentFinalizer<String, Map<MutableString, Long>> finalizer = new ReferentFinalizer<String, Map<MutableString, Long>>() {
087                public void finalize(String fileName, Map<MutableString, Long> map) {
088                    try {
089                        makeCache(fileName, map);
090                    } catch (IOException e) {
091                        LOG.warn(e);
092                    }
093                }

095            _constructMapCache = new FinalizableSoftValueReferenceMap<String, Map<MutableString, Long>>(finalizer);
096            Runtime.getRuntime().addShutdownHook(new Thread() {
097                public void run() {
098                    for(Map.Entry<String, Map<MutableString, Long>> e : _constructMapCache.entrySet()) {
099                        try {
100                            makeCache(e.getKey(), e.getValue());
101                        } catch (IOException ioe) {
102                            LOG.debug(ioe);
103                        }
104                    }
105                }
106            });

145        public PagedStringChunk2() {

153        public PagedStringChunk2(VarSegments paged, PropertyMap properties) {
154            if(paged == null) {
155                throw new IllegalArgumentException();
156            }
157            this._tranfered = false;
158            this.paged_ = paged;
159    
160            String sp = properties.getProperty(KEY_STRPOOL_WRITTEN, "-1");
161            this.strBlockPtr_ = Integer.parseInt(sp);
162            String cp = properties.getProperty(KEY_CHUNK_POINTER, "0");
163            this.ccPointer_ = Long.parseLong(cp);
164            String pc = properties.getProperty(KEY_PENDING_CCPTR, "0");
165            this._pendingCharsPtr = Integer.parseInt(pc);
166    
167            if(_pendingCharsPtr > 0) {
168                long page = (ccPointer_ - 1) >> BLOCK_SHIFT;
169                byte[] b = read(page);
170                assert (b != null) : "ccPointer:" + ccPointer_ + ",  page:" + page;
171                byte[] decompressed = _compressor.decompress(b);
172                Primitives.getChars(decompressed, _pendingChars);
173            }
174    
175            this._constructionMap = reconstruct(paged);
176        }

178        private Map<MutableString, Long> reconstruct(VarSegments paged) {
179            String fileName = paged.getFile().getAbsolutePath() + CACHE_FILE_SUFFIX;
180            Map<MutableString, Long> map = _constructMapCache.get(fileName);
181            if(map != null) {
182                return map;
183            }
184            File cacheFile = new File(fileName);
185            if(!cacheFile.exists()) {
186                return new LRUMap<MutableString, Long>(Integer.getInteger("xbird.strchk.cmapsize", 32768));
187            }
188            final FileInputStream fis;
189            try {
190                fis = new FileInputStream(cacheFile);
191            } catch (FileNotFoundException e) {
192                throw new IllegalStateException(e);
193            }
194            final FastBufferedInputStream bis = new FastBufferedInputStream(fis);
195            Object obj = ObjectUtils.readObject(bis);
196            return (Map<MutableString, Long>) obj;
197        }

216        public void get(long addr, StringBuilder sb) {
217            if(addr < 0) {
218                char[] c = getStringInternal(addr);
219                sb.append(c);
220            } else {
221                char[] c = getCharChunkInternal(addr);
222                int block = (int) (addr & BLOCK_MASK);
223                int length = c[block]; // the first char is the length of the string
224                sb.append(c, block + 1, length);
225            }
226        }

228        public String getString(long addr) {
229            if(addr < 0) {
230                char[] c = getStringInternal(addr);
231                return new String(c);
232            } else {
233                char[] c = getCharChunkInternal(addr);
234                int block = (int) (addr & BLOCK_MASK);
235                int length = c[block]; // the first char is the length of the string
236                return new String(c, block + 1, length);
237            }
238        }

240        private char[] getStringInternal(final long addr) {
241            char[] c = _referenceMap.get(addr);
242            if(c != null) {
243                return c;
244            }
245            final byte[] b;
246            if(_tranfered) {
247                b = readv(addr, _serContext);
248            } else {
249                b = read(addr);
250            }
251            c = _compressor.decompressAsChars(b);
252            _referenceMap.put(addr, c);
253            return c;
254        }

256        private char[] getCharChunkInternal(final long addr) {
257            long page = addr >> BLOCK_SHIFT;
258            char[] c = _referenceMap.get(page);
259            if(c != null) {
260                return c;
261            }
262            final byte[] b;
263            if(_tranfered) {
264                b = readv(page, _serContext);
265            } else {
266                b = read(page);
267            }
268            c = _compressor.decompressAsChars(b);
269            _referenceMap.put(page, c);
270            return c;
271        }

273        private byte[] read(final long addr) {
274            try {// read from disk
275                return paged_.read(addr);
276            } catch (IOException e) {
277                throw new IllegalStateException("read from disk failed", e);
278            }
279        }

281        private byte[] readv(final long addr, final SerializationContext serContext) {
282            SortedSet<Long> tails = serContext.textBufferAddresses().tailSet(addr);
283            int nPages = Math.min(tails.size(), 16); // minimum 16 MB 
284            final long[] addrs;
285            if(nPages == 0) {
286                addrs = new long[] { addr };
287            } else {
288                addrs = new long[nPages];
289                int n = 0;
290                for(long l : tails) {
291                    addrs[n++] = l;
292                    if(n == nPages) {
293                        break;
294                    }
295                }
296            }
297            final byte[][] tmp;
298            try {// read from disk
299                tmp = paged_.readv(addrs);
300            } catch (IOException e) {
301                throw new IllegalStateException("read from disk failed", e);
302            }
303            for(int i = 1; i < nPages; i++) {
304                char[] cc = _compressor.decompressAsChars(tmp[i]);
305                long tmpAddr = addrs[i];
306                _referenceMap.put(tmpAddr, cc);
307            }
308            return tmp[0];
309        }

311        public synchronized long store(char[] ch, int start, int length) {
312            if(length < CHUNKED_THRESHOLD) {
313                return storeCharChunk(ch, start, length);
314            } else {
315                byte[] b = Primitives.toBytes(ch, start, length);
316                return storeStringChunk(b);
317            }
318        }

320        public synchronized long store(String s) {
321            final int strlen = s.length();
322            if(strlen < CHUNKED_THRESHOLD) {
323                s.getChars(0, strlen, _tmpBuf, 0);
324                return storeCharChunk(_tmpBuf, 0, strlen);
325            } else {
326                byte[] b = StringUtils.getBytes(s);
327                return storeStringChunk(b);
328            }
329        }
330    
331        private int storeStringChunk(final byte[] s) {
332            if(_pendingFlushPages.size() >= FLUSH_THRESHOLD) {
333                flushPendingPages();
334            }
335            final byte[] b = _compressor.compress(s);
336            _pendingFlushPages.add(new Long(strBlockPtr_), b);
337            return strBlockPtr_--;
338        }
339    
340        private long storeCharChunk(final char[] ch, final int start, final int length) {
341            MutableString probe = new MutableString(ch, start, length);
342            Long addr = _constructionMap.get(probe);
343            if(addr != null) {
344                return addr.longValue();
345            }
346    
347            int deltaToPlus = length + 1;
348            int nextPtrProbe = _pendingCharsPtr + deltaToPlus;
349            if(nextPtrProbe >= BLOCK_SIZE) {
350                if(_pendingFlushPages.size() >= FLUSH_THRESHOLD) {
351                    flushPendingPages();
352                }
353                final byte[] b = compress(_compressor, _pendingChars);
354                long page = (ccPointer_ - 1) >> BLOCK_SHIFT;
355                _pendingFlushPages.add(page, b);
356                ccPointer_ = (page + 1) << BLOCK_SHIFT; // set counter to next page
357                _pendingCharsPtr = 0;
358            }
359    
360            _pendingChars[_pendingCharsPtr] = (char) length;
361            System.arraycopy(ch, start, _pendingChars, _pendingCharsPtr + 1, length);
362            _pendingCharsPtr += deltaToPlus;
363    
364            long curPointer = ccPointer_;
365            _constructionMap.put(probe, curPointer);
366            ccPointer_ += deltaToPlus;
367            return curPointer;
368        }
369    
370        private static byte[] compress(final CompressionCodec compressor, final char[] c) {
371            final byte[] b = Primitives.toBytes(c);
372            return compressor.compress(b);
373        }
374    
375        private void flushPendingPages() {
376            final PairList<Long, byte[]> pendings = _pendingFlushPages;
377            final int size = pendings.size();
378            for(int i = 0; i < size; i++) {
379                try {
380                    paged_.write(pendings.getKey(i), pendings.getValue(i));
381                } catch (IOException ioe) {
382                    throw new IllegalStateException(ioe);
383                }
384            }
385            pendings.clear();
386        }

407            RemoteVarSegments paged = RemoteVarSegments.read(in);
408            PrivilegedAccessor.unsafeSetField(this, _pagedField, paged);

411        public void writeExternal(ObjectOutput out) throws IOException {
412            if(paged_ instanceof RemoteVarSegments) {
413                ((RemoteVarSegments) paged_).writeExternal(out);
414            } else {
415                File dataFile = paged_.getFile();
416                String filePath = dataFile.getAbsolutePath();
417                RemoteVarSegments remoteSeg = new RemoteVarSegments(RemotePagingService.PORT, filePath, false);
418                remoteSeg.writeExternal(out);
419            }
420        }

422        public synchronized void flush(DbCollection coll, String docName, PropertyMap properties)
423                throws IOException {
424            if(_pendingCharsPtr > 0) {
425                final byte[] b = compress(_compressor, _pendingChars);
426                long page = (ccPointer_ - 1) >> BLOCK_SHIFT;
427                _pendingFlushPages.add(page, b);
428                assert (_pendingCharsPtr < BLOCK_SIZE) : _pendingCharsPtr;
429            }
430            if(!_pendingFlushPages.isEmpty()) {
431                flushPendingPages();
432            }
433            properties.setProperty(KEY_STRPOOL_WRITTEN, Integer.toString(strBlockPtr_));
434            properties.setProperty(KEY_CHUNK_POINTER, Long.toString(ccPointer_));
435            properties.setProperty(KEY_PENDING_CCPTR, Integer.toString(_pendingCharsPtr));
436    
437            paged_.flush(false);
438            close();
439        }

441        public void close() throws IOException {
442            if(_refcount.decrementAndGet() == 0) {
443                String fileName = paged_.getFile().getAbsolutePath() + CACHE_FILE_SUFFIX;
444                _constructMapCache.put(fileName, _constructionMap);
445                forseClose();
446            }
447        }
448    
449        private void forseClose() throws IOException {
450            this._constructionMap = null;
451            this._referenceMap = null;
452            if(paged_ != null) {
453                paged_.close();
454                PrivilegedAccessor.unsafeSetField(this, _pagedField, null);
455            }
456        }

458        private static void makeCache(final String fileName, final Map<MutableString, Long> map)
459                throws IOException {
460            File cacheFile = new File(fileName);
461            FileOutputStream fos = new FileOutputStream(cacheFile, false);
462            FastBufferedOutputStream bos = new FastBufferedOutputStream(fos);
463            ObjectUtils.toStream(map, bos);
464            bos.flush();
465            bos.close();
466        }
