021    package xbird.xquery.func.context;
022    
023    import xbird.xquery.DynamicError;
024    import xbird.xquery.XQueryException;
025    import xbird.xquery.dm.value.Item;
026    import xbird.xquery.dm.value.Sequence;
027    import xbird.xquery.dm.value.literal.XInteger;
028    import xbird.xquery.dm.value.sequence.ValueSequence;
029    import xbird.xquery.func.BuiltInFunction;
030    import xbird.xquery.func.FunctionSignature;
031    import xbird.xquery.meta.DynamicContext;
032    import xbird.xquery.meta.IFocus;
033    import xbird.xquery.type.xs.IntegerType;
034    
035    /**
036     * fn:last() as xs:integer.
037     * <DIV lang="en"></DIV>
038     * <DIV lang="ja"></DIV>
039     * 
040     * @author Makoto YUI (yuin405+xbird@gmail.com)
041     * @link http://www.w3.org/TR/xquery-operators/#func-last
042     */
043    public final class Last extends BuiltInFunction {
044        private static final long serialVersionUID = -3638592334567227069L;
045        public static final String SYMBOL = "fn:last";
046    
047        public Last() {
048            super(SYMBOL, IntegerType.INTEGER);
049            this._evalPocily = EvaluationPolicy.dynamic;
050        }
051    
052        protected FunctionSignature[] signatures() {
053            final FunctionSignature[] s = new FunctionSignature[1];
054            s[0] = new FunctionSignature(getName());
055            return s;
056        }
057    
058        public Sequence eval(Sequence<? extends Item> contextSeq, ValueSequence argv, DynamicContext dynEnv)
059                throws XQueryException {
060            final IFocus focus = dynEnv.getFocus();
061            if(focus == null) {
062                throw new DynamicError("err:XPDY0002", "Focus of fileter is not set");
063            }
064            int last = focus.getLast();
065            if(last != -1) {
066                return new XInteger(last);
067            }
068            final Sequence<? extends Item> src = dynEnv.peekSequence();
069            if(src == null) {
070                throw new DynamicError("err:XPDY0002", "Source of fileter is not set");
071            }
072            last = 0;
073            IFocus<? extends Item> srcItor = src.iterator();
074            for(Item it : srcItor) {
075                ++last;
076            }
077            srcItor.closeQuietly();
078            focus.setLast(last);
079            return new XInteger(last);
080        }
081    
082    }
