021    package xbird.xquery.func.seq;
022    
023    import xbird.xquery.XQueryException;
024    import xbird.xquery.dm.value.Item;
025    import xbird.xquery.dm.value.Sequence;
026    import xbird.xquery.dm.value.sequence.ValueSequence;
027    import xbird.xquery.dm.value.xsi.BooleanValue;
028    import xbird.xquery.func.BuiltInFunction;
029    import xbird.xquery.func.FunctionSignature;
030    import xbird.xquery.meta.DynamicContext;
031    import xbird.xquery.type.SequenceType;
032    import xbird.xquery.type.Type;
033    import xbird.xquery.type.xs.BooleanType;
034    
035    /**
036     * fn:empty($arg as item()*) as xs:boolean.
037     * <DIV lang="en"></DIV>
038     * <DIV lang="ja"></DIV>
039     * 
040     * @author Makoto YUI (yuin405+xbird@gmail.com)
041     * @link http://www.w3.org/TR/xquery-operators/#func-empty
042     */
043    public final class Empty extends BuiltInFunction {
044        private static final long serialVersionUID = -9106469800991353019L;
045    
046        public static final String SYMBOL = "fn:empty";
047    
048        public Empty() {
049            super(SYMBOL, BooleanType.BOOLEAN);
050        }
051    
052        protected FunctionSignature[] signatures() {
053            final FunctionSignature[] s = new FunctionSignature[1];
054            s[0] = new FunctionSignature(getName(), new Type[] { SequenceType.ANY_ITEMS });
055            return s;
056        }
057    
058        public BooleanValue eval(final Sequence<? extends Item> contextSeq, final ValueSequence argv, final DynamicContext dynEnv)
059                throws XQueryException {
060            assert (argv != null && argv.size() == 1);
061            final Item arg = argv.getItem(0);
062            return arg.isEmpty() ? BooleanValue.TRUE : BooleanValue.FALSE;
063        }
064    
065    }
