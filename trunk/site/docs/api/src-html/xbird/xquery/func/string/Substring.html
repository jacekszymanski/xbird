021    package xbird.xquery.func.string;
022    
023    import xbird.xquery.XQueryException;
024    import xbird.xquery.dm.value.Item;
025    import xbird.xquery.dm.value.Sequence;
026    import xbird.xquery.dm.value.literal.XNumber;
027    import xbird.xquery.dm.value.literal.XString;
028    import xbird.xquery.dm.value.sequence.ValueSequence;
029    import xbird.xquery.func.BuiltInFunction;
030    import xbird.xquery.func.FunctionSignature;
031    import xbird.xquery.meta.DynamicContext;
032    import xbird.xquery.type.Type;
033    import xbird.xquery.type.TypeRegistry;
034    import xbird.xquery.type.xs.*;
035    
036    /**
037     * fn:substring.
038     * <DIV lang="en">
039     * Returns the portion of the value of $sourceString beginning at the position 
040     * indicated by the value of $startingLoc and continuing for the number of characters 
041     * indicated by the value of $length.<br/>
042     * Note that the first character of a string is located at position 1, not position 0.
043     * <ul>
044     * <li>fn:substring($sourceString as xs:string?, $startingLoc as xs:double) as xs:string</li>
045     * <li>fn:substring($sourceString as xs:string?, $startingLoc as xs:double, $length as xs:double) as xs:string</li>
046     * </ul>
047     * </DIV>
048     * <DIV lang="ja"></DIV>
049     * 
050     * @author Makoto YUI (yuin405+xbird@gmail.com)
051     * @link http://www.w3.org/TR/xquery-operators/#func-substring
052     */
053    public final class Substring extends BuiltInFunction {
054        private static final long serialVersionUID = -7070216833028372984L;
055        public static final String SYMBOL = "fn:substring";
056    
057        public Substring() {
058            super(SYMBOL, StringType.STRING);
059        }
060    
061        protected FunctionSignature[] signatures() {
062            final FunctionSignature[] s = new FunctionSignature[4];
063            s[0] = new FunctionSignature(getName(), new Type[] { TypeRegistry.safeGet("xs:string?"),
064                    DoubleType.DOUBLE });
065            s[1] = new FunctionSignature(getName(), new Type[] { TypeRegistry.safeGet("xs:string?"),
066                    DoubleType.DOUBLE, DoubleType.DOUBLE });
067            // workaround
068            s[2] = new FunctionSignature(getName(), new Type[] { TypeRegistry.safeGet("xs:string?"),
069                    NumericType.getInstance() });
070            s[3] = new FunctionSignature(getName(), new Type[] { TypeRegistry.safeGet("xs:string?"),
071                    NumericType.getInstance(), NumericType.getInstance() });
072            return s;
073        }
074    
075        public XString eval(Sequence<? extends Item> contextSeq, ValueSequence argv, DynamicContext dynEnv)
076                throws XQueryException {
077            final Item firstItem = argv.getItem(0);
078            if(firstItem.isEmpty()) {
079                // If the value of $sourceString is the empty sequence, the zero-length string is returned.
080                return XString.valueOf("");
081            }
082            final String sourceString = firstItem.stringValue();
083            final int maxSourceLen = sourceString.length();
084            final Item secItem = argv.getItem(1);
085            final XNumber secNum = (XNumber) secItem;
086            // first character of a string is located at position 1, not position 0.
087            final int startingLoc = Math.max(0, (int) Math.round(secNum.asDouble() - 1));
088            final int endingLoc;
089            final int arglen = argv.size();
090            if(arglen == 3) {
091                final Item thirdItem = argv.getItem(2);
092                final XNumber thirdNum = (XNumber) thirdItem;
093                final int length = (int) Math.round(thirdNum.asDouble());
094                endingLoc = Math.max(0, Math.min(maxSourceLen, length));
095            } else {
096                endingLoc = maxSourceLen;
097            }
098            if(endingLoc <= startingLoc) {
099                return XString.valueOf("");
100            }
101            final String ret = sourceString.substring(startingLoc, endingLoc);
102            return XString.valueOf(ret);
103        }
104    
105    }
