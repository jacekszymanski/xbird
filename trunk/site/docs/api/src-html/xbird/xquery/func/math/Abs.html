021    package xbird.xquery.func.math;
022    
023    import xbird.xquery.dm.value.literal.XNumber;
024    import xbird.xquery.type.TypeRegistry;
025    
026    /**
027     * fn:abs($arg as numeric?) as numeric?.
028     * <DIV lang="en">
029     * Returns the absolute value of $arg. If $arg is negative returns -$arg otherwise returns $arg.
030     * If type of $arg is one of the four numeric types xs:float, xs:double, xs:decimal or xs:integer 
031     * the type of the result is the same as the type of $arg.
032     * </DIV>
033     * <DIV lang="ja"></DIV>
034     * 
035     * @author Makoto YUI (yuin405+xbird@gmail.com)
036     * @link http://www.w3.org/TR/xquery-operators/#func-abs
037     * @link http://www.w3.org/TR/xquery-semantics/#sec_fn_abs_ceil_floor_round
038     */
039    public final class Abs extends NumericFunction {
040        private static final long serialVersionUID = 7360186556404994838L;
041        public static final String SYMBOL = "fn:abs";
042    
043        public Abs() {
044            super(SYMBOL, TypeRegistry.safeGet("xs:double?"));
045        }
046    
047        protected XNumber promote(XNumber value) {
048            assert (value != null);
049            final double v = value.getNumber().doubleValue();
050            if (v < 0 || v == Double.NEGATIVE_INFINITY) {// TODO negative zero
051                return value.negate();
052            } else {
053                return value;
054            }
055        }
056    
057    }
