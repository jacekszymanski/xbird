021    package xbird.xquery.func.constructor;

023    import javax.xml.datatype.DatatypeConstants;
024    import javax.xml.datatype.XMLGregorianCalendar;
025    
026    import xbird.xquery.DynamicError;
027    import xbird.xquery.XQueryException;
028    import xbird.xquery.dm.value.Item;
029    import xbird.xquery.dm.value.Sequence;

032    import xbird.xquery.func.BuiltInFunction;
033    import xbird.xquery.func.FunctionSignature;
034    import xbird.xquery.meta.DynamicContext;
035    import xbird.xquery.type.Type;
036    import xbird.xquery.type.xs.*;
037    
038    /**
039     * fn:dateTime($arg1 as xs:date, $arg2 as xs:time) as xs:dateTime.
040     * <DIV lang="en">
041     * A special constructor function is provided for constructing a xs:dateTime 
042     * value from a xs:date value and a xs:time value.
043     * </DIV>
044     * <DIV lang="ja"></DIV>
045     * 
046     * @author Makoto YUI (yuin405+xbird@gmail.com)
047     * @link http://www.w3.org/TR/xquery-operators/#func-dateTime
048     */
049    public final class DateTime extends BuiltInFunction {
050        private static final long serialVersionUID = -8630350815622049802L;
051    
052        public static final String SYMBOL = "fn:dateTime";
053    
054        public DateTime() {
055            super(SYMBOL, DateTimeType.DATETIME);
056        }
057    
058        protected FunctionSignature[] signatures() {
059            final FunctionSignature[] s = new FunctionSignature[1];
060            s[0] = new FunctionSignature(getName(), new Type[] { DateType.DATE, TimeType.TIME });
061            return s;
062        }
063    
064        public Sequence eval(Sequence<? extends Item> contextSeq, ValueSequence argv, DynamicContext dynEnv)
065                throws XQueryException {
066            Item arg1 = argv.getItem(0);
067            Item arg2 = argv.getItem(1);
068            DateTimeValue date = (DateTimeValue) arg1;
069            DateTimeValue time = (DateTimeValue) arg2;
070            XMLGregorianCalendar dateValue = date.getValue();
071            XMLGregorianCalendar newValue = (XMLGregorianCalendar) dateValue.clone();
072            XMLGregorianCalendar timeValue = time.getValue();
073            final int timeTz = timeValue.getTimezone();
074            if(timeTz != DatatypeConstants.FIELD_UNDEFINED) {
075                final int dateTz = newValue.getTimezone();
076                if(dateTz != DatatypeConstants.FIELD_UNDEFINED && dateTz != timeTz) {
077                    throw new DynamicError("err:FORG0008", "fn:dateTime(" + date + ", " + time
078                            + ") for `" + dateValue.getTimeZone(dateTz).getID() + " and `"
079                            + timeValue.getTimeZone(timeTz).getID() + "` is not allowed.");
080                }
081                newValue.setTimezone(timeTz);
082            }
083            newValue.setTime(timeValue.getHour(), timeValue.getMinute(), timeValue.getSecond(), timeValue.getMillisecond());
084            DateTimeValue dt = new DateTimeValue(newValue, DateTimeType.DATETIME);
085            return dt;
086        }
087    
088    }
