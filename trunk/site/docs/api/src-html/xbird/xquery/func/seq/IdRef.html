041    /**
042     * fn:idref.
043     * <DIV lang="en">
044     * An element or attribute typically acquires the is-idrefs property by being validated
045     * against the schema type xs:IDREF or xs:IDREFS, or (for attributes only) by being described 
046     * as of type IDREF or IDREFS in a DTD.
047     * <ul>
048     * <li>fn:idref($arg as xs:string*) as node()*</li>
049     * <li>fn:idref($arg as xs:string*, $node as node()) as node()*</li>
050     * </ul>
051     * </DIV>
052     * <DIV lang="ja"></DIV>
053     * 
054     * @author Makoto YUI (yuin405+xbird@gmail.com)
055     * @link http://www.w3.org/TR/xquery-operators/#func-idref
056     */
057    public final class IdRef extends BuiltInFunction {
058        private static final long serialVersionUID = 9211859623622428079L;
059        public static final String SYMBOL = "fn:idref";
060        
061        public IdRef() {
062            super(SYMBOL, TypeRegistry.safeGet("node()*"));
063        }
064    
065        protected FunctionSignature[] signatures() {
066            final FunctionSignature[] s = new FunctionSignature[2];
067            Type stra = TypeRegistry.safeGet("xs:string*");
068            s[0] = new FunctionSignature(getName(), new Type[] { stra });
069            s[1] = new FunctionSignature(getName(), new Type[] { stra,
070                    NodeType.ANYNODE });
071            return s;
072        }
073    
074        public Sequence eval(Sequence<? extends Item> contextSeq, ValueSequence argv, DynamicContext dynEnv) throws XQueryException {
075            assert (argv != null);
076            final int arglen = argv.size();
077            assert (arglen == 1 || arglen == 2);
078            final XQNode node;
079            if (arglen == 2) {
080                Item second = argv.getItem(1);
081                if (!(second instanceof XQNode)) {
082                    throw new DynamicError("err:XPTY0004", "Second argument is expected to be a node, but was "
083                            + second.getType());
084                }
085                node = (XQNode) second;
086            } else {
087                Item contextItem = dynEnv.contextItem();
088                if (contextItem == null) {
089                    throw new DynamicError("err:XPDY0002", "ContextItem is not set");
090                } else if (!(contextItem instanceof XQNode)) {
091                    throw new DynamicError("err:XPTY0004", "Context item is expected to be a node, but was "
092                            + contextItem.getType());
093                }
094                node = (XQNode) contextItem;
095            }
096            // An error is raised if node's root is not a document node
097            XQNode root = node.getRoot();
098            if (root.nodeKind() != NodeKind.DOCUMENT) {
099                throw new DynamicError("err:FODC0001", "root node was not document-node.");
100            }
101            Item firstItem = argv.getItem(0);
102            NodeSequence<XQNode> nodes = new NodeSequence<XQNode>(dynEnv);
103            for (Item it : firstItem) {
104                assert (it instanceof XString);
105                String ids = it.stringValue();
106                final StringTokenizer tok = new StringTokenizer(ids, " ");
107                while (tok.hasMoreTokens()) {
108                    String id = tok.nextToken();
109                    if (!IDType.ID.isValid(id)) {
110                        // If any of the tokens is not a lexically valid IDREF, it is ignored. 
111                        continue;
112                    }
113                    List<XQNode> founds = Id.getElementsById(root, id, true);
114                    for (XQNode n : founds) {
115                        nodes.addItem(n);
116                    }
117                }
118            }
119            return nodes;
120        }
121    }
