021    package xbird.xquery.func.qname;
022    
023    import java.util.List;
024    
025    import xbird.xquery.XQueryException;
026    import xbird.xquery.dm.value.Item;
027    import xbird.xquery.dm.value.Sequence;
028    import xbird.xquery.dm.value.literal.XString;
029    import xbird.xquery.dm.value.sequence.ValueSequence;
030    import xbird.xquery.dm.value.xsi.QNameValue;
031    import xbird.xquery.expr.XQExpression;
032    import xbird.xquery.func.*;
033    import xbird.xquery.meta.DynamicContext;
034    import xbird.xquery.meta.StaticContext;
035    import xbird.xquery.misc.QNameTable.QualifiedName;
036    import xbird.xquery.type.Type;
037    import xbird.xquery.type.TypeRegistry;
038    import xbird.xquery.type.xs.NCNameType;
039    
040    /**
041     * fn:prefix-from-QName($arg as xs:QName?) as xs:NCName?
042     * <DIV lang="en"></DIV>
043     * <DIV lang="ja"></DIV>
044     * 
045     * @author Makoto YUI (yuin405+xbird@gmail.com)
046     * @link http://www.w3.org/TR/xquery-operators/#func-prefix-from-QName
047     */
048    public final class PrefixFromQName extends BuiltInFunction {
049        private static final long serialVersionUID = 4809594696986732057L;
050        public static final String SYMBOL = "fn:prefix-from-QName";
051    
052        public PrefixFromQName() {
053            super(SYMBOL, TypeRegistry.safeGet("xs:NCName?"));
054        }
055    
056        protected FunctionSignature[] signatures() {
057            final FunctionSignature[] s = new FunctionSignature[1];
058            s[0] = new FunctionSignature(getName(), new Type[] { TypeRegistry.safeGet("xs:QName?") });
059            return s;
060        }
061    
062        @Override
063        public PrefixFromQName staticAnalysis(StaticContext context, List<? extends XQExpression> params) throws XQueryException {
064            strictCheckArguments(context, params);
065            return this;
066        }
067    
068        public Sequence eval(Sequence<? extends Item> contextSeq, ValueSequence argv, DynamicContext dynEnv)
069                throws XQueryException {
070            assert (argv != null && argv.size() == 1);
071            Item firstItem = argv.getItem(0);
072            if(firstItem.isEmpty()) {
073                return ValueSequence.EMPTY_SEQUENCE;
074            }
075            QNameValue arg = (QNameValue) firstItem;
076            QualifiedName qname = arg.getValue();
077            String prefix = qname.getPrefix();
078            if(prefix.length() == 0) {
079                return ValueSequence.EMPTY_SEQUENCE;
080            }
081            return new XString(prefix, NCNameType.NCNAME);
082        }
083    
084    }
