161        public XQExpression staticAnalysis(StaticContext statEnv) throws XQueryException {
162            if(!_analyzed) {
163                this._analyzed = true;
164    
165                int loopDepth = statEnv.getLoopDepth();
166                // #1 for, let                                    
167                ForClause forClause = null;
168                assert (_clauses.size() > 0) : "Binding clause is not found.";
169                for(Binding clause : _clauses) {
170                    clause.staticAnalysis(statEnv);
171                    if(clause.getExpressionType() == Binding.FOR_CLAUSE) {
172                        if(forClause != null) {
173                            throw new IllegalStateException("nested forClause: " + this);
174                        }
175                        forClause = (ForClause) clause;
176                    }
177                }
178                // #2 where, return
179                if(_filteredReturnExpr == null) {
180                    throw new IllegalStateException();
181                }
182                XQExpression normRetExpr = _filteredReturnExpr.staticAnalysis(statEnv);
183                this._type = normRetExpr.getType();
184                this._filteredReturnExpr = normRetExpr;
185                // #3 order by
186                for(OrderSpec orderby : _orderSpecs) {
187                    orderby.staticAnalysis(statEnv);
188                }
189                statEnv.setLoopDepth(loopDepth);
190    
191                XQExpression normed = normalize();
192                XQExpression analyzed = normed.staticAnalysis(statEnv);
193                // join detection
194                if(!ENV_NO_JOIN) {
195                    JoinDetector detector = new JoinDetector();
196                    analyzed.visit(detector, null);
197                }
198                // loop-invariants code motion
199                if(ENV_ENABLE_LOOPOPTIM) {//TODO
200                    if(forClause != null) {
201                        LoopInvariantsTransformer transformer = new LoopInvariantsTransformer(this, forClause);
202                        transformer.visit(this, null);
203                        transformer.transform();
204                    }
205                }
206                return analyzed;
207            }
208            return this;
209        }

211        public Sequence<? extends Item> eval(Sequence<? extends Item> contextSeq, DynamicContext dynEnv)
212                throws XQueryException {
213            Sequence input = new DummySequence(dynEnv);
214            for(Binding bc : _clauses) {
215                final int type = bc.getExpressionType();
216                if(type == Binding.LET_CLAUSE) {
217                    LetVariable lv = ((LetClause) bc).getVariable();
218                    int refcnt = lv.getReferenceCount();
219                    if(refcnt == 0) {
220                        continue;
221                    } else {
222                        input = new ActionSequence(input, bc, dynEnv);
223                    }
224                } else {
225                    input = new PipedActionSequence(input, bc, dynEnv);
226                }
227            }
228            // where + return
229            Sequence ret = new PipedActionSequence(input, _filteredReturnExpr, dynEnv);
230            // order by 
231            // TODO PERFORMANCE eagar ordering 
232            final List<OrderSpec> orderSpecs = _orderSpecs;
233            if(!orderSpecs.isEmpty()) {
234                return sorted(ret, orderSpecs, contextSeq, dynEnv);
235            }
236            return ret;
237        }
238    
239        @Override
240        public void evalAsEvents(XQEventReceiver handler, Sequence<? extends Item> contextSeq, DynamicContext dynEnv)
241                throws XQueryException {
242            if(!_orderSpecs.isEmpty()) {
243                super.evalAsEvents(handler, contextSeq, dynEnv);
244                return;
245            }
246            Sequence<? extends Item> input = new DummySequence(dynEnv);
247            for(Binding bc : _clauses) {
248                final int bindingType = bc.getExpressionType();
249                if(bindingType == Binding.LET_CLAUSE) {
250                    final LetVariable lv = ((LetClause) bc).getVariable();
251                    final int refcnt = lv.getReferenceCount();
252                    if(refcnt == 0) {
253                        continue;
254                    }
255                    input = new ActionSequence(input, bc, dynEnv);
256                } else if(bindingType == Binding.FOR_CLAUSE) {
257                    input = new PipedActionSequence(input, bc, dynEnv);
258                } else {
259                    throw new IllegalStateException("Illegal bindingType: " + bindingType);
260                }
261            }
262    
263            final boolean old = handler.setBlankRequired(false);
264            final IFocus<? extends Item> inputItor = input.iterator();
265            final XQExpression filteredReturnExpr = _filteredReturnExpr;
266            for(Item it : inputItor) {
267                filteredReturnExpr.evalAsEvents(handler, it, dynEnv);
268                handler.setBlankRequired(true);
269            }
270            inputItor.closeQuietly();
271            handler.setBlankRequired(old);
272        }

274        private static Sequence sorted(Sequence src, List<OrderSpec> orderSpecs, Sequence contextSeq, DynamicContext dynEnv) {
275            assert (!orderSpecs.isEmpty());
276            if(src instanceof ISorted) {
277                src = ((ISorted) src).getSource(); // TODO REVIEWME
278            }
279            return new SortedSequence(src, orderSpecs, contextSeq, dynEnv);
280        }

285        public XQExpression normalize() throws XQueryException {
286            final List<XQExpression> nodeps = new LinkedList<XQExpression>();
287            final Map<Binding, XQExpression> dependancies = FLWORArranger.getDependentInWhereExpr(this, nodeps);
288            this._whereExpr = null;
289            XQExpression ret = FLWORArranger.applyForNormalization(this, dependancies);
290            if(!nodeps.isEmpty()) {// create ifExpr with remaining where           
291                int nodepLen = nodeps.size();
292                final XQExpression cond;
293                if(nodepLen == 1) {
294                    cond = nodeps.get(0);
295                } else {
296                    cond = new AndExpr(nodeps);
297                }
298                ret = new IfExpr(cond, ret);
299                this._whereExpr = null;
300            }
301            assert (this._filteredReturnExpr != null);
302            assert (this._whereExpr == null);
303            if(_clauses.isEmpty()) {
304                return _filteredReturnExpr;
305            }
306            this._returnExpr = null;
307            return ret;
308        }

310        private static final class PipedActionSequence extends ProxySequence<Item> {
311            private static final long serialVersionUID = -231486273287919000L;
312    
313            private final XQExpression _actionExpr;
314    
315            public PipedActionSequence(Sequence<? extends Item> input, XQExpression actionExpr, DynamicContext dynEnv) {
316                super(input, dynEnv);
317                assert (actionExpr != null);
318                this._actionExpr = actionExpr;
319            }
320    
321            @Override
322            public ActionFocus iterator() {
323                return new ActionFocus(this, _delegate.iterator(), _dynEnv);
324            }
325    
326            public boolean next(IFocus focus) throws XQueryException {
327                final ActionFocus actFocus = (ActionFocus) focus;
328                if(actFocus.reachedEnd()) {
329                    return false;
330                }
331                final IFocus<? extends Item> inputItor = (IFocus<? extends Item>) actFocus.getBaseFocus();
332                IFocus<? extends Item> actionItor = actFocus.getActionItor();
333                for(int i = 0;; i++) {
334                    final int pos = actFocus.getContextPosition();
335                    if((i != 0 || pos > 0) && actionItor.hasNext()) {
336                        Item actionItem = actionItor.next();
337                        actFocus.setContextItem(actionItem);
338                        return true;
339                    }
340                    if(!inputItor.hasNext()) {
341                        break;
342                    }
343                    final Item nextit = inputItor.next();
344                    // iterate on action in the outer loop.
345                    final Sequence<? extends Item> action = _actionExpr.eval(nextit, _dynEnv);
346                    actionItor = action.iterator();
347                    actFocus.setActionItor(actionItor);
348                }
349                inputItor.closeQuietly();
350                actionItor.closeQuietly();
351                focus.setReachedEnd(true);
352                return false;
353            }
354    
355            @Override
356            public Type getType() {
357                return _actionExpr.getType();
358            }
359        }

385        private static final class ActionSequence extends ProxySequence<Item> {
386            private static final long serialVersionUID = -2722355672697772918L;
387    
388            private final Binding _actionExpr;
389    
390            public ActionSequence(Sequence<? extends Item> input, Binding actionExpr, DynamicContext dynEnv) {
391                super(input, dynEnv);
392                if(actionExpr == null) {
393                    throw new IllegalArgumentException();
394                }
395                this._actionExpr = actionExpr;
396            }
397    
398            public boolean next(IFocus focus) throws XQueryException {
399                final IFocus<? extends Item> inputItor = (IFocus<? extends Item>) focus.getBaseFocus();
400                if(inputItor.hasNext()) {
401                    Item nextit = inputItor.next();
402                    // invoke let action in the outer loop.
403                    _actionExpr.eval(nextit, _dynEnv);
404                    focus.setContextItem(nextit);
405                    return true;
406                }
407                inputItor.closeQuietly();
408                return false;
409            }
410    
411            @Override
412            public Type getType() {
413                return _actionExpr.getType();
414            }
415        }
