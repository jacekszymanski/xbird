094        public BDQExpr(XQExpression endpoint, BindingVariable hostVar, XQExpression expr)
095                throws XQueryException {
096            super();

098                throw new IllegalArgumentException();

101                throw new IllegalArgumentException();
102            }
103            this._endpoint = endpoint;
104            this._hostVar = hostVar;
105            this._queryExpr = expr;
106            this._type = expr.getType();

150        public XQExpression staticAnalysis(StaticContext statEnv) throws XQueryException {
151            if(!_analyzed) {
152                this._analyzed = true;
153    
154                XQExpression analyzedEndpoint = _endpoint.staticAnalysis(statEnv);
155                this._endpoint = analyzedEndpoint;
156    
157                // prepare to ship local variables
158                final List<ShippedVariable> shippedVars = new ArrayList<ShippedVariable>(4);
159                ParametricVariableDetector detector = new ParametricVariableDetector(shippedVars, statEnv);
160                detector.visit(_queryExpr, null);
161    
162                final XQExpression compiledExpr;
163                if(analyzedEndpoint instanceof LiteralExpr) {// distributed compile                 
164                    LiteralExpr epLiteral = (LiteralExpr) analyzedEndpoint;
165                    String endpoint = epLiteral.getValue().stringValue();
166    
167                    final XQEngineClient client = new XQEngineClient(endpoint);
168                    final CompileRequest request = new CompileRequest(_queryExpr);
169    
170                    if(LOG.isInfoEnabled()) {
171                        LOG.info("Invoking remote compilation at [" + endpoint + "]:\n " + _queryExpr);
172                    }
173    
174                    // invokes remote compilation
175                    final Object result;
176                    try {
177                        result = client.execute(request);
178                    } catch (RemoteException e) {
179                        throw new XQueryException(e.getMessage(), e.getCause());
180                    } finally {
181                        try {
182                            client.close();
183                        } catch (RemoteException e) {
184                            LOG.warn("shutdown failed for `" + endpoint + '\'', e);
185                        }
186                    }
187    
188                    compiledExpr = (XQExpression) result;
189                    // reset local expression
190                    if(!shippedVars.isEmpty()) {
191                        ShippedVariableModifier modifier = new ShippedVariableModifier(shippedVars);
192                        modifier.visit(compiledExpr, null);
193                    }
194    
195                    this._parallel = false;
196    
197                } else { // compile locally (some access paths such as index access is disabled) 
198                    boolean prevState = statEnv.isIndicesAccessible();
199                    statEnv.setIndicesAccessible(false); //TODO remote compilation for static endpoint
200                    compiledExpr = _queryExpr.staticAnalysis(statEnv);
201                    statEnv.setIndicesAccessible(prevState);
202                }
203    
204                this._queryExpr = compiledExpr;
205                this._type = compiledExpr.getType();
206                final ThreadedVariable threadedVar = new ThreadedVariable(this);
207                statEnv.addThreadedVariable(threadedVar);
208                return threadedVar;
209            }
210            return this;
211        }

213        private static final class ParametricVariableDetector extends AbstractXQueryParserVisitor {
214            private final List<ShippedVariable> _variablesHolder;
215            private final StaticContext _statEnv;
216            private int _counter = 0;
217    
218            ParametricVariableDetector(List<ShippedVariable> variablesHolder, StaticContext statEnv) {
219                super();
220                this._variablesHolder = variablesHolder;
221                this._statEnv = statEnv;
222            }
223    
224            @Override
225            public ForClause visit(ForClause clause, XQueryContext ctxt) throws XQueryException {
226                ForVariable var = clause.getVariable();
227                var.setInsideRemoteExpr(true);
228                XQExpression e = clause.getExpression();
229                e.visit(this, ctxt);
230                return clause;
231            }
232    
233            @Override
234            public LetClause visit(LetClause clause, XQueryContext ctxt) throws XQueryException {
235                LetVariable lv = clause.getVariable();
236                lv.setInsideRemoteExpr(true);
237                XQExpression e = clause.getExpression();
238                e.visit(this, ctxt);
239                return clause;
240            }
241    
242            @Override
243            public XQExpression visit(VarRef ref, XQueryContext ctxt) throws XQueryException {
244                Variable var = ref.getValue();
245                if(!var.isInsideRemoteExpr()) {
246                    if(var.isAnalyzed()) {
247                        var.staticAnalysis(_statEnv);
248                        ShippedVariable shippedVar = new ShippedVariable(var, _counter++);
249                        _variablesHolder.add(shippedVar);
250                        ref.setValue(shippedVar);
251                    }
252                }
253                return ref;
254            }
255        }

257        private static final class ShippedVariableModifier extends AbstractXQueryParserVisitor {
258            private final List<ShippedVariable> _variablesHolder;
259    
260            ShippedVariableModifier(List<ShippedVariable> variablesHolder) {
261                super();
262                this._variablesHolder = variablesHolder;
263            }
264    
265            @Override
266            public XQExpression visit(VarRef ref, XQueryContext ctxt) throws XQueryException {
267                final Variable var = ref.getValue();
268                if(var instanceof ShippedVariable) {
269                    ShippedVariable shippedVar = (ShippedVariable) var;
270                    int shippedId = shippedVar.getIdentifier();
271                    ShippedVariable original = _variablesHolder.get(shippedId);
272                    XQExpression originalExpr = original.getValue();
273                    shippedVar.setValue(originalExpr);
274                    shippedVar.setLocal(true);
275                }
276                return ref;
277            }
278        }

310    
311            private boolean foundShippedVariable = false;
312            private boolean foundNonDownwardAxis = false;
313    
314            public InsideNonDownwardAxisDetector() {
315                super();

346        public Sequence<? extends Item> eval(Sequence<? extends Item> contextSeq, DynamicContext dynEnv)
347                throws XQueryException {
348            XQExpression outsideQueryExpr = dynEnv.getQueryExpression();
349            OutsideNonDownwardAxisDetector outsideNdaDetector = new OutsideNonDownwardAxisDetector();
350            outsideNdaDetector.visit(outsideQueryExpr, null);
351            final PreparedQueryRequest request = new PreparedQueryRequest(_queryExpr, outsideNdaDetector.foundNonDownwardAxis() ? ReturnType.REMOTE_PADED_SEQUENCE
352                    : RemoteEval.RETURN_TYPE);
353    
354            InsideNonDownwardAxisDetector insideNdaDetector = new InsideNonDownwardAxisDetector();
355            insideNdaDetector.visit(_queryExpr, null);
356            PreparedQueryRequest svRequest = insideNdaDetector.foundNonDownwardAxis() ? new PreparedQueryRequest(_queryExpr, ReturnType.REMOTE_PADED_SEQUENCE)
357                    : request;
358    
359            final ArrayList<ShippedVariable> shippedVars = new ArrayList<ShippedVariable>(4);
360            ShippedVariableCollector collector = new ShippedVariableCollector(shippedVars);
361            collector.visit(_queryExpr, null);
362            final int numVars = shippedVars.size();
363            for(int i = 0; i < numVars; i++) {
364                ShippedVariable sv = shippedVars.get(i);
365                prepareVariablesToShip(svRequest, sv, contextSeq, dynEnv);
366            }
367    
368            final List<String> endpoints = getEndpoints(_endpoint, contextSeq, dynEnv);
369            final int numEndpoints = endpoints.size();
370            if(numEndpoints == 0) {
371                return ValueSequence.EMPTY_SEQUENCE;
372            }
373            final Sequence result;
374            if(numEndpoints == 1) {
375                result = invokeRequest(endpoints.get(0), request);
376            } else if(_parallel) {
377                result = invokeRequestsInParallel(endpoints, _hostVar, request, dynEnv);
378            } else {
379                Sequence lastSeq = invokeRequest(endpoints.get(0), request);
380                for(int i = 1; i < numEndpoints; i++) {
381                    Sequence secSeq = invokeRequest(endpoints.get(i), request);
382                    lastSeq = new ChainedSequence(lastSeq, secSeq, dynEnv);
383                }
384                result = lastSeq;
385            }
386            return result;
387        }

389        private static Sequence invokeRequestsInParallel(final List<String> endpoints, final BindingVariable hostVar, final PreparedQueryRequest request, final DynamicContext dynEnv) {
390            final int numEndpoints = endpoints.size();
391            final ExecutorService ex = ExecutorFactory.newFixedThreadPool(numEndpoints, "parallelRemoteExec");
392            final Sequence[] results = new Sequence[numEndpoints];
393            //final AtomicReferenceArray<Sequence> resultsRef = new AtomicReferenceArray<Sequence>(results);
394            try {
395                for(int i = 0; i < numEndpoints; i++) {
396                    final String endpoint = endpoints.get(i);
397                    final int at = i;
398                    Runnable r = new Runnable() {
399                        public void run() {
400                            final Sequence res;
401                            try {
402                                res = invokeRequest(endpoint, request);
403                            } catch (XQueryException e) {
404                                throw new IllegalStateException("An error caused while evaluating CompiledQueryRequest#"
405                                        + request.getIdentifier() + " at " + endpoint, e);
406                            }
407                            results[at] = res;
408                        }
409                    };
410                    ex.execute(r);
411                }
412            } finally {
413                ExecutorUtils.shutdownAndAwaitTermination(ex);
414            }
415            assert (ex.isTerminated());
416            return new PipelinedSequence(dynEnv, results);
417        }

419        private static Sequence invokeRequest(final String endpoint, final PreparedQueryRequest request)
420                throws XQueryException {
421            if(LOG.isDebugEnabled()) {
422                LOG.debug("Invoking remote execution at [" + endpoint + "]:\n "
423                        + request.getCompiledExpression());
424            }
425            final XQEngineClient client = new XQEngineClient(endpoint);
426            final Object result;
427            try {
428                result = client.execute(request);
429            } catch (RemoteException e) {
430                throw new XQueryException(e.getMessage(), e.getCause());
431            } finally {
432                try {
433                    client.close();
434                } catch (RemoteException e) {
435                    LOG.warn("shutdown failed for `" + endpoint + '\'', e);
436                }
437            }
438            Sequence resultSeq = (Sequence) result;
439            return resultSeq;
440        }

442        private static final class ShippedVariableCollector extends AbstractXQueryParserVisitor {
443            private final List<ShippedVariable> _variablesHolder;
444    
445            public ShippedVariableCollector(final List<ShippedVariable> holder) {
446                super();
447                this._variablesHolder = holder;
448            }
449    
450            @Override
451            public XQExpression visit(VarRef ref, XQueryContext ctxt) throws XQueryException {
452                final Variable var = ref.getValue();
453                if(var instanceof ShippedVariable) {
454                    _variablesHolder.add((ShippedVariable) var);
455                }
456                return ref;
457            }
458        }

460        private static List<String> getEndpoints(final XQExpression endpointExpr, final Sequence<? extends Item> contextSeq, final DynamicContext dynEnv)
461                throws XQueryException {
462            final Sequence ep = endpointExpr.eval(contextSeq, dynEnv);
463            final IFocus<? extends Item> epFocus = ep.iterator();
464            if(!epFocus.hasNext()) {
465                epFocus.closeQuietly();
466                throw new DynamicError("Invalid XQueryD expression. Endpoint does not found");
467            }
468            final List<String> endpoints = new ArrayList<String>(4);
469            do {
470                Item firstItem = epFocus.next();
471                String endpointStr = firstItem.stringValue();
472                endpoints.add(endpointStr);
473            } while(epFocus.hasNext());
474            epFocus.closeQuietly();
475            return endpoints;
476        }
477    
478        private static void prepareVariablesToShip(final QueryRequest request, final ShippedVariable shippedVar, final Sequence<? extends Item> contextSeq, final DynamicContext dynEnv)
479                throws XQueryException {
480            final Sequence result = shippedVar.eval(contextSeq, dynEnv);
481            final ReturnType rettype = request.getReturnType();
482            if(RemoteEval.ENV_NOWRAP_VARSHIP || !rettype.isRemoteSequnece()) {
483                shippedVar.setResult(new MarshalledSequence(result, dynEnv));
484            } else if(result instanceof SingleItem) {
485                shippedVar.setResult(result);
486            } else {
487                final RemoteSequenceProxy proxy = new RemoteSequenceProxy(result, request);
488                try {
489                    UnicastRemoteObject.exportObject(proxy, 0, TimeoutSocketProdiver.createClientSocketFactory(), TimeoutSocketProdiver.createServerSocketFactory());
490                } catch (RemoteException e) {
491                    throw new XQueryException("failed exporting variable: " + shippedVar.getName(), e);
492                }
493                final RemoteSequence remote = new RemoteSequence(proxy, result.getType());
494                shippedVar.setResult(remote);
495            }
496        }
