076        public PathExpr(List<XQExpression> stepList) {
077            if(stepList == null || stepList.isEmpty()) {
078                stepList = Collections.emptyList();
079            }
080            this._steps = stepList;
081            this._type = NodeType.ANYNODE;
082        }

084        public PathExpr(XQExpression... steps) {
085            if(steps.length == 0) {
086                throw new IllegalArgumentException();
087            }
088            final List<XQExpression> list = new LinkedList<XQExpression>();
089            for(XQExpression step : steps) {
090                list.add(step);
091            }
092            this._steps = list;
093            this._type = NodeType.ANYNODE;
094        }

097            super();
098        }

105        public void readExternal(ObjectInput in) throws IOException, ClassNotFoundException {
106            final int numSteps = in.readInt();
107            final ArrayList<XQExpression> steps = new ArrayList<XQExpression>(numSteps);
108            for(int i = 0; i < numSteps; i++) {
109                XQExpression step = (XQExpression) in.readObject();
110                steps.add(step);
111            }
112            this._steps = steps;
113            final boolean hasAnalyzedExpr = in.readBoolean();
114            if(hasAnalyzedExpr) {
115                this._analyzedExpr = (XQExpression) in.readObject();
116            }
117        }
118    
119        public void writeExternal(ObjectOutput out) throws IOException {
120            final List<XQExpression> steps = _steps;
121            final int numSteps = steps.size();
122            out.writeInt(numSteps);
123            for(int i = 0; i < numSteps; i++) {
124                XQExpression step = steps.get(i);
125                out.writeObject(step);
126            }
127            final XQExpression analyzed = _analyzedExpr;
128            if(analyzed == null) {
129                out.writeBoolean(false);
130            } else {
131                out.writeBoolean(true);
132                out.writeObject(analyzed);
133            }
134        }

136        public XQExpression setStep(int i, XQExpression step) {
137            return _steps.set(i, step);
138        }

141            return _steps;

144        public void addPredicate(XQExpression pred) {
145            int lastidx = _steps.size() - 1;
146            XQExpression lastStep = _steps.get(lastidx);
147            if(lastStep instanceof FilterExpr) {
148                ((FilterExpr) lastStep).addPredicate(pred);
149            } else {
150                FilterExpr filter = new FilterExpr(lastStep);
151                filter.addPredicate(pred);
152                _steps.set(lastidx, filter);
153            }
154        }

159        public XQExpression staticAnalysis(StaticContext statEnv) throws XQueryException {
160            this._analyzed = true;
161            final List<XQExpression> step = _steps;
162            if(step.isEmpty()) {
163                return new ContextItemExpr();
164            }
165            // simplify PathExpr
166            simplify();
167            // pre-static analyse
168            final int stepsize = step.size();
169            final int lastpos = stepsize - 1;
170            for(int i = 0; i < stepsize; i++) {
171                XQExpression e = step.get(i);
172                XQExpression analyzed = e.staticAnalysis(statEnv);
173                if(i != lastpos) {
174                    final Type type = analyzed.getType();
175                    if(type instanceof AtomicType) {
176                        reportError("err:XPTY0019", "Illegal result type was detected in '" + analyzed
177                                + "', its type: " + type);
178                    }
179                }
180                step.set(i, analyzed);
181            }
182            // take the first/last StepExpr off from PathExpr
183            if(stepsize == 1) { // TODO needs review (relative path case, and so on..)
184                XQExpression last = step.get(0);
185                if(last instanceof PathExpr) {
186                    throw new IllegalStateException("Unexpected expression class: "
187                            + last.getClass().getName());
188                }
189                this._analyzedExpr = last;
190                return last;
191            }
192            // formal semantics annotation
193            final boolean annotate = isDistinctDocOrderRequired();
194            boolean reverse = false;
195            if(annotate) {
196                final int size = step.size();
197                if(size > 1) {
198                    XQExpression last = step.get(size - 1);
199                    if(last instanceof ReverseAxis) {
200                        reverse = true;
201                    }
202                }
203            }
204            // normalize
205            XQExpression normed = normalize(statEnv);
206            XQExpression analysed = normed.staticAnalysis(statEnv);
207            assert (analysed != this);
208            this._type = analysed.getType();
209            this._analyzedExpr = analysed;
210            if(annotate) {
211                if(LOG.isDebugEnabled()) {
212                    LOG.debug("Applying Distinct-document-order: " + analysed);
213                }
214                analysed = applyDistinctDocOrder(analysed, reverse);
215            }
216            return analysed;
217        }

219        /**
220         * Rewrite(simplify) PathExpr based on huristic rules.
221         */
222        private void simplify() throws XQueryException {
223            XQExpression prevStep = null;
224            final List<XQExpression> step = _steps;
225            for(int i = 0; i < step.size(); i++) {
226                XQExpression currStep = step.get(i);
227                if(i > 0) {
228                    if(prevStep instanceof DescendantOrSelfStep) {
229                        if(currStep instanceof ChildStep) {
230                            NodeTest prevNodeTest = ((DescendantOrSelfStep) prevStep).getNodeTest();
231                            if(NodeTest.ANYNODE.equals(prevNodeTest)) {
232                                //  rewrite rule #1
233                                //    - '/descendant-or-self::node()/child::` => `/descendant::`
234                                XQExpression removed = step.remove(--i);
235                                assert (removed == prevStep);
236                                NodeTest replacedNodeTest = ((ChildStep) currStep).getNodeTest();
237                                DescendantStep replaced = new DescendantStep(replacedNodeTest);
238                                step.set(i, replaced);
239                            }
240                        } else if(currStep instanceof DescendantOrSelfStep
241                                || currStep instanceof DescendantStep) {
242                            //  rewrite rule #2
243                            //   - '/descendant-or-self::node()/descendant-or-self::` => `/descendant-or-self::`
244                            //  rewrite rule #3
245                            //   - '/descendant-or-self::node()/descendant::` => `/descendant::`
246                            NodeTest prevNodeTest = ((DescendantOrSelfStep) prevStep).getNodeTest();
247                            if(NodeTest.ANYNODE.equals(prevNodeTest)) {
248                                XQExpression removed = step.remove(i - 1);
249                                assert (removed == prevStep);
250                            }
251                        }
252                    }
253                }
254                prevStep = currStep;
255            }
256        }

258        /**
259         * Annotates fn:distinct-docorder.
260         * 
261         * @link http://www.w3.org/TR/xquery-semantics/#sec_distinct_docorder_or_atomic_sequence
262         */
263        private boolean isDistinctDocOrderRequired() {
264            XQExpression rawPrevStep = null;
265            final List<XQExpression> step = _steps;
266            final int len = step.size();
267            for(int i = 0; i < len; i++) {
268                XQExpression curStep = step.get(i);
269                if(i > 0) {
270                    final boolean sortRequired = sortRequired(rawPrevStep, curStep);
271                    if(sortRequired) {
272                        return true;
273                    }
274                }
275                rawPrevStep = curStep;
276            }
277            return false;
278        }

280        /**
281         * Apply KNormalization.
282         */
283        private AbstractXQExpression normalize(StaticContext statEnv) throws XQueryException {
284            final List<XQExpression> step = _steps;
285            if(step.isEmpty()) {
286                throw new IllegalStateException();
287            }
288            PathVariable var = null;
289            final int steplen = step.size();
290            final RewriteInfo info = new RewriteInfo();
291            boolean indexAccessable = ENV_DISABLE_INDEX_ACCESS ? false : true;
292            int lastRewritten = -1;
293            for(int i = 0; i < steplen; i++) {
294                final XQExpression curStep = step.get(i);
295                if(indexAccessable) {
296                    indexAccessable &= curStep.isPathIndexAccessable(statEnv, info);
297                }
298                if(indexAccessable && info.isLookaheadRequired() && (i + 1) < steplen) {
299                    indexAccessable = false;
300                    int ni = i + 1;
301                    XQExpression nextStep = step.get(ni);
302                    if(nextStep instanceof DescendantStep) {
303                        indexAccessable = nextStep.isPathIndexAccessable(statEnv, info);
304                        if(indexAccessable) {
305                            FilteredPathIndexAcccessExpr filtered = new FilteredPathIndexAcccessExpr(info, curStep.getType());
306                            var = PathVariable.create(filtered, statEnv, false);
307                            step.set(ni, var);
308                            lastRewritten = ni;
309                            ++i;
310                            continue;
311                        }
312                    }
313                }
314                if(i == 0) {
315                    var = PathVariable.create(curStep, statEnv, false);
316                } else {
317                    if(indexAccessable) {
318                        boolean isFilter = (curStep instanceof FilterExpr);
319                        // If filtered, no more index access is enabled for the time being
320                        indexAccessable = !isFilter;
321                        if(i >= ENV_INDEXING_THRESHOLD) {
322                            if(isFilter) {
323                                var = PathVariable.create(curStep, statEnv, false);
324                            } else {
325                                PathIndexAccessExpr idxExpr = new PathIndexAccessExpr(info, curStep.getType());
326                                var = PathVariable.create(idxExpr, statEnv, true);
327                            }
328                            lastRewritten = i;
329                        } else {
330                            final CompositePath cp = new CompositePath(var, curStep);
331                            var = PathVariable.create(cp, statEnv, false);
332                        }
333                    } else {
334                        final CompositePath cp = new CompositePath(var, curStep);
335                        var = PathVariable.create(cp, statEnv, false);
336                    }
337                }
338                step.set(i, var);
339            }
340            for(int i = 0; i < lastRewritten; i++) {
341                step.remove(0);
342            }
343            assert (var != null);
344            return var;
345        }

348                throws XQueryException {
349            if(_analyzedExpr == null) {

359        /**
360         * If the input is a sequence of nodes, is sorts those nodes by document order 
361         * and removes duplicates.
362         * 
363         * @link http://www.w3.org/TR/xquery-semantics/#sec_distinct_docorder_or_atomic_sequence
364         */
365        private static boolean sortRequired(XQExpression prev, XQExpression curr) {
366            assert (prev != null && curr != null);
367            // Note that sorting by document order enforces the restriction
368            // that input and output sequences contains only nodes, 
369            // and that the last step in a path expression may actually return atomic values.       
370            Type currType = curr.getType();
371            if(TypeUtil.subtypeOf(currType, TypeRegistry.safeGet("node()*"))) {
372                if(prev instanceof FilterExpr) {
373                    return false;
374                } else if(prev instanceof AxisStep && curr instanceof AxisStep) {
375                    int prevAxisKind = ((AxisStep) prev).getAxisKind();
376                    int currAxisKind = ((AxisStep) curr).getAxisKind();
377                    switch(currAxisKind) {
378                        case AxisStep.CHILD:
379                        case AxisStep.ATTR:
380                            return false;
381                        default:
382                            break;
383                    }
384                    switch(prevAxisKind) {
385                        case AxisStep.CHILD:
386                            if(currAxisKind == AxisStep.CHILD || currAxisKind == AxisStep.SELF) {
387                                return false;
388                            }
389                            break;
390                        case AxisStep.FOLLOWING_SIBLING:
391                        case AxisStep.PRECEDING_SIBLING:
392                            if(currAxisKind == AxisStep.CHILD || currAxisKind == AxisStep.DESC
393                                    || currAxisKind == AxisStep.DESC_OR_SELF
394                                    || currAxisKind == AxisStep.SELF) {
395                                return false;
396                            }
397                            break;
398                        case AxisStep.DESC:
399                        case AxisStep.DESC_OR_SELF:
400                        case AxisStep.FOLLOWING:
401                        case AxisStep.PRECEDING:
402                            if(currAxisKind == AxisStep.SELF) { /* || currAxisKind == AxisStep.CHILD */
403                                return false;
404                            }
405                            break;
406                        case AxisStep.PARENT:
407                        case AxisStep.SELF:
408                            return false;
409                        case AxisStep.ANCESTOR:
410                        case AxisStep.ANCESTOR_OR_SELF:
411                            if(currAxisKind == AxisStep.CHILD // TODO REVIEW
412                                    || currAxisKind == AxisStep.FOLLOWING_SIBLING
413                                    || currAxisKind == AxisStep.PRECEDING_SIBLING
414                                    || currAxisKind == AxisStep.PARENT || currAxisKind == AxisStep.SELF) {
415                                return false;
416                            }
417                            break;
418                        case AxisStep.ATTR:
419                            return false;
420                        default:
421                            throw new IllegalStateException("Invalid axis type: " + prevAxisKind);
422                    }
423                    return true;
424                } else {
425                    return false;
426                }
427            }
428            return false;
429        }

434        public static final class CompositePath extends AbstractXQExpression {
435            private static final long serialVersionUID = 303229194702711722L;
436    
437            private final Variable srcVar;
438            private XQExpression filterExpr;
439    
440            public CompositePath(Variable srcVar, XQExpression filterExpr) {
441                if(filterExpr instanceof CompositePath) {
442                    throw new IllegalStateException();
443                }
444                this.srcVar = srcVar;
445                this.filterExpr = filterExpr;
446            }
447    
448            public XQExpression visit(XQueryParserVisitor visitor, XQueryContext ctxt)
449                    throws XQueryException {
450                return visitor.visit(this, ctxt);
451            }
452    
453            public Variable getSourceVariable() {
454                return srcVar;
455            }
456    
457            public XQExpression getSourceExpr() {
458                return srcVar.getValue();
459            }
460    
461            public XQExpression getFilterExpr() {
462                return filterExpr;
463            }
464    
465            public void setFilterExpr(XQExpression filter) {
466                this.filterExpr = filter;
467            }
468    
469            public XQExpression staticAnalysis(StaticContext statEnv) throws XQueryException {
470                if(!_analyzed) {
471                    this._analyzed = true;
472                    // #1 static type analysis
473                    //srcVar.staticAnalysis(statEnv);
474                    final XQExpression analysedFilter = filterExpr.staticAnalysis(statEnv);
475                    final XQExpression arrangedExpr = rewrite(srcVar, analysedFilter, statEnv);
476                    if(arrangedExpr != null) {
477                        return arrangedExpr.staticAnalysis(statEnv);
478                    } else {
479                        XQExpression prev = srcVar.getValue();
480                        srcVar.staticAnalysis(statEnv);
481                        XQExpression curr = srcVar.getValue();
482                        if(curr != prev) {
483                            this._analyzed = false;
484                            return this.staticAnalysis(statEnv);
485                        }
486                    }
487                    this.filterExpr = analysedFilter;
488                    this._type = analysedFilter.getType();
489                }
490                return this;
491            }
492    
493            private static XQExpression rewrite(final Variable srcVar, final XQExpression filterExpr, final StaticContext statEnv)
494                    throws XQueryException {
495                if(srcVar instanceof ForVariable) {
496                    return null; //TODO REVIEWME
497                }
498                final XQExpression srcExpr = srcVar.getValue();
499                if(srcExpr instanceof VarRef) {
500                    VarRef ref = (VarRef) srcExpr;
501                    Variable var = ref.getValue();
502                    assert (var != null);
503                    return rewrite(var, filterExpr, statEnv);
504                } else if(srcExpr instanceof Variable) {
505                    Variable var = (Variable) srcExpr;
506                    return rewrite(var, filterExpr, statEnv);
507                } else if(srcExpr instanceof BDQExpr) {
508                    final BDQExpr distExpr = (BDQExpr) srcExpr;
509                    final XQExpression bodyExpr = distExpr.getBodyExpression();
510                    if(bodyExpr instanceof PathVariable) {
511                        PathVariable bodyVar = (PathVariable) bodyExpr;
512                        CompositePath optExpr = new CompositePath(bodyVar, filterExpr);
513                        distExpr.setBodyExpression(optExpr);
514                        return distExpr;
515                    } else {
516                        AnonymousLetVariable wrappedVar = new AnonymousLetVariable(bodyExpr);
517                        CompositePath optExpr = new CompositePath(wrappedVar, filterExpr);
518                        distExpr.setBodyExpression(optExpr);
519                        return distExpr;
520                    }
521                }
522                return null;
523            }
524    
525            public CompositeSequence<XQNode> eval(Sequence<? extends Item> contextSeq, DynamicContext dynEnv)
526                    throws XQueryException {
527                final Sequence<? extends Item> src = srcVar.eval(contextSeq, dynEnv);
528                final CompositeSequence<XQNode> cs = new CompositeSequence<XQNode>(src, filterExpr, dynEnv);
529                return cs;
530            }
