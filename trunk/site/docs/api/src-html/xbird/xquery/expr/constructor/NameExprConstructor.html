044     */
045    public abstract class NameExprConstructor extends AbstractXQExpression implements NodeConstructor {
046        private static final long serialVersionUID = 1L;
047    
048        protected XQExpression _nameExpr = null;
049    
050        public NameExprConstructor() {
051            super();
052        }
053    
054        public XQExpression getNameExpr() {
055            return this._nameExpr;
056        }
057    
058        public void setNameExpr(XQExpression e) {
059            this._nameExpr = e;
060        }
061    
062        protected QualifiedName processNameExpr(Sequence contextSeq, DynamicContext dynEnv)
063                throws XQueryException {
064            final Sequence instNameSeq = _nameExpr.eval(contextSeq, dynEnv);
065            final Sequence<AtomicValue> qnameSeq = AtomizedSequence.wrap(instNameSeq, dynEnv);
066            final IFocus<AtomicValue> qnameItor = qnameSeq.iterator();
067            if(!qnameItor.hasNext()) {
068                qnameItor.closeQuietly();
069                reportError("err:XPTY0004", "Conversion of the atomized name expression to an expanded QName was not successful.");
070            }
071            final AtomicValue it = qnameItor.next();
072            if(qnameItor.hasNext()) {
073                qnameItor.closeQuietly();
074                reportError("err:XPTY0004", "Conversion of the atomized name expression to an expanded QName was not successful.");
075            }
076            qnameItor.closeQuietly();
077            final QualifiedName namev;
078            final Type t = it.getType();
079            if(TypeUtil.subtypeOf(t, QNameType.QNAME)) {
080                final QNameValue qname = ((AtomicValue) it).castAs(QNameType.QNAME, dynEnv);
081                namev = qname.getValue();
082            } else if(TypeUtil.subtypeOf(t, StringType.STRING) || t == UntypedAtomicType.UNTYPED_ATOMIC) {
083                final StaticContext sc = dynEnv.getStaticContext();
084                try {
085                    final CharSequence namecs = it.stringValue();
086                    final String namestr = (namecs == null) ? null : namecs.toString();
087                    namev = QNameUtil.parse(namestr, sc.getStaticalyKnownNamespaces(), sc.getDefaultElementNamespace());
088                } catch (Exception e) {
089                    throw new DynamicError("err:XQDY0074", e);
090                }
091            } else {
092                reportError("err:XPTY0004", "Convertion to name expression is failed.");
093                throw new IllegalStateException();
094            }
095            return namev;
096        }
097    
098    }
