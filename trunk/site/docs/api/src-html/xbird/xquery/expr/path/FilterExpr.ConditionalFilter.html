102        public XQExpression staticAnalysis(StaticContext statEnv) throws XQueryException {
103            if(!_analyzed) {
104                this._analyzed = true;
105                XQExpression src = _srcExpr.staticAnalysis(statEnv);
106                EagarAnnotator annotator = new EagarAnnotator();
107                annotator.visit(src, null);
108                Type inferredType = src.getType();
109                this._srcExpr = src;
110                for(int i = 0; i < _predicates.size(); i++) {
111                    XQExpression p = _predicates.get(i);
112                    statEnv.setContextItemStaticType(inferredType);
113                    XQExpression anylyzed = p.staticAnalysis(statEnv);
114                    _predicates.set(i, anylyzed);
115                }
116                //this._type = inferredType;
117                return simplify(statEnv);
118            }
119            return this;
120        }

140        @Override
141        public boolean isPathIndexAccessable(StaticContext statEnv, RewriteInfo info) {
142            if(!info.hasPreviousStep()) {
143                return false;
144            }
145            if(_hasContextualFilter) {
146                return false;
147            }
148            if(_srcExpr.isPathIndexAccessable(statEnv, info)) {
149                PathIndexAccessExpr idxAccessExpr = new PathIndexAccessExpr(info, _srcExpr.getType());
150                _srcExpr = PathVariable.create(idxAccessExpr, statEnv, true);
151                return true;
152            } else {
153                return false;
154            }
155        }

157        public Sequence<? extends Item> eval(Sequence<? extends Item> contextSeq, DynamicContext dynEnv)
158                throws XQueryException {
159            Sequence<? extends Item> res = _srcExpr.eval(contextSeq, dynEnv); // TODO REVIEWME        
160            for(XQExpression p : _predicates) {
161                res = p.eval(res, dynEnv); // apply filtering
162            }
163            return res;
164        }

166        /**
167         * Some optimization(de-normalization) is applied.
168         */
169        private AbstractXQExpression simplify(StaticContext statEnv) throws XQueryException {
170            if(TypeUtil.subtypeOf(_type, SequenceType.EMPTY)) {
171                return SequenceExpression.EMPTY_SEQUENCE;
172            }
173            Type baseType = _srcExpr.getType();
174            for(int i = 0; i < _predicates.size(); i++) {
175                XQExpression e = _predicates.get(i);
176                if(e instanceof LiteralExpr) {
177                    AtomicValue literal = ((LiteralExpr) e).getValue();
178                    if(literal instanceof XNumber) {
179                        DynamicContext probe = DynamicContext.PROBE;
180                        probe.setStaticContext(statEnv);
181                        XNumber xnum = (XNumber) literal;
182                        double xdouble = xnum.asDouble();
183                        long xint = xnum.asLong();
184                        if(xdouble != xint) {
185                            return SequenceExpression.EMPTY_SEQUENCE;
186                        }
187                        if(xint > Integer.MAX_VALUE) {
188                            throw new IllegalStateException();
189                        }
190                        int pos = (int) xint;
191                        if(pos < 1) {
192                            return SequenceExpression.EMPTY_SEQUENCE;
193                        }
194                        final PositionFilter filter = new PositionFilter(pos, baseType);
195                        _predicates.set(i, filter);
196                        baseType = filter.getType();
197                        _hasContextualFilter = true;
198                        continue;
199                    }
200                } else if(e instanceof FunctionCall) {
201                    FunctionCall fc = (FunctionCall) e;
202                    BuiltInFunction fnlast = PredefinedFunctions.lookup(Last.SYMBOL);
203                    if(fnlast.getName().equals(fc.getFuncName())) { // if fn:last()
204                        final LastFilter filter = new LastFilter(baseType);
205                        _predicates.set(i, filter);
206                        baseType = filter.getType();
207                        _hasContextualFilter = true;
208                        continue;
209                    }
210                } else if(e instanceof PathVariable) {
211                    final ExistsFilter filter = new ExistsFilter(e, baseType);
212                    _predicates.set(i, filter);
213                    _hasContextualFilter = true;
214                    continue;
215                }
216                final ConditionalFilter filter = new ConditionalFilter(e, baseType);
217                _predicates.set(i, filter);
218            }
219            this._type = baseType;
220            return this;
221        }

240        private static final class PositionFilter extends Instruction {
241            private static final long serialVersionUID = -8028124545542650688L;
242    
243            private final int pos;
244    
245            public PositionFilter(int pos, Type baseType) {
246                super();
247                if(pos < 1) {
248                    throw new IllegalArgumentException("Illegal position: " + pos);
249                }
250                this.pos = pos;
251                if(baseType instanceof ListType) {
252                    ListType list = ((ListType) baseType);
253                    int size = list.size();
254                    if(pos > size) {
255                        // REVIEWME item seems to be not exist in the position
256                        this._type = baseType.prime();
257                        //this._type = SequenceType.EMPTY;
258                    } else {
259                        int i = pos - 1;
260                        this._type = list.get(i);
261                    }
262                } else {
263                    this._type = baseType.prime();
264                }
265            }
266    
267            public Sequence<? extends Item> eval(Sequence<? extends Item> contextSeq, DynamicContext dynEnv)
268                    throws XQueryException {
269                if(pos <= 0) {
270                    return ValueSequence.EMPTY_SEQUENCE;
271                }
272                return new Sliced(contextSeq, pos, pos, dynEnv, _type);
273            }
274    
275            @Override
276            public String toString() {
277                return Integer.toString(pos);
278            }
279        }

281        private static final class LastFilter extends Instruction {
282            private static final long serialVersionUID = 8332527761105927517L;
283    
284            public LastFilter(Type baseType) {
285                if(baseType instanceof ListType) {
286                    ListType list = ((ListType) baseType);
287                    int size = list.size();
288                    if(size > 0) {
289                        this._type = list.get(size - 1);
290                    } else {
291                        // REVIEWME item seems to be not exist
292                        this._type = baseType.prime();
293                        //this._type = SequenceType.EMPTY;
294                    }
295                } else {
296                    this._type = baseType.prime();
297                }
298            }
299    
300            public LastOf eval(Sequence<? extends Item> contextSeq, DynamicContext dynEnv)
301                    throws XQueryException {
302                return new LastOf(contextSeq, dynEnv, _type);
303            }
304    
305            @Override
306            public String toString() {
307                return "last()";
308            }

367        public static final class Sliced extends ProxySequence {
368            private static final long serialVersionUID = 7613358886773972861L;
369    
370            private final int _from, _to;
371            private final Type type;
372    
373            public Sliced(Sequence<? extends Item> src, int from, int to, DynamicContext dynEnv, Type type) {
374                super(src, dynEnv);
375                if(from > to || from < 0) {
376                    throw new IllegalArgumentException("Illegal from: " + from + ", to: " + to);
377                }
378                this._from = from;
379                this._to = to;
380                this.type = type;
381            }
382    
383            public boolean next(IFocus focus) throws XQueryException {
384                final IFocus<? extends Item> srcItor = (IFocus<? extends Item>) focus.getBaseFocus();
385                while(true) {
386                    final int curidx = focus.incrPosition();
387                    if(curidx > _to || !srcItor.hasNext()) {
388                        srcItor.closeQuietly();
389                        focus.setReachedEnd(true);
390                        return false;
391                    }
392                    Item it = srcItor.next();
393                    if(curidx >= _from) {
394                        focus.setContextItem(it);
395                        return true;
396                    }
397                }
398            }
399    
400            @Override
401            public Type getType() {
402                return type;
403            }
404    
405        }

407        public static final class LastOf extends ProxySequence {
408            private static final long serialVersionUID = 4988642397046873632L;
409            private final Type type;
410    
411            public LastOf(Sequence delegate, DynamicContext dynEnv, Type type) {
412                super(delegate, dynEnv);
413                this.type = type;
414            }
415    
416            public boolean next(IFocus focus) throws XQueryException {
417                if(focus.reachedEnd()) {
418                    return false;
419                }
420                Iterator<? extends Item> itor = focus.getBaseFocus();
421                while(itor.hasNext()) {
422                    final Item it = itor.next();
423                    if(!itor.hasNext()) {
424                        focus.setContextItem(it);
425                        focus.setReachedEnd(true);
426                        return true;
427                    }
428                }
429                focus.setReachedEnd(true);
430                return false;
431            }
432    
433            @Override
434            public Type getType() {
435                return type;
436            }

445            public Exists(XQExpression predicate, Sequence delegate, DynamicContext dynEnv, Type type) {

451            public boolean next(IFocus focus) throws XQueryException {
452                final Iterator<? extends Item> itor = focus.getBaseFocus();
453                while(itor.hasNext()) {
454                    final Item it = itor.next();
455                    final Sequence cond = predicate.eval(it, _dynEnv);
456                    if(!cond.isEmpty()) {
457                        focus.setContextItem(it);
458                        focus.setReachedEnd(true);
459                        return true;
460                    }
461                }
462                focus.setReachedEnd(true);
463                return false;
464            }

468                return type;
469            }

473        public static final class Filtered extends ProxySequence {
474            private static final long serialVersionUID = -7637704463461420787L;
475    
476            private final XQExpression predicate;
477            private final Type type;
478    
479            public Filtered(XQExpression predicate, Sequence delegate, DynamicContext dynEnv, Type type) {
480                super(delegate, dynEnv);
481                this.predicate = predicate;
482                this.type = type;
483            }
484    
485            public boolean next(IFocus focus) throws XQueryException {
486                final DynamicContext dynEnv = _dynEnv;
487                Iterator<? extends Item> itor = focus.getBaseFocus();
488                while(itor.hasNext()) {
489                    int curidx = focus.incrPosition();
490                    Item it = itor.next();
491                    focus.setContextItem(it);
492    
493                    // REVIEWME workaround for fn:last()
494                    dynEnv.pushSequence(_delegate);
495    
496                    // evaluates predicate
497                    IFocus baseFocus = dynEnv.getFocus();
498                    dynEnv.setFocus(focus); // change focus
499                    Sequence cond = predicate.eval(it, dynEnv);
500                    dynEnv.setFocus(baseFocus);
501    
502                    // workaround for fn:last()
503                    dynEnv.popSequence();
504    
505                    if(cond instanceof XInteger) { // positional filtering
506                        final long pos = ((XInteger) cond).getValue();
507                        if(curidx == pos) {
508                            return true;
509                        } else {
510                            continue;
511                        }
512                    }
513    
514                    dynEnv.pushSequence(_delegate);
515                    final boolean ebv = FnBoolean.effectiveBooleanValue(cond, curidx);
516                    dynEnv.popSequence();
517                    if(ebv) {
518                        return true;
519                    } else {
520                        continue;
521                    }
522                }
523                return false;
524            }
525    
526            @Override
527            public Type getType() {
528                return type;
529            }
