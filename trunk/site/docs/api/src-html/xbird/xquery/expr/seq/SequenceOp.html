021    package xbird.xquery.expr.seq;
022    
023    import xbird.xquery.TypeError;
024    import xbird.xquery.XQueryException;
025    import xbird.xquery.expr.AbstractXQExpression;
026    import xbird.xquery.expr.XQExpression;
027    import xbird.xquery.meta.StaticContext;
028    import xbird.xquery.meta.XQueryContext;
029    import xbird.xquery.misc.TypeUtil;
030    import xbird.xquery.parser.XQueryParserVisitor;
031    import xbird.xquery.type.Type;
032    
033    /**
034     * 
035     * <DIV lang="en"></DIV>
036     * <DIV lang="ja"></DIV>
037     * 
038     * @author Makoto YUI (yuin405+xbird@gmail.com)
039     * @link http://www.w3.org/TR/xquery-semantics/#id-sequence-expressions
040     */
041    public abstract class SequenceOp extends AbstractXQExpression {
042        private static final long serialVersionUID = 1L;
043    
044        protected XQExpression _leftOperand, _rightOperand;
045    
046        protected SequenceOp(XQExpression leftOperand, XQExpression rightOperand, Type type) {
047            this._leftOperand = leftOperand;
048            this._rightOperand = rightOperand;
049            this._type = type;
050        }
051    
052        public XQExpression getLeftOperand() {
053            return this._leftOperand;
054        }
055    
056        public XQExpression getRightOperand() {
057            return this._rightOperand;
058        }
059    
060        public abstract String getOperator();
061    
062        public XQExpression visit(XQueryParserVisitor visitor, XQueryContext ctxt)
063                throws XQueryException {
064            return visitor.visit(this, ctxt);
065        }
066    
067        public XQExpression staticAnalysis(StaticContext statEnv) throws XQueryException {
068            if(!_analyzed) {
069                this._analyzed = true;
070                this._leftOperand = _leftOperand.staticAnalysis(statEnv);
071                this._rightOperand = _rightOperand.staticAnalysis(statEnv);
072                if(!TypeUtil.subtypeOf(_leftOperand.getType(), _type)) {
073                    throw new TypeError("Inferred type of left operand is invalid: "
074                            + _leftOperand.getType());
075                }
076                if(!TypeUtil.subtypeOf(_rightOperand.getType(), _type)) {
077                    throw new TypeError("Inferred type of left operand is invalid: "
078                            + _rightOperand.getType());
079                }
080            }
081            return this;
082        }
083    
084    }
