138            @Override
139            public XQExpression visit(FilterExpr filterExpr, XQueryContext ctxt) throws XQueryException {
140                boolean isFirst = true;
141                int shipId = -1;
142                List<XQExpression> srcFilter = Collections.<XQExpression> emptyList();
143                List<XQExpression> persistKeyExprs = Collections.<XQExpression> emptyList();
144                List<XQExpression> searchKeyExprs = Collections.<XQExpression> emptyList();
145                final List<XQExpression> predicates = filterExpr.getPredicates();
146                String opr = null;
147                for(XQExpression p : predicates) {
148                    if(p instanceof ConditionalFilter) {
149                        XQExpression pred = ((ConditionalFilter) p).getPredicate();
150                        if((pred instanceof GeneralComp) || (pred instanceof NodeComp)) {
151                            ComparisonOp cmpOp = (ComparisonOp) pred;
152                            String newOpr = cmpOp.getOperator();
153                            if(opr != null && opr != newOpr) {
154                                break;
155                            }
156                            opr = newOpr;
157                            XQExpression left = cmpOp.getLeftOperand();
158                            if(left instanceof VarRef) {
159                                left = ((VarRef) left).getValue();
160                            }
161                            if(left instanceof PathVariable) {
162                                left = ((PathVariable) left).getValue();
163                            }
164                            XQExpression right = cmpOp.getRightOperand();
165                            if(right instanceof VarRef) {
166                                right = ((VarRef) right).getValue();
167                            }
168                            if(right instanceof PathVariable) {
169                                right = ((PathVariable) right).getValue();
170                            }
171                            XQExpression leftSrc = left, rightSrc = right;
172                            while(leftSrc instanceof CompositePath) {
173                                leftSrc = ((CompositePath) leftSrc).getSourceExpr();
174                            }
175                            while(rightSrc instanceof CompositePath) {
176                                rightSrc = ((CompositePath) rightSrc).getSourceExpr();
177                            }
178                            VarRef pathSrc = null;
179                            XQExpression persistKeyExpr = null, searchKeyExpr = null;
180                            if(leftSrc instanceof VarRef && !(rightSrc instanceof VarRef)) {
181                                pathSrc = (VarRef) leftSrc;
182                                persistKeyExpr = right;
183                                searchKeyExpr = left;
184                            } else if(rightSrc instanceof VarRef && !(leftSrc instanceof VarRef)) {
185                                pathSrc = (VarRef) rightSrc;
186                                persistKeyExpr = left;
187                                searchKeyExpr = right;
188                            }
189                            if(pathSrc != null) {
190                                if(isFirst) {
191                                    persistKeyExprs = new ArrayList<XQExpression>(4);
192                                    searchKeyExprs = new ArrayList<XQExpression>(4);
193                                }
194                                Variable pv = pathSrc.getValue();
195                                if(pv instanceof BindingVariable) {
196                                    if(searchKeyExpr instanceof CompositePath
197                                            && _currentForClause != null) {
198                                        assert (_currentBindings != null);
199                                        final LetVariable lv = new AnonymousLetVariable(searchKeyExpr);
200                                        lv.incrementReferenceCount();
201                                        final int bsize = _currentBindings.size();
202                                        assert (bsize > 0) : bsize;
203                                        boolean found = false;
204                                        for(int i = 0; i < bsize; i++) {
205                                            Binding binding = _currentBindings.get(i);
206                                            if(binding == _currentForClause) {
207                                                LetClause lc = new LetClause(lv);
208                                                _currentBindings.add(i, lc);
209                                                found = true;
210                                                break;
211                                            }
212                                        }
213                                        assert (found);
214                                        searchKeyExpr = lv;
215                                    }
216                                    persistKeyExprs.add(persistKeyExpr);
217                                    searchKeyExprs.add(searchKeyExpr);
218                                    isFirst = false;
219                                    continue;
220                                } else if(pv instanceof ShippedVariable) {
221                                    shipId = ((ShippedVariable) pv).getIdentifier();
222                                    persistKeyExprs.add(persistKeyExpr);
223                                    searchKeyExprs.add(searchKeyExpr);
224                                    isFirst = false;
225                                    continue;
226                                }
227                            }
228                            if(isFirst) {
229                                if(srcFilter == Collections.EMPTY_LIST) {
230                                    srcFilter = new ArrayList<XQExpression>(4);
231                                }
232                                srcFilter.add(p);
233                            } else {
234                                break;
235                            }
236                        }
237                    }
238                }
239                if(!isFirst && _lastPathVariable != null) { // found join
240                    XQExpression pvExpr = _lastPathVariable.getValue();
241                    final XQExpression srcExpr;
242                    if(pvExpr instanceof CompositePath) {
243                        CompositePath srcPath = (CompositePath) pvExpr;
244                        XQExpression filterBaseExpr = filterExpr.getSourceExpr();
245                        srcPath.setFilterExpr(filterBaseExpr);
246                        srcExpr = srcPath;
247                    } else if(pvExpr instanceof FilterExpr) {//FIXME REVIEWME
248                        srcExpr = filterExpr.getSourceExpr();
249                    } else {
250                        throw new IllegalStateException("Unexpected expression is binded to last PathVariable "
251                                + _lastPathVariable + ": " + pvExpr);
252                    }
253                    PromoteJoinExpression promoteJoinExpr = new PromoteJoinExpression(srcExpr, srcFilter, persistKeyExprs, opr, searchKeyExprs);
254                    promoteJoinExpr._shippedId = shipId;
255                    _lastPathVariable.setValue(promoteJoinExpr);
256                    _lastPathVariable = null;
257                }
258                return filterExpr;
259            }

263            private static final long serialVersionUID = 2994212583454668688L;

268            private final List<XQExpression> _srcFilters;
269            private final List<XQExpression> _pkeyExprs;
270            private final List<Type> _keyTypes;
271            private final String _opr;

278            public PromoteJoinExpression(XQExpression srcExpr, List<XQExpression> srcFilter, List<XQExpression> persistKeyExprs, String opr, List<XQExpression> searchKeyExprs) {
279                if(opr == null) {
280                    throw new IllegalArgumentException("Operator is not set");
281                }
282                if(persistKeyExprs.size() != searchKeyExprs.size()) {
283                    throw new IllegalArgumentException("Keys length is different.\npersistentKeyExpr: "
284                            + persistKeyExprs + ", searchKeys: " + searchKeyExprs);
285                }
286                this._srcExpr = srcExpr;
287                this._srcFilters = srcFilter;
288                this._opr = opr;
289                this._pkeyExprs = persistKeyExprs;
290                this._skeyExprs = searchKeyExprs;
291                final List<Type> keyTypes = new ArrayList<Type>(persistKeyExprs.size());
292                for(XQExpression keyexpr : persistKeyExprs) {
293                    final Type t = keyexpr.getType();
294                    if(!(t instanceof NodeType) && !(t instanceof AtomicType)) {
295                        keyTypes.add(StringType.STRING);
296                    } else {
297                        keyTypes.add(t);
298                    }
299                }
300                this._keyTypes = keyTypes;
301                if(srcFilter.isEmpty()) {
302                    this._type = srcExpr.getType();
303                } else {
304                    final int last = srcFilter.size() - 1;
305                    this._type = srcFilter.get(last).getType();
306                }
307                this._isNodeComp = (opr == IS.sgn || opr == PRECEDES.sgn || opr == FOLLOWS.sgn);
308            }

332                final StringBuilder buf = new StringBuilder(128);
333                buf.append(_srcExpr.toString());
334                for(XQExpression filter : _srcFilters) {
335                    buf.append('[');
336                    buf.append(filter.toString());
337                    buf.append(']');
338                }
339                final int condsize = _pkeyExprs.size();
340                for(int i = 0; i < condsize; i++) {
341                    buf.append('[');
342                    buf.append(_pkeyExprs.get(i));
343                    buf.append(' ');
344                    buf.append(_opr);
345                    buf.append(' ');
346                    buf.append(_skeyExprs.get(i));
347                    buf.append(']');
348                }
349                return buf.toString();

361            public Sequence<? extends Item> eval(final Sequence<? extends Item> contextSeq, final DynamicContext dynEnv)
362                    throws XQueryException {
363                final Map<Integer, JoinTable> tables = _tables;
364                final int shippedId = _shippedId;
365                if(shippedId != -1) {
366                    this._joinTable = tables.get(shippedId);
367                }
368                if(_joinTable != null) {
369                    return map(contextSeq, dynEnv);
370                }
371                if(ENV_DISABLE_JOIN_THREADING) {
372                    final JoinTable joinTable = makeJoinTable(contextSeq, dynEnv);
373                    this._joinTable = joinTable;
374                    if(shippedId != -1) {
375                        tables.put(shippedId, joinTable);
376                    }
377                    return map(contextSeq, dynEnv);
378                }
379                // threaded execution
380                final ExecutorService ex = Executors.newSingleThreadExecutor();
381                final Callable<JoinTable> call = new Callable<JoinTable>() {
382                    public JoinTable call() throws Exception {
383                        final JoinTable joinTable = makeJoinTable(contextSeq, dynEnv);
384                        if(shippedId != -1) {
385                            tables.put(shippedId, joinTable);
386                        }
387                        return joinTable;
388                    }
389                };
390                final Future<JoinTable> future = ex.submit(call);
391                final Sequence<? extends Item> mapped;
392                try {
393                    mapped = map(contextSeq, dynEnv, future);
394                } catch (InterruptedException e) {
395                    ex.shutdownNow();
396                    throw new IllegalStateException(e);
397                } catch (ExecutionException e) {
398                    ex.shutdownNow();
399                    throw new IllegalStateException(e);
400                }
401                ex.shutdown(); // REVIEWME is really required?
402                return mapped;
403            }

405            protected JoinTable makeJoinTable(Sequence<? extends Item> contextSeq, DynamicContext dynEnv)
406                    throws XQueryException {
407                Sequence<? extends Item> src = _srcExpr.eval(contextSeq, dynEnv);
408                for(XQExpression f : _srcFilters) {
409                    src = f.eval(src, dynEnv);
410                }
411                final int keysize = _pkeyExprs.size();
412                assert (keysize > 0);
413                final JoinTable joinTable = new JoinTable(_opr, _type);
414                final IFocus<? extends Item> srcItor = src.iterator();
415                int counter = 0;
416                outer: for(Item it : srcItor) {
417                    final Comparable[] persistKeys = new Comparable[keysize];
418                    if(keysize == 1) {
419                        final XQExpression key = _pkeyExprs.get(0);
420                        final IFocus<? extends Item> pkvItor = key.eval(it, dynEnv).iterator();
421                        for(Item fpv : pkvItor) {
422                            if(_isNodeComp || fpv instanceof AtomicValue) {
423                                persistKeys[0] = fpv;
424                            } else {
425                                if(counter == 0) {
426                                    _keyTypes.set(0, StringType.STRING);
427                                }
428                                persistKeys[0] = new XString(fpv.stringValue());
429                            }
430                            joinTable.put(persistKeys, it);
431                        }
432                        pkvItor.closeQuietly();
433                    } else {// TODO FIXME multiple join keys    
434                        for(int i = 0; i < keysize; i++) {
435                            final XQExpression key = _pkeyExprs.get(i);
436                            final Sequence<? extends Item> pkv = key.eval(it, dynEnv);
437                            final Iterator<? extends Item> pkvItor = pkv.iterator();
438                            if(!pkvItor.hasNext()) {
439                                continue outer;
440                            }
441                            final Item fpv = pkvItor.next();
442                            if(pkvItor.hasNext()) {
443                                persistKeys[i] = new SingleCollection(pkv, dynEnv);
444                            } else if(_isNodeComp || fpv instanceof AtomicValue) {
445                                persistKeys[i] = fpv;
446                            } else {
447                                if(counter == 0) {
448                                    _keyTypes.set(i, StringType.STRING);
449                                }
450                                persistKeys[i] = new XString(fpv.stringValue());
451                            }
452                        }
453                        joinTable.put(persistKeys, it);
454                    }
455                    counter++;
456                }
457                srcItor.closeQuietly();
458                return joinTable;
459            }

461            protected ValueSequence map(Sequence<? extends Item> contextSeq, DynamicContext dynEnv)
462                    throws XQueryException {
463                final int searchKeyLen = _skeyExprs.size();
464                Iterator<? extends Item> skvItor = null;
465                for(int i = 0; i < searchKeyLen; i++) {
466                    final Sequence<? extends Item> skv = _skeyExprs.get(i).eval(contextSeq, dynEnv);
467                    if(skvItor == null) {
468                        skvItor = skv.iterator();
469                    } else {
470                        skvItor = new NestedIterator<Item>(skvItor, skv.iterator());
471                    }
472                }
473                JoinTable joinTable = _joinTable;
474                final ValueSequence vs = new ValueSequence(dynEnv);
475                while(skvItor.hasNext()) {
476                    final Comparable searchKey;
477                    if(searchKeyLen == 1) {
478                        final Item fsv = skvItor.next();
479                        searchKey = mapArgument(fsv, 0, dynEnv);
480                    } else {
481                        final Comparable[] searchKeys = new Comparable[searchKeyLen];
482                        for(int i = 0; i < searchKeyLen; i++) {
483                            Item fsv = skvItor.next();
484                            searchKeys[i] = mapArgument(fsv, i, dynEnv);
485                        }
486                        searchKey = new JoinTable.Key(searchKeys, _opr);
487                    }
488                    final IFocus<Item> jsItor = joinTable.entrySequence(searchKey, dynEnv).iterator();
489                    for(Item entry : jsItor) {
490                        vs.addItem(entry);
491                    }
492                    jsItor.closeQuietly();
493                }
494                if(vs.isEmpty()) {
495                    return ValueSequence.EMPTY_SEQUENCE;
496                }
497                return vs;
498            }

500            private Item mapArgument(Item item, int index, DynamicContext dynEnv)
501                    throws XQueryException {
502                if(_isNodeComp) {
503                    return item;
504                }
505                final Type t = _keyTypes.get(index);
506                if(t == StringType.STRING) {
507                    return new XString(item.stringValue());
508                }
509                final AtomicValue it = (item instanceof AtomicValue) ? (AtomicValue) item
510                        : new XString(item.stringValue());
511                if(t instanceof AtomicType) {
512                    final AtomicType trgType = (AtomicType) t;
513                    final AtomicValue converted = it.castAs(trgType, dynEnv);
514                    return converted;
515                }
516                return it;
517            }
518    
519            protected Sequence<? extends Item> map(Sequence<? extends Item> contextSeq, DynamicContext dynEnv, Future<JoinTable> future)
520                    throws XQueryException, InterruptedException, ExecutionException {
521                final int searchKeyLen = _skeyExprs.size();
522                Iterator<? extends Item> skvItor = null;
523                for(int i = 0; i < searchKeyLen; i++) {
524                    Sequence<? extends Item> skv = _skeyExprs.get(i).eval(contextSeq, dynEnv);
525                    final Iterator<? extends Item> itor = skv.iterator();//skv.materialize().iterator();
526                    if(skvItor == null) {
527                        skvItor = itor;
528                    } else {
529                        skvItor = new NestedIterator<Item>(skvItor, itor);
530                    }
531                }
532                final ValueSequence vs = new ValueSequence(dynEnv);
533                while(skvItor.hasNext()) {
534                    final Comparable searchKey;
535                    if(searchKeyLen == 1) {
536                        final Item fsv = skvItor.next();
537                        searchKey = mapArgument(fsv, 0, dynEnv);
538                    } else {
539                        final Comparable[] searchKeys = new Comparable[searchKeyLen];
540                        for(int i = 0; i < searchKeyLen; i++) {
541                            Item fsv = skvItor.next();
542                            searchKeys[i] = mapArgument(fsv, i, dynEnv);
543                        }
544                        searchKey = new JoinTable.Key(searchKeys, _opr);
545                    }
546                    if(_joinTable == null) {
547                        this._joinTable = future.get();
548                    }
549                    final IFocus<Item> js = _joinTable.entrySequence(searchKey, dynEnv).iterator();
550                    for(Item entry : js) {
551                        vs.addItem(entry);
552                    }
553                    js.closeQuietly();
554                }
555                if(vs.isEmpty()) {
556                    return ValueSequence.EMPTY_SEQUENCE;
557                }
558                return vs;
559            }

562        private static final class JoinTable implements Serializable {
563            private static final long serialVersionUID = -1029891058586613017L;
564    
565            private final Map<Comparable, Entry> _table = new HashMap<Comparable, Entry>(32, 0.8f);
566            private final SortedSet<Comparable> _orderedKey;
567            private final Key _probe;
568            private final String _opr;
569            private final boolean _equiJoin;
570            private final Type _type;
571    
572            public JoinTable(String opr, Type type) {
573                if(opr == null || type == null) {
574                    throw new IllegalArgumentException();
575                }
576                this._opr = opr;
577                if(EQ.sgn.equals(opr) || IS.sgn.equals(opr)) {
578                    this._equiJoin = true;
579                    this._orderedKey = null;
580                } else {
581                    this._equiJoin = false;
582                    this._orderedKey = new TreeSet<Comparable>();
583                }
584                this._type = type;
585                this._probe = new Key(null, opr);
586            }

588            public static final class Key implements Comparable, Serializable {
589                private static final long serialVersionUID = -7951500027264400851L;
590    
591                private Comparable[] keys;
592                private final int[] expected;
593                private final int unexpected;
594    
595                public Key(Comparable[] keys, String opr) {
596                    this.keys = keys;
597                    if(EQ.sgn.equals(opr) || IS.sgn.equals(opr)) {
598                        this.expected = new int[] { 0 };
599                        this.unexpected = -1;
600                    } else if(NE.sgn.equals(opr)) {
601                        this.expected = new int[] { 1, -1 };
602                        this.unexpected = 0;
603                    } else if(GT.sgn.equals(opr) || FOLLOWS.sgn.equals(opr)) {
604                        this.expected = new int[] { 1 };
605                        this.unexpected = -1;
606                    } else if(LT.sgn.equals(opr) || PRECEDES.sgn.equals(opr)) {
607                        this.expected = new int[] { -1 };
608                        this.unexpected = 1;
609                    } else if(GE.sgn.equals(opr)) {
610                        this.expected = new int[] { 1, 0 };
611                        this.unexpected = -1;
612                    } else if(LE.sgn.equals(opr)) {
613                        this.expected = new int[] { 0, -1 };
614                        this.unexpected = 1;
615                    } else {
616                        throw new IllegalArgumentException("Illegal operator: " + opr);
617                    }
618                }

784            private Sequence<Item> entrySequence(Comparable key, DynamicContext dynEnv) {
785                if(_equiJoin) { // ==
786                    final Entry entry = find(key);
787                    if(entry == null) {
788                        return ValueSequence.EMPTY_SEQUENCE;
789                    }
790                    Item[] items = entry.getItems();
791                    return new ValueSequence(Arrays.asList(items), dynEnv);
792                }
793                assert (_orderedKey != null);
794                final SortedSet<Comparable> sorted;
795                if(_opr.indexOf('<') != -1) { // <
796                    SortedSet<Comparable> heads = _orderedKey.headSet(key); // note that headSet contains key itself.
797                    if(_opr.indexOf('=') == -1) {
798                        heads.remove(key);
799                    }
800                    sorted = heads;
801                } else if(_opr.indexOf('>') != -1) { // > 
802                    SortedSet<Comparable> tails = _orderedKey.tailSet(key); // note that tailSet doesn't contain key itself.
803                    if(_opr.indexOf('=') != -1) {
804                        tails.add(key);
805                    }
806                    sorted = tails;
807                } else if(NE.sgn.equals(_opr)) { // !=
808                    SortedSet<Comparable> excepts = new TreeSet<Comparable>(_orderedKey);
809                    excepts.remove(key);
810                    sorted = excepts;
811                } else {
812                    throw new IllegalStateException("Illegal operator: " + _opr);
813                }
814                if(sorted.isEmpty()) {
815                    return ValueSequence.EMPTY_SEQUENCE;
816                }
817                return new JoinEntrySequence(_table, sorted, dynEnv, _type);
818            }
