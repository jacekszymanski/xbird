103        /** create index resources and close it. */
104        public boolean create() throws DbException {
105            return create(true);
106        }
107    
108        public boolean create(boolean close) throws DbException {
109            ensureResourceOpen();
110            try {
111                _fileHeader.write();
112            } catch (IOException e) {
113                throw new DbException(e);
114            }
115            if(close) {
116                close();
117            } else {
118                this._opened = true;
119            }
120            return true;
121        }
122    
123        public boolean open() throws DbException {
124            ensureResourceOpen();
125            if(exists()) {
126                try {
127                    _fileHeader.read();
128                } catch (IOException e) {
129                    throw new DbException(e);
130                }
131                this._opened = true;
132                return true;
133            } else {
134                this._opened = false;
135                return false;
136            }
137        }
138    
139        protected final RandomAccessFile ensureResourceOpen() throws DbException {
140            if(_raf == null) {
141                try {
142                    this._raf = new RandomAccessFile(_file, "rw");
143                } catch (FileNotFoundException e) {
144                    throw new DbException(e);
145                }
146            }
147            if(_fc == null) {
148                this._fc = _raf.getChannel();
149            }
150            return _raf;
151        }
152    
153        public synchronized boolean close() throws DbException {
154            if(_opened) {
155                this._opened = false;
156                // close resources
157                try {
158                    _raf.close();
159                    _fc.close();
160                } catch (IOException e) {
161                    throw new DbException(e);
162                }
163                reset();
164                return true;
165            } else {
166                return false;
167            }
168        }
169    
170        protected final void checkOpened() throws DbException {
171            if(!_opened) {
172                throw new DbException("Not opened");
173            }
174        }
175    
176        private final void reset() {
177            this._raf = null;
178            this._fc = null;
179        }
180    
181        public boolean drop() throws DbException {
182            close();
183            if(exists()) {
184                return getFile().delete();
185            } else {
186                return true;
187            }
188        }
189    
190        public final boolean exists() {
191            return _file.exists();
192        }
193    
194        public void flush() throws DbException {
195            try {
196                if(_fileHeader._fhDirty) {
197                    _fileHeader.write();
198                }
199                _fc.force(true);
200            } catch (IOException e) {
201                throw new DbException(e);
202            }
203        }

219        /**
220         * getPage returns the page specified by pageNum.
221         */
222        protected final Page getPage(long pageNum) throws DbException {
223            Page p = null;
224            synchronized(this) {
225                // if not check if it's already loaded in the page cache
226                Reference<Page> ref = _pages.get(pageNum); // Check if required page is in the volatile cache
227                if(ref != null) {
228                    p = ref.get();
229                }
230                if(p == null) {
231                    // if still not found we need to create it and add it to the page cache.
232                    p = new Page(pageNum);
233                    try {
234                        p.read(); // Load the page from disk if necessary
235                    } catch (IOException e) {
236                        throw new DbException(e);
237                    }
238                    _pages.put(pageNum, new WeakReference<Page>(p));
239                }
240            }
241            return p;
242        }
243    
244        /**
245         * getFreePage returns the first free Page from secondary storage.
246         * If no Pages are available, the file is grown as appropriate.
247         */
248        protected final Page getFreePage() throws DbException {
249            Page p = null;
250            // Synchronize read and write to the fileHeader.firstFreePage
251            synchronized(_fileHeader) {
252                if(_fileHeader._firstFreePage != NO_PAGE) {
253                    // Steal a deleted page
254                    p = getPage(_fileHeader._firstFreePage);
255                    _fileHeader.setFirstFreePage(p._pageHeader._nextPage);
256                    if(_fileHeader._firstFreePage == NO_PAGE) {
257                        _fileHeader.setLastFreePage(NO_PAGE);
258                    }
259                }
260            }
261            if(p == null) { // No deleted pages, grow the file
262                p = getPage(_fileHeader.incrTotalPageCount());
263            }
264            p.initPage(); // Initialize The Page Header (Cleanly)
265            return p;
266        }
267    
268        /**
269         * unlinkPages unlinks a set of pages starting at the specified
270         * page number.
271         */
272        protected final void unlinkPages(long pageNum) throws DbException {
273            unlinkPages(getPage(pageNum));
274        }
275    
276        /**
277         * unlinkPages unlinks a set of pages starting at the specified Page.
278         */
279        protected final void unlinkPages(Page page) throws DbException {
280            Page nextPage = page;
281            if(nextPage != null) {
282                // Walk the chain and add it to the unused list
283                long firstPage = nextPage._pageNum;
284                long nextPageNum = nextPage.getPageHeader().getNextPage();
285                while(nextPageNum != NO_PAGE) {
286                    nextPage = getPage(nextPageNum);
287                    nextPageNum = nextPage.getPageHeader().getNextPage();
288                }
289                long lastPage = nextPage.getPageNum();
290                // Free the chain
291                synchronized(_fileHeader) {
292                    if(_fileHeader._lastFreePage != NO_PAGE) {
293                        Page p = getPage(_fileHeader._lastFreePage);
294                        p._pageHeader.setNextPage(firstPage);
295                        p.write();
296                    }
297                    if(_fileHeader._firstFreePage == NO_PAGE) {
298                        _fileHeader.setFirstFreePage(firstPage);
299                    }
300                    _fileHeader.setLastFreePage(lastPage);
301                }
302            }
303        }

305        /**
306         * writeValue writes the multi-Paged Value starting at the specified
307         * Page.
308         *
309         * @param page The starting Page
310         * @param value The Value to write
311         */
312        public final void writeValue(Page page, Value value) throws DbException {
313            InputStream is = value.getInputStream();
314    
315            // Write as much as we can onto the primary page.
316            PageHeader hdr = page.getPageHeader();
317            hdr.setRecordLength(value.getLength());
318            try {
319                page.readData(is);
320            } catch (IOException e) {
321                throw new DbException(e);
322            }
323    
324            // Write out the rest of the value onto any needed overflow pages
325            Page lastPage = page;
326            while(true) {
327                final int available;
328                try {
329                    available = is.available();
330                } catch (IOException e) {
331                    throw new DbException(e);
332                }
333                if(available == 0) {
334                    break;
335                }
336                LOG.debug("page overflowed");
337    
338                Page lpage = lastPage;
339                PageHeader lhdr = hdr;
340    
341                // Find an overflow page to use
342                long np = lhdr.getNextPage();
343                if(np != NO_PAGE) {
344                    // Use an existing page
345                    lastPage = getPage(np);
346                } else {
347                    // Create a new overflow page
348                    lastPage = getFreePage();
349                    lhdr.setNextPage(lastPage.getPageNum());
350                }
351    
352                // Mark the page as an overflow page
353                hdr = lastPage.getPageHeader();
354                hdr.setStatus(OVERFLOW);
355    
356                // Write some more of the value to the overflow page
357                try {
358                    lastPage.readData(is);
359                } catch (IOException e) {
360                    throw new DbException(e);
361                }
362                lpage.write();
363            }
364    
365            // Cleanup any unused overflow pages. i.e. the value is smaller then the
366            // last time it was written.
367            long np = hdr.getNextPage();
368            if(np != NO_PAGE) {
369                unlinkPages(np);
370            }
371            hdr.setNextPage(NO_PAGE);
372            lastPage.write();
373        }

393        /**
394         * readValue reads the multi-Paged Value starting at the specified
395         * Page.
396         *
397         * @param page The starting Page
398         * @return The Value
399         */
400        public final Value readValue(Page page) throws DbException {
401            PageHeader sph = page.getPageHeader();
402            FastMultiByteArrayOutputStream bos = new FastMultiByteArrayOutputStream(sph.getRecordLength());
403    
404            // Loop until we've read all the pages into memory
405            Page p = page;
406            while(true) {
407                // Add the contents of the page onto the stream
408                try {
409                    p.writeData(bos);
410                } catch (IOException e) {
411                    throw new DbException(e);
412                }
413                // Continue following the list of pages until we get to the end
414                PageHeader ph = p.getPageHeader();
415                long nextPage = ph.getNextPage();
416                if(nextPage == NO_PAGE) {
417                    break;
418                }
419                p = getPage(nextPage);
420            }
421            // Return a Value with the collected contents of all pages
422            return new Value(bos.toByteArray());
423        }

441        public abstract class FileHeader {
442    
443            private boolean _fhDirty = true;
444            private int _workSize;
445    
446            //--------------------------------------------
447            // persistant entry
448    
449            /** The size of the <code>FileHeader</code>. Usually 1 OS page (4096 byte). */
450            private short _fhSize;
451            /** The size of a <code>Page</code> */
452            private int _pageSize;
453            /** The number of total pages in the file */
454            private long _totalPageCount;
455            /** The first free page in unused secondary space */
456            private long _firstFreePage = NO_PAGE;
457            /** The last free page in unused secondary space */
458            private long _lastFreePage = NO_PAGE;
459            /** The size of the <code>PageHeader</code> */
460            private byte _pageHeaderSize = PageHeader.DEFAULT_PAGE_HEADER_SIZE;
461    
462            //--------------------------------------------
463    
464            public FileHeader(int pageSize) {
465                this._pageSize = pageSize;
466                this._fhSize = (short) 4096;
467                this._workSize = calculateWorkSize();
468            }
469    
470            public synchronized final void write() throws IOException {
471                if(!_fhDirty) {
472                    return;
473                }
474                synchronized(_raf) {
475                    _raf.seek(0);
476                    write(_raf);
477                }
478                if(LOG.isDebugEnabled()) {
479                    LOG.debug("wrote file header");
480                }
481                this._fhDirty = false;
482            }
483    
484            protected synchronized void write(RandomAccessFile raf) throws IOException {
485                raf.writeShort(_fhSize);
486                raf.writeInt(_pageSize);
487                raf.writeLong(_totalPageCount);
488                raf.writeLong(_firstFreePage);
489                raf.writeLong(_lastFreePage);
490                raf.writeByte(_pageHeaderSize);
491            }
492    
493            public synchronized final void read() throws IOException {
494                synchronized(_raf) {
495                    _raf.seek(0);
496                    read(_raf);
497                    this._workSize = calculateWorkSize();
498                }
499            }
500    
501            protected synchronized void read(RandomAccessFile raf) throws IOException {
502                this._fhSize = raf.readShort();
503                this._pageSize = raf.readInt();
504                this._totalPageCount = raf.readLong();
505                this._firstFreePage = raf.readLong();
506                this._lastFreePage = raf.readLong();
507                this._pageHeaderSize = raf.readByte();
508            }
509    
510            //--------------------------------------------
511    
512            public synchronized final void setFirstFreePage(long page) {
513                this._firstFreePage = page;
514                this._fhDirty = true;
515            }
516    
517            public long getFirstFreePage() {
518                return _firstFreePage;
519            }
520    
521            public synchronized final void setLastFreePage(long page) {
522                this._lastFreePage = page;
523                this._fhDirty = true;
524            }
525    
526            public long getLastFreePage() {
527                return _lastFreePage;
528            }
529    
530            public synchronized final long incrTotalPageCount() {
531                this._fhDirty = true;
532                return _totalPageCount++;
533            }
534    
535            public synchronized final void setDirty(boolean dirty) {
536                this._fhDirty = dirty;
537            }
538    
539            public synchronized void setTotalPageCount(long pageCount) {
540                this._fhDirty = true;
541                this._totalPageCount = pageCount;
542            }
543    
544            public synchronized final long getTotalPageCount() {
545                return _totalPageCount;
546            }
547    
548            public final int getPageSize() {
549                return _pageSize;
550            }
551    
552            public final int getWorkSize() {
553                return _workSize;
554            }
555    
556            private final int calculateWorkSize() {
557                return _pageSize - _pageHeaderSize;
558            }
559        }

561        public static abstract class PageHeader {
562    
563            public static final int DEFAULT_PAGE_HEADER_SIZE = 64;
564    
565            //--------------------------------------------
566            // persistant entry        
567    
568            private byte _status = UNUSED;
569            private int _dataLen;
570            private int _recordLen;
571            private long _nextPage = NO_PAGE;
572    
573            //--------------------------------------------
574    
575            public PageHeader() {}
576    
577            public PageHeader(ByteBuffer buf) {
578                read(buf);
579            }
580    
581            public synchronized void read(ByteBuffer buf) {
582                this._status = buf.get();
583                if(_status == UNUSED) {
584                    return;
585                }
586                this._dataLen = buf.getInt();
587                this._recordLen = buf.getInt();
588                this._nextPage = buf.getLong();
589            }
590    
591            public synchronized void write(ByteBuffer buf) {
592                buf.put(_status);
593                buf.putInt(_dataLen);
594                buf.putInt(_recordLen);
595                buf.putLong(_nextPage);
596            }
597    
598            //--------------------------------------------
599            // getter, setter
600    
601            /** The status of this page (UNUSED, RECORD, DELETED, etc...) */
602            public final void setStatus(byte status) {
603                this._status = status;
604            }
605    
606            /** The status of this page (UNUSED, RECORD, DELETED, etc...) */
607            public final byte getStatus() {
608                return _status;
609            }
610    
611            /** The next page for this Record (if overflowed) */
612            public final void setNextPage(long nextPage) {
613                this._nextPage = nextPage;
614            }
615    
616            /** The next page for this Record (if overflowed) */
617            public final long getNextPage() {
618                return _nextPage;
619            }
620    
621            /** The length of the Data */
622            public final void setDataLength(int dataLen) {
623                this._dataLen = dataLen;
624            }
625    
626            public final int getDataLength() {
627                return _dataLen;
628            }
629    
630            public final void setRecordLength(int length) {
631                this._recordLen = length;
632            }
633    
634            public final int getRecordLength() {
635                return _recordLen;
636            }
637        }

639        public final class Page implements Comparable {
640    
641            private final long _pageNum;
642            private final PageHeader _pageHeader;
643            /** The offset into the file that this page starts */
644            private final long _pageOffset;
645    
646            /** The data for this page */
647            private ByteBuffer _pageData = null;
648            /** The position (relative) of the Data in the data array */
649            private int _dataPos;
650    
651            public Page(long pageNum) {
652                this._pageNum = pageNum;
653                this._pageHeader = createPageHeader();
654                this._pageOffset = _fileHeader._fhSize + (pageNum * _fileHeader._pageSize);
655            }
656    
657            public synchronized void read() throws IOException {
658                if(_pageData == null) {
659                    if(LOG.isDebugEnabled()) {
660                        LOG.debug("read in page#" + _pageNum + " from page offset " + _pageOffset);
661                    }
662                    byte[] buf = new byte[_fileHeader._pageSize];
663                    _raf.seek(_pageOffset);
664                    _raf.read(buf);
665                    this._pageData = ByteBuffer.wrap(buf);
666                    _pageHeader.read(_pageData);
667                    this._dataPos = _fileHeader._pageHeaderSize;
668                }
669            }
670    
671            public void write() throws DbException {
672                synchronized(this) {
673                    _pageData.rewind();
674                    _pageHeader.write(_pageData);
675                }
676                synchronized(_raf) {
677                    try {
678                        _raf.seek(_pageOffset);
679                        _raf.write(_pageData.array());
680                    } catch (IOException e) {
681                        throw new DbException(e);
682                    }
683                }
684            }
685    
686            /**
687             * Flushes content of the dirty page into the file
688             */
689            public synchronized void flush() throws IOException {
690                if(LOG.isDebugEnabled()) {
691                    LOG.debug("write out page#" + _pageNum + " to page offset " + _pageOffset);
692                }
693                _raf.seek(_pageOffset);
694                _raf.write(_pageData.array());
695            }
696    
697            public synchronized void writeData(OutputStream os) throws IOException {
698                if(_pageHeader._dataLen > 0) {
699                    byte[] b = new byte[_pageHeader._dataLen];
700                    _pageData.position(_dataPos);
701                    _pageData.get(b);
702                    os.write(b);
703                }
704            }
705    
706            public synchronized void readData(InputStream is) throws IOException {
707                // set data length in the page header
708                int avail = is.available();
709                int datalen = _fileHeader._workSize;
710                if(avail < datalen) {
711                    datalen = avail;
712                }
713                _pageHeader.setDataLength(datalen);
714                // read data from stream
715                if(datalen > 0) {
716                    byte[] b = new byte[datalen];
717                    is.read(b);
718                    _pageData.position(getDataPos());
719                    _pageData.put(b);
720                }
721            }
722    
723            public PageHeader getPageHeader() {
724                return _pageHeader;
725            }
726    
727            public long getPageNum() {
728                return _pageNum;
729            }
730    
731            private int getDataPos() {
732                return _dataPos;
733            }
734    
735            protected void initPage() {
736                _pageHeader.setNextPage(NO_PAGE);
737                _pageHeader.setStatus(UNUSED);
738            }
739    
740            public int compareTo(Object o) {
741                return (int) (_pageNum - ((Page) o)._pageNum);
742            }
743    
744            @Override
745            public String toString() {
746                return "page#" + _pageNum;
747            }
748    
749        }
