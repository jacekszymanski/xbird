052    public final class VarSegments implements Segments {
053        private static final Log LOG = LogFactory.getLog(VarSegments.class);
054        private static final String ENV_DESCRIPTOR_TYPE = System.getProperty("xbird.descriotor.type", "hash");
055    
056        public static final int DEFAULT_BLOCK_CACHE_SIZE = 256;
057        private static final int INTEGER_BYTES = Integer.SIZE / 8;
058    
059        private static final String DESCRIPTOR_APPENDIX_HASH = ".di_h";
060        private static final String DESCRIPTOR_APPENDIX_RAF = ".di_f";
061        private static final String DESCRIPTOR_APPENDIX_BTREE = ".di_b";
062    
063        private final File file;
064        private final IDescriptor directory;
065    
066        private RandomAccessFile raf = null;
067        private long recorded = 8;
068    
069        private transient boolean open = false;
070    
071        public VarSegments(File file) {
072            this(file, DEFAULT_BLOCK_CACHE_SIZE);
073        }
074    
075        public VarSegments(File file, int cacheSize) {
076            this.file = file;
077            try {
078                if(file.exists() && file.length() > 0) {
079                    this.recorded = loadRecoreded();
080                }
081                this.directory = initDescriptor(file, cacheSize >>> 2);
082            } catch (IOException e) {
083                throw new IllegalStateException("preparing descriptor failed", e);
084            }
085        }
086    
087        public VarSegments(File file, DescriptorType type) {
088            this(file, DEFAULT_BLOCK_CACHE_SIZE, type);
089        }
090    
091        public VarSegments(File file, int cacheSize, DescriptorType type) {
092            this.file = file;
093            if(file.exists()) {
094                try {
095                    this.recorded = loadRecoreded();
096                } catch (IOException e) {
097                    throw new IllegalStateException(e);
098                }
099            }
100            final IDescriptor desc;
101            switch(type) {
102                case hash:
103                    desc = new DescriptorHash(file, cacheSize);
104                    break;
105                case btree:
106                    desc = new DescriptorBTree(file, cacheSize);
107                    break;
108                case file:
109                    try {
110                        desc = new DescriptorRAF(file, cacheSize);
111                    } catch (IOException e) {
112                        throw new IllegalStateException(e);
113                    }
114                    break;
115                default:
116                    throw new IllegalStateException("Illegal type: " + type);
117            }
118            this.directory = desc;
119        }
120    
121        private long loadRecoreded() throws IOException {
122            ensureOpen();
123            raf.seek(0);
124            long recorded = raf.readLong();
125            return recorded;
126        }
127    
128        public static IDescriptor initDescriptor(File baseFile) throws IOException {
129            return initDescriptor(baseFile, DEFAULT_BLOCK_CACHE_SIZE >>> 2);
130        }
131    
132        public static IDescriptor initDescriptor(File baseFile, int cacheSize) throws IOException {
133            String baseFileName = baseFile.getName();
134            File probe = new File(baseFile.getParent(), baseFileName + DESCRIPTOR_APPENDIX_BTREE);
135            if(probe.exists() || ENV_DESCRIPTOR_TYPE.equals("btree")) {
136                return new DescriptorBTree(baseFile, cacheSize);
137            }
138            probe = new File(baseFile.getParent(), baseFileName + DESCRIPTOR_APPENDIX_RAF);
139            if(probe.exists() || ENV_DESCRIPTOR_TYPE.equals("file")) {
140                return new DescriptorRAF(baseFile, cacheSize);
141            }
142            return new DescriptorHash(baseFile, cacheSize);
143        }
144    
145        public File getFile() {
146            return file;
147        }

149        public long write(final long idx, final byte[] b) throws IOException {
150            ensureOpen();
151            final long prevAddr = directory.getRecordAddr(idx);
152            if(prevAddr != -1L && prevAddr < recorded) {
153                synchronized(raf) {
154                    raf.seek(prevAddr);
155                    int prevLength = raf.readInt();
156                    if(b.length == prevLength) {
157                        raf.seek(prevAddr + 4);
158                        raf.write(b);
159                        return prevAddr;
160                    }
161                }
162            }
163            final long ptr = recorded;
164            // write to disk
165            synchronized(raf) {//insert
166                raf.seek(ptr);
167                raf.writeInt(b.length);
168                raf.write(b);
169            }
170            directory.addRecord(idx, ptr);
171            this.recorded = ptr + INTEGER_BYTES + b.length;
172            return ptr;
173        }
174    
175        public long writeIfAbsent(final long idx, final byte[] b) throws IOException {
176            ensureOpen();
177            final long prevAddr = directory.getRecordAddr(idx);
178            if(prevAddr != -1L && prevAddr < recorded) {
179                return -1L;
180            }
181            final long ptr = recorded;
182            // write to disk
183            synchronized(raf) {//insert
184                raf.seek(ptr);
185                raf.writeInt(b.length);
186                raf.write(b);
187            }
188            directory.addRecord(idx, ptr);
189            this.recorded = ptr + INTEGER_BYTES + b.length;
190            return ptr;
191        }
192    
193        /**
194         * reads persistent record
195         */
196        public byte[] read(final long idx) throws IOException {
197            final long ptr = directory.getRecordAddr(idx);
198            if(ptr == -1) {
199                return null;
200            }
201            return directRead(ptr);
202        }
203    
204        public byte[][] readv(long[] idx) throws IOException {
205            final int len = idx.length;
206            final byte[][] pages = new byte[len][];
207            for(int i = 0; i < len; i++) {
208                pages[i] = read(idx[i]);
209            }
210            return pages;
211        }
212    
213        private byte[] directRead(final long addr) throws IOException {
214            if(LOG.isDebugEnabled()) {
215                LOG.debug("paged in document segment of physical addr #" + addr + " from disk");
216            }
217            ensureOpen();
218            final byte[] b;
219            synchronized(raf) {
220                raf.seek(addr);
221                int len = raf.readInt();
222                b = new byte[len];
223                raf.read(b);
224            }
225            return b;
226        }
227    
228        private void ensureOpen() throws IOException {
229            if(open) {
230                return;
231            }
232            synchronized(this) {
233                if(!open) {
234                    if(raf == null) {
235                        if(!file.exists()) {
236                            boolean created = file.createNewFile();
237                            if(!created) {
238                                throw new IllegalStateException("could'nt create file: "
239                                        + file.getAbsolutePath());
240                            }
241                        }
242                        this.raf = new RandomAccessFile(file, "rw");
243                    }
244                    this.open = true;
245                }
246            }
247        }
248    
249        public void close() throws IOException {
250            if(raf != null) {
251                raf.close();
252            }
253            directory.close();
254            this.open = false;
255        }
256    
257        public void delete() throws IOException {
258            close();
259            if(!file.delete()) {
260                LOG.warn("delete file failed: " + file.getAbsolutePath());
261            }
262            directory.delete();
263        }
264    
265        public synchronized void flush(boolean close) throws IOException {
266            if(!open) {
267                throw new IllegalStateException("not opened: " + file.getAbsolutePath());
268            }
269            directory.flush(close);
270            syncRecordLength();
271            raf.getChannel().force(true);
272            if(close) {
273                close();
274            }
275        }
276    
277        private void syncRecordLength() throws IOException {
278            raf.seek(0);
279            raf.writeLong(recorded);
280        }

282        public interface IDescriptor {
283    
284            public void addRecord(final long idx, final long addr) throws IOException;
285    
286            public long getRecordAddr(final long idx) throws IOException;
287    
288            public void flush(boolean close) throws IOException;
289    
290            public void close() throws IOException;
291    
292            public void delete() throws IOException;
293    
294        }

296        public enum DescriptorType {
297            hash, btree, file
298        }

300        private static final class DescriptorHash implements IDescriptor {
301            private Long2LongOpenHash recordMap;
302            private final File descFile;
303    
304            public DescriptorHash(File baseFile, int cacheSize) {
305                final String descName = baseFile.getName() + DESCRIPTOR_APPENDIX_HASH;
306                final File file = new File(baseFile.getParent(), descName);
307                if(file.exists()) {
308                    final FileInputStream fis;
309                    try {
310                        fis = new FileInputStream(file);
311                    } catch (FileNotFoundException e) {
312                        throw new IllegalStateException();
313                    }
314                    final FastBufferedInputStream bis = new FastBufferedInputStream(fis, 4096);
315                    this.recordMap = ObjectUtils.readObject(bis);
316                    IOUtils.closeQuietly(bis);
317                } else {
318                    this.recordMap = new Long2LongOpenHash(cacheSize, 0.7f, 1.9f);
319                }
320                this.descFile = file;
321            }
322    
323            public void addRecord(final long idx, final long addr) {
324                recordMap.put(idx, addr);
325            }
326    
327            public long getRecordAddr(final long idx) {
328                return recordMap.get(idx);
329            }
330    
331            public void close() throws IOException {
332                this.recordMap = null;
333            }
334    
335            public void delete() throws IOException {
336                close();
337                if(!descFile.delete()) {
338                    LOG.warn("delete file failed: " + descFile.getAbsolutePath());
339                }
340            }
341    
342            public void flush(boolean close) throws IOException {
343                FileOutputStream fos = new FileOutputStream(descFile, false);
344                FastBufferedOutputStream bos = new FastBufferedOutputStream(fos, 8192);
345                ObjectUtils.toStream(recordMap, bos);
346                bos.flush();
347                bos.close();
348    
349                if(close) {
350                    close();
351                }
352            }
353        }

355        private static final class DescriptorBTree implements IDescriptor {
356    
357            private final File descFile;
358            private final BTree btree;
359    
360            public DescriptorBTree(File baseFile, int cacheSize) {
361                String descName = baseFile.getName() + DESCRIPTOR_APPENDIX_BTREE;
362                File descFile = new File(baseFile.getParent(), descName);
363                this.descFile = descFile;
364                BTree tree = new BTree(descFile, 2048, cacheSize, false);
365                try {
366                    tree.init(false);
367                } catch (DbException e) {
368                    throw new IllegalStateException("failed on initializing b+-tree: "
369                            + descFile.getAbsolutePath(), e);
370                }
371                this.btree = tree;
372            }
373    
374            public void addRecord(long idx, long addr) throws IOException {
375                try {
376                    btree.addValue(new Value(idx), addr);
377                } catch (DbException e) {
378                    throw new IOException("failed to add a record addr#" + addr + " to the idx#" + idx);
379                }
380            }
381    
382            public long getRecordAddr(long idx) throws IOException {
383                try {
384                    return btree.findValue(new Value(idx));
385                } catch (DbException e) {
386                    throw new IOException("failed to get a record of idx#" + idx);
387                }
388            }
389    
390            public void close() throws IOException {
391                try {
392                    btree.close();
393                } catch (DbException e) {
394                    throw new IOException("failed to close: " + btree.getFile().getAbsolutePath());
395                }
396            }
397    
398            public void flush(boolean close) throws IOException {
399                try {
400                    btree.flush(true, true);
401                    if(close) {
402                        btree.close();
403                    }
404                } catch (DbException e) {
405                    throw new IOException("failed to flush: " + btree.getFile().getAbsolutePath());
406                }
407            }
408    
409            public void delete() throws IOException {
410                close();
411                if(!descFile.delete()) {
412                    LOG.warn("delete file failed: " + descFile.getAbsolutePath());
413                }
414            }
415        }

417        /** key-address directory */
418        @Deprecated
419        private static final class DescriptorRAF implements IDescriptor {
420    
421            private final File descFile;
422            private final RandomAccessFile raf;
423            private final int purgeUnits;
424    
425            private SortedMap<Long, Long> addrCache = new TreeMap<Long, Long>();
426    
427            public DescriptorRAF(File baseFile, int cacheSize) throws IOException {
428                String descName = baseFile.getName() + DESCRIPTOR_APPENDIX_RAF;
429                File descFile = new File(baseFile.getParent(), descName);
430                if(!descFile.exists()) {
431                    boolean created = descFile.createNewFile();
432                    if(!created) {
433                        throw new IllegalStateException("create file failed: "
434                                + descFile.getAbsolutePath());
435                    }
436                }
437                this.descFile = descFile;
438                this.raf = new RandomAccessFile(descFile, "rw");
439                this.purgeUnits = cacheSize;
440            }
441    
442            public void addRecord(final long idx, final long addr) throws IOException {
443                if(idx < 0) {
444                    throw new IllegalArgumentException("Illegal idx: " + idx);
445                }
446                if(addrCache.size() >= purgeUnits) {
447                    purge();
448                }
449                addrCache.put(idx, addr);
450                //TODO update handling
451            }
452    
453            public long getRecordAddr(final long idx) throws IOException {
454                if(idx < 0) {
455                    throw new IllegalArgumentException("Illegal idx: " + idx);
456                }
457                Long laddr = addrCache.get(idx);
458                if(laddr != null) {
459                    return laddr.longValue();
460                }
461                long offset = idx * 8;
462                if(offset >= raf.length()) {
463                    return -1;// not found
464                }
465                final long addr;
466                synchronized(raf) {
467                    raf.seek(offset);
468                    addr = raf.readLong();
469                }
470                return addr;
471            }
472    
473            private synchronized void purge() throws IOException {
474                long prevIdx = -1;
475                for(Entry<Long, Long> e : addrCache.entrySet()) {
476                    final long idx = e.getKey();
477                    final long addr = e.getValue();
478                    synchronized(raf) {
479                        if(idx != (prevIdx + 8)) {
480                            final long offset = idx * 8;
481                            raf.seek(offset);
482                        }
483                        raf.writeLong(addr);
484                    }
485                    prevIdx = idx;
486                }
487                addrCache.clear();
488            }
489    
490            public void flush(boolean close) throws IOException {
491                if(!addrCache.isEmpty()) {
492                    purge();
493                }
494                raf.getChannel().force(true);
495                if(close) {
496                    raf.close();
497                }
498            }
499    
500            public void close() throws IOException {
501                addrCache = null;
502                raf.close();
503            }
504    
505            public void delete() throws IOException {
506                close();
507                if(!descFile.delete()) {
508                    LOG.warn("delete file failed: " + descFile.getAbsolutePath());
509                }
510            }
511        }
