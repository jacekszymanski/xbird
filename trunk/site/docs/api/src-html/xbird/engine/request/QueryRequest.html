118        /**
119         * The default fetch method is {@code FetchMethod#bytes}.
120         * {@code FetchMethod#asyncStream} is not recommended.
121         * 
122         * @see RemoteFocusProxy
123         * @see RemoteFocus
124         */
125        public enum FetchMethod {
126            bytes, compressed_bytes, syncStream, @Deprecated
127            asyncStream;
128    
129            FetchMethod() {}
130    
131            public static FetchMethod appropriateMethod() {
132                final String m = Settings.get("xbird.remote.fetch_method", "bytes");
133                if("bytes".equals(m)) {
134                    return bytes;
135                } else if("compressed_bytes".equals(m)) {
136                    return compressed_bytes;
137                } else if("syncStream".equals(m)) {
138                    return syncStream;
139                } else if("asyncStream".equals(m)) {
140                    return asyncStream;
141                } else {
142                    throw new IllegalStateException("Unsupported fetch method: " + m);
143                }
144            }
145    
146            public static FetchMethod readFrom(final ObjectInput in) throws IOException {
147                final int type = in.readByte();
148                switch(type) {
149                    case 1:
150                        return bytes;
151                    case 2:
152                        return compressed_bytes;
153                    case 3:
154                        return syncStream;
155                    case 4:
156                        return asyncStream;
157                    default:
158                        throw new IllegalStateException("Illegal pattern: " + type);
159                }
160            }
161    
162            public void writeTo(final ObjectOutput out) throws IOException {
163                switch(this) {
164                    case bytes:
165                        out.writeByte(1);
166                        break;
167                    case compressed_bytes:
168                        out.writeByte(2);
169                        break;
170                    case syncStream:
171                        out.writeByte(3);
172                        break;
173                    case asyncStream:
174                        out.writeByte(4);
175                        break;
176                    default:
177                        throw new IllegalStateException("Unexpected: " + this);
178                }
179            }
180        }

213        @Override
214        public void readExternal(ObjectInput in) throws IOException, ClassNotFoundException {
215            super.readExternal(in);
216            readQuery(in);
217            this._fetchMethod = FetchMethod.readFrom(in);
218            final int frag = in.readByte();
219            if(frag != 0) {
220                if((frag & 1) == 1) {
221                    this._baseUri = (URI) in.readObject();
222                }
223                if((frag & 2) == 2) {
224                    this._fetchSize = in.readInt();
225                }
226                if((frag & 4) == 4) {
227                    this._fetchGrow = in.readFloat();
228                }
229                if((frag & 8) == 8) {
230                    final int varsSize = in.readInt();
231                    final ShippedVariable[] vars = new ShippedVariable[varsSize];
232                    for(int i = 0; i < varsSize; i++) {
233                        vars[i] = ShippedVariable.readFrom(in);
234                    }
235                    this._varToShip = vars;
236                }
237            }
238        }
239    
240        protected void readQuery(ObjectInput in) throws IOException, ClassNotFoundException {
241            this._query = ObjectUtils.readString(in);
242        }
243    
244        @Override
245        public void writeExternal(ObjectOutput out) throws IOException {
246            super.writeExternal(out);
247            writeQuery(out);
248            _fetchMethod.writeTo(out);
249    
250            final URI baseUri = _baseUri;
251            final int fetchSize = _fetchSize;
252            final float fetchGrow = _fetchGrow;
253            final ShippedVariable[] varToShip = _varToShip;
254            int frag = 0;
255            if(baseUri != null) {
256                frag |= 1;
257            }
258            if(fetchSize != RemoteFocus.DEFAULT_FETCH_SIZE) {
259                frag |= 2;
260            }
261            if(fetchGrow != RemoteFocus.DEFAULT_FETCH_GROWFACTOR) {
262                frag |= 4;
263            }
264            if(varToShip != null) {
265                frag |= 8;
266            }
267            out.writeByte(frag);
268            if(frag != 0) {
269                if(baseUri != null) {
270                    out.writeObject(baseUri);
271                }
272                if(fetchSize != RemoteFocus.DEFAULT_FETCH_SIZE) {
273                    out.writeInt(fetchSize);
274                }
275                if(fetchGrow != RemoteFocus.DEFAULT_FETCH_GROWFACTOR) {
276                    out.writeFloat(fetchGrow);
277                }
278                if(varToShip != null) {
279                    final int varsSize = varToShip.length;
280                    out.writeInt(varsSize);
281                    for(int i = 0; i < varsSize; i++) {
282                        varToShip[i].writeExternal(out);
283                    }
284                }
285            }
286        }
287    
288        protected void writeQuery(ObjectOutput out) throws IOException {
289            assert (_query != null);
290            ObjectUtils.writeString(out, _query);
291        }
