044        public static byte[] toBytes(final char[] c, final int off, final int len) {
045            final byte[] b = new byte[len * 2];
046            for(int i = 0; i < len; i++) {
047                putChar(b, i << 1, c[off + i]);
048            }
049            return b;
050        }

069            final byte[] b = new byte[len * 8];
070            for(int i = 0; i < len; i++) {
071                putLong(b, i << 3, array[off + i]);
072            }
073            return b;
074        }

076        public static void putChar(final byte[] b, final int off, final char val) {
077            b[off + 1] = (byte) (val >>> 0);
078            b[off] = (byte) (val >>> 8);
079        }
080    
081        public static void putInt(byte[] b, int off, int val) {
082            b[off + 3] = (byte) (val >>> 0);
083            b[off + 2] = (byte) (val >>> 8);
084            b[off + 1] = (byte) (val >>> 16);
085            b[off] = (byte) (val >>> 24);
086        }
087    
088        public static void putLong(byte[] b, int off, long val) {
089            b[off + 7] = (byte) (val >>> 0);
090            b[off + 6] = (byte) (val >>> 8);
091            b[off + 5] = (byte) (val >>> 16);
092            b[off + 4] = (byte) (val >>> 24);
093            b[off + 3] = (byte) (val >>> 32);
094            b[off + 2] = (byte) (val >>> 40);
095            b[off + 1] = (byte) (val >>> 48);
096            b[off] = (byte) (val >>> 56);
097        }

099        public static byte[] toBytes(int v) {
100            final byte[] b = new byte[4];
101            b[0] = (byte) (v >>> 24);
102            b[1] = (byte) (v >>> 16);
103            b[2] = (byte) (v >>> 8);
104            b[3] = (byte) (v >>> 0);
105            return b;
106        }

109            final byte[] b = new byte[8];
110            b[0] = (byte) (v >>> 56);
111            b[1] = (byte) (v >>> 48);
112            b[2] = (byte) (v >>> 40);
113            b[3] = (byte) (v >>> 32);
114            b[4] = (byte) (v >>> 24);
115            b[5] = (byte) (v >>> 16);
116            b[6] = (byte) (v >>> 8);
117            b[7] = (byte) (v >>> 0);
118            return b;
119        }

125        public static char[] toChars(final byte[] b, final int off, final int len) {
126            final int clen = len >>> 1;
127            final char[] c = new char[clen];
128            for(int i = 0; i < clen; i++) {
129                final int j = off + (i << 1);
130                c[i] = (char) ((b[j + 1] & 0xFF) + ((b[j + 0]) << 8));
131            }
132            return c;
133        }

137            for(int i = 0; i < clen; i++) {
138                final int j = i << 1;
139                dest[i] = (char) ((b[j + 1] & 0xFF) + ((b[j + 0]) << 8));
140            }
141        }

147        public static int[] toInts(final byte[] b, final int off, final int len) {
148            final int clen = len >>> 2;
149            final int[] c = new int[clen];
150            for(int i = 0, bi = off; i < clen; i++, bi += 4) {
151                c[i] = (b[bi] << 24) + ((b[bi + 1] & 0xFF) << 16) + ((b[bi + 2] & 0xFF) << 8)
152                        + (b[bi + 3] & 0xFF);
153            }
154            return c;
155        }

157        public static int[] toIntsVectorized(final byte[] b, final int off, final int len) {
158            final int clen = len >>> 2;
159            final int[] c = new int[clen];
160            int i = 0, bi = off;
161            final int limit = clen - 7;
162            for(; i < limit; i += 8, bi += 32) {
163                c[i] = (b[bi] << 24) + ((b[bi + 1] & 0xFF) << 16) + ((b[bi + 2] & 0xFF) << 8)
164                        + (b[bi + 3] & 0xFF);
165                c[i + 1] = (b[bi + 4] << 24) + ((b[bi + 5] & 0xFF) << 16) + ((b[bi + 6] & 0xFF) << 8)
166                        + (b[bi + 7] & 0xFF);
167                c[i + 2] = (b[bi + 8] << 24) + ((b[bi + 9] & 0xFF) << 16) + ((b[bi + 10] & 0xFF) << 8)
168                        + (b[bi + 11] & 0xFF);
169                c[i + 3] = (b[bi + 12] << 24) + ((b[bi + 13] & 0xFF) << 16)
170                        + ((b[bi + 14] & 0xFF) << 8) + (b[bi + 15] & 0xFF);
171                c[i + 4] = (b[bi + 16] << 24) + ((b[bi + 17] & 0xFF) << 16)
172                        + ((b[bi + 18] & 0xFF) << 8) + (b[bi + 19] & 0xFF);
173                c[i + 5] = (b[bi + 20] << 24) + ((b[bi + 21] & 0xFF) << 16)
174                        + ((b[bi + 22] & 0xFF) << 8) + (b[bi + 23] & 0xFF);
175                c[i + 6] = (b[bi + 24] << 24) + ((b[bi + 25] & 0xFF) << 16)
176                        + ((b[bi + 26] & 0xFF) << 8) + (b[bi + 27] & 0xFF);
177                c[i + 7] = (b[bi + 28] << 24) + ((b[bi + 29] & 0xFF) << 16)
178                        + ((b[bi + 30] & 0xFF) << 8) + (b[bi + 31] & 0xFF);
179            }
180            for(; i < clen; i++, bi += 4) {
181                c[i] = (b[bi] << 24) + ((b[bi + 1] & 0xFF) << 16) + ((b[bi + 2] & 0xFF) << 8)
182                        + (b[bi + 3] & 0xFF);
183            }
184            return c;
185        }

191        public static long[] toLongs(final byte[] b, final int off, final int len) {
192            final int clen = len >>> 3;
193            final long[] c = new long[clen];
194            for(int i = 0, bi = off; i < clen; i++, bi += 8) {
195                c[i] = (b[bi + 7] & 0xFFL) + ((b[bi + 6] & 0xFFL) << 8) + ((b[bi + 5] & 0xFFL) << 16)
196                        + ((b[bi + 4] & 0xFFL) << 24) + ((b[bi + 3] & 0xFFL) << 32)
197                        + ((b[bi + 2] & 0xFFL) << 40) + ((b[bi + 1] & 0xFFL) << 48)
198                        + (((long) b[bi]) << 56);
199            }
200            return c;
201        }

251        public static int getInt(final byte[] b, final int off) {
252            return ((b[off + 3] & 0xFF) << 0) + ((b[off + 2] & 0xFF) << 8)
253                    + ((b[off + 1] & 0xFF) << 16) + ((b[off + 0]) << 24);
254        }

256        public static char getChar(final byte[] b, final int off) {
257            return (char) (((b[off + 1] & 0xFF) << 0) + ((b[off + 0]) << 8));
258        }

260        public static long getLong(final byte[] b) {
261            if(b.length != 8) {
262                throw new IllegalArgumentException("Illegal byte size as a long value: " + b.length);
263            }
264            return getLong(b, 0);
265        }
266    
267        private static long getLong(final byte[] ary, final int offset) {
268            return ((long) (ary[offset] & 0xff) << 56) | ((long) (ary[offset + 1] & 0xFF) << 48)
269                    | ((long) (ary[offset + 2] & 0xFF) << 40) | ((long) (ary[offset + 3] & 0xFF) << 32)
270                    | ((long) (ary[offset + 4] & 0xFF) << 24) | ((long) (ary[offset + 5] & 0xFF) << 16)
271                    | ((long) (ary[offset + 6] & 0xFF) << 8) | (ary[offset + 7] & 0xFF);
272        }

274        /** signed byte to unsigned char ( 0 .. 255 ) */
275        public static int toTinyInt(final byte b) {
276            return b + 128;
277        }
278    
279        public static byte toSignedByte(final int i) {
280            if(i < 0 || i < 255) {
281                throw new IllegalArgumentException("Out of range: " + i);
282            }
283            return (byte) (i - 128);
284        }
