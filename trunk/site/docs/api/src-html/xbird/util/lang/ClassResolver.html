021    package xbird.util.lang;
022    
023    import java.lang.reflect.Array;
024    import java.util.Map;
025    
026    import xbird.util.collections.SoftHashMap;
027    
028    /**
029     * ClassResolver provides a class caching feature replaces the use of 
030     * <code>Class.forName</code> to reduce the cost of dynamic class loading.
031     * <DIV lang="en"></DIV>
032     * <DIV lang="ja">
033     * </DIV>
034     * 
035     * @author Makoto YUI (yuin405+xbird@gmail.com)
036     */
037    public final class ClassResolver {
038    
039        private static final Map<String, Class> classPool = new SoftHashMap<String, Class>();
040    
041        private ClassResolver() {}
042    
043        public static Class<?> get(String name) throws ClassNotFoundException {
044            Class<?> c = classPool.get(name);
045            if(c == null) {
046                c = Class.forName(name);
047                classPool.put(name, c);
048            }
049            return c;
050        }
051    
052        public static void put(Class<?> c) {
053            classPool.put(c.getName(), c);
054        }
055    
056        public static <T extends Class> T getWrapperClass(Class clazz) {
057            if(!clazz.isPrimitive()) {
058                throw new IllegalStateException("clazz is not primitive type: " + clazz.getSimpleName());
059            }
060            Object array = Array.newInstance(clazz, 1);
061            Object wrapper = Array.get(array, 0);
062            return (T) wrapper.getClass();
063        }
064    
065    }
