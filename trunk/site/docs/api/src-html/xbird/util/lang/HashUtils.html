034        public static boolean isPowerOfTwo(final int x) {
035            if(x < 1) {
036                return false;
037            }
038            return (x & (x - 1)) == 0;
039        }
040    
041        public static boolean isPowerOfTwo(final long x) {
042            if(x < 1) {
043                return false;
044            }
045            return (x & (x - 1)) == 0;
046        }
047    
048        public static int nextPowerOfTwo(final int targetSize) {
049            return nextPowerOfTwo(1, targetSize);
050        }
051    
052        public static int nextPowerOfTwo(final int minSizeLog, final int targetSize) {
053            int i;
054            for(i = minSizeLog; (1 << i) < targetSize; i++)
055                ;
056            return 1 << i;
057        }
058    
059        public static int shiftsForNextPowerOfTwo(final int targetSize) {
060            int i;
061            for(i = 0; (1 << i) < targetSize; i++)
062                ;
063            return i;
064        }

066        public static int hashCode(final long l) {
067            return (int) (l ^ (l >>> 32));
068        }
069    
070        public static int hashCode(final char[] ch, final int offset, final int length) {
071            if(ch == null) {
072                return 0;
073            }
074            int result = 1;
075            final int limit = offset + length;
076            for(int i = offset; i < limit; i++) {
077                result = 31 * result + ch[i];
078            }
079            return result;
080        }
081    
082        public static int hashCode(final byte[] b, final int offset, final int length) {
083            if(b == null) {
084                return 0;
085            }
086            int result = 1;
087            final int limit = offset + length;
088            for(int i = offset; i < limit; i++) {
089                result = 31 * result + b[i];
090            }
091            return result;
092        }
093    
094        /**
095         * Returns a hash index for the given thread.  Uses a one-step
096         * FNV-1a hash code (http://www.isthe.com/chongo/tech/comp/fnv/)
097         * based on the given thread's Thread.getId().  These hash codes
098         * have more uniform distribution properties with respect to small
099         * moduli (here 1-31) than do other simple hashing functions.
100         *
101         * <p>To return an index between 0 and max, we use a cheap
102         * approximation to a mod operation, that also corrects for bias
103         * due to non-power-of-2 remaindering (see {@link
104         * java.util.Random#nextInt}).  Bits of the hashcode are masked
105         * with "nbits", the ceiling power of two of table size (looked up
106         * in a table packed into three ints).  If too large, this is
107         * retried after rotating the hash by nbits bits, while forcing new
108         * top bit to 0, which guarantees eventual termination (although
109         * with a non-random-bias).  This requires an average of less than
110         * 2 tries for all table sizes, and has a maximum 2% difference
111         * from perfectly uniform slot probabilities when applied to all
112         * possible hash codes for sizes less than 32.
113         *
114         * @return a per-thread-random index, 0 <= index <= max
115         */
116        public static final int hash(final Thread thrd, final int max) {
117            final long id = thrd.getId();
118            int hash = (((int) (id ^ (id >>> 32))) ^ 0x811c9dc5) * 0x01000193;
119    
120            final int nbits = (((0xfffffc00 >> max) & 4) | // Compute ceil(log2(m+1))
121                    ((0x000001f8 >>> max) & 2) | // The constants hold
122            ((0xffff00f2 >>> max) & 1)); // a lookup table
123            int index;
124            while((index = hash & ((1 << nbits) - 1)) > max) {// May retry on
125                hash = (hash >>> nbits) | (hash << (33 - nbits)); // non-power-2 m
126            }
127            return index;
128        }
129    
130        // Hash spreader
131        public static final int hash(final Thread thrd) {
132            int h = System.identityHashCode(thrd);
133            // You would think that System.identityHashCode on the current thread
134            // would be a good hash fcn, but actually on SunOS 5.8 it is pretty lousy
135            // in the low bits.
136            h ^= (h >>> 20) ^ (h >>> 12); // Bit spreader, borrowed from Doug Lea
137            h ^= (h >>> 7) ^ (h >>> 4);
138            return h << 2; // Pad out cache lines.  The goal is to avoid cache-line contention
139        }
