059        /**
060         * <p>Returns the length of the specified array.
061         * This method can deal with <code>Object</code> arrays and with primitive arrays.</p>
062         *
063         * <p>If the input array is <code>null</code>, <code>0</code> is returned.</p>
064         *
065         * <pre>
066         * ArrayUtils.getLength(null)            = 0
067         * ArrayUtils.getLength([])              = 0
068         * ArrayUtils.getLength([null])          = 1
069         * ArrayUtils.getLength([true, false])   = 2
070         * ArrayUtils.getLength([1, 2, 3])       = 3
071         * ArrayUtils.getLength(["a", "b", "c"]) = 3
072         * </pre>
073         *
074         * @param array  the array to retrieve the length from, may be null
075         * @return The length of the array, or <code>0</code> if the array is <code>null</code>
076         * @throws IllegalArgumentException if the object arguement is not an array.
077         */
078        public static int getLength(final Object array) {
079            if(array == null) {
080                return 0;
081            } else {
082                return Array.getLength(array);
083            }
084        }

086        public static <T> int indexOf(final T[] array, final T value) {
087            final int alen = array.length;
088            for(int i = 0; i < alen; i++) {
089                if(value.equals(array[i])) {
090                    return i;
091                }
092            }
093            return INDEX_NOT_FOUND;
094        }

096        public static <T> int indexOf(final T[] array, final T value, final int startIndex) {
097            final int alen = array.length;
098            for(int i = startIndex; i < alen; i++) {
099                if(value.equals(array[i])) {
100                    return i;
101                }
102            }
103            return INDEX_NOT_FOUND;
104        }

106        public static int indexOf(final byte[] array, final byte valueToFind, int startIndex) {
107            if(array == null) {
108                return INDEX_NOT_FOUND;
109            }
110            if(startIndex < 0) {
111                startIndex = 0;
112            }
113            for(int i = startIndex; i < array.length; i++) {
114                if(valueToFind == array[i]) {
115                    return i;
116                }
117            }
118            return INDEX_NOT_FOUND;
119        }

123         * This method can deal with <code>Object</code> arrays and with primitive arrays.

125         *

129         * ArrayUtils.lastIndex([null])          = 0
130         * ArrayUtils.lastIndex([true, false])   = 1
131         * ArrayUtils.lastIndex([1, 2, 3])       = 2
132         * ArrayUtils.lastIndex(["a", "b", "c"]) = 2

137         * @throws IllegalArgumentException if the object arguement is not an array.

140            return ArrayUtils.getLength(array) - 1;

143        /**
144         * <p>Inserts the specified element at the specified position in the array. 
145         * Shifts the element currently at that position (if any) and any subsequent
146         * elements to the right (adds one to their indices).</p>
147         *
148         * <p>This method returns a new array with the same elements of the input
149         * array plus the given element on the specified position. The component 
150         * type of the returned array is always the same as that of the input 
151         * array.</p>
152         *
153         * <p>If the input array is <code>null</code>, a new one element array is returned
154         *  whose component type is the same as the element.</p>
155         * 
156         * <pre>
157         * ArrayUtils.insert(null, 0, null)      = [null]
158         * ArrayUtils.insert(null, 0, "a")       = ["a"]
159         * ArrayUtils.insert(["a"], 1, null)     = ["a", null]
160         * ArrayUtils.insert(["a"], 1, "b")      = ["a", "b"]
161         * ArrayUtils.insert(["a", "b"], 3, "c") = ["a", "b", "c"]
162         * </pre>
163         * 
164         * @param array  the array to add the element to, may be <code>null</code>
165         * @param index  the position of the new object
166         * @param element  the object to add
167         * @return A new array containing the existing elements and the new element
168         * @throws IndexOutOfBoundsException if the index is out of range 
169         * (index < 0 || index > array.length).
170         */
171        public static <T> T[] insert(final Object array, final int index, final Object element) {
172            if(array == null) {
173                if(index != 0) {
174                    throw new IndexOutOfBoundsException("Index: " + index + ", Length: 0");
175                }
176                Object joinedArray = Array.newInstance(element != null ? element.getClass()
177                        : Object.class, 1);
178                Array.set(joinedArray, 0, element);
179                return (T[]) joinedArray;
180            }
181            int length = getLength(array);
182            if(index > length || index < 0) {
183                throw new IndexOutOfBoundsException("Index: " + index + ", Length: " + length);
184            }
185            Object result = Array.newInstance(array.getClass().getComponentType(), length + 1);
186            System.arraycopy(array, 0, result, 0, index);
187            Array.set(result, index, element);
188            if(index < length) {
189                System.arraycopy(array, index, result, index + 1, length - index);
190            }
191            return (T[]) result;
192        }

195            long[] newArray = (long[]) copyArrayGrow1(array, Long.TYPE);

224        /**
225         * <p>Removes the element at the specified position from the specified array.
226         * All subsequent elements are shifted to the left (substracts one from
227         * their indices).</p>
228         *
229         * <p>This method returns a new array with the same elements of the input
230         * array except the element on the specified position. The component 
231         * type of the returned array is always the same as that of the input 
232         * array.</p>
233         *
234         * <p>If the input array is <code>null</code>, an IndexOutOfBoundsException
235         * will be thrown, because in that case no valid index can be specified.</p>
236         * 
237         * @param array  the array to remove the element from, may not be <code>null</code>
238         * @param index  the position of the element to be removed
239         * @return A new array containing the existing elements except the element
240         *         at the specified position.
241         * @throws IndexOutOfBoundsException if the index is out of range 
242         * (index < 0 || index >= array.length), or if the array is <code>null</code>.
243         * @since 2.1
244         */
245        public static <T> T[] remove(final T[] array, final int index) {
246            int length = getLength(array);
247            if(index < 0 || index >= length) {
248                throw new IndexOutOfBoundsException("Index: " + index + ", Length: " + length);
249            }
250            Object result = Array.newInstance(array.getClass().getComponentType(), length - 1);
251            System.arraycopy(array, 0, result, 0, index);
252            if(index < length - 1) {
253                System.arraycopy(array, index + 1, result, index, length - index - 1);
254            }
255            return (T[]) result;
256        }

258        public static <T> T[] remove(final T[] array, final int from, final int to) {
259            assert (to >= from) : to + " - " + from;
260            int length = getLength(array);
261            if(from < 0 || to >= length) {
262                throw new IndexOutOfBoundsException("from: " + from + ", to: " + to + ", Length: "
263                        + length);
264            }
265            int remsize = to - from + 1;
266            Object result = Array.newInstance(array.getClass().getComponentType(), length - remsize);
267            System.arraycopy(array, 0, result, 0, from);
268            if(to < length - 1) {
269                System.arraycopy(array, to + 1, result, from, length - to - 1);
270            }
271            return (T[]) result;
272        }

297        /**
298         * Returns a copy of the given array of size 1 greater than the argument. 
299         * The last value of the array is left to the default value.
300         * 
301         * @param array The array to copy, must not be <code>null</code>.
302         * @param newArrayComponentType If <code>array</code> is <code>null</code>, create a 
303         * size 1 array of this type.
304         * @return A new copy of the array of size 1 greater than the input.
305         */
306        private static Object copyArrayGrow1(final Object array, final Class newArrayComponentType) {
307            if(array != null) {
308                int arrayLength = Array.getLength(array);
309                Object newArray = Array.newInstance(array.getClass().getComponentType(), arrayLength + 1);
310                System.arraycopy(array, 0, newArray, 0, arrayLength);
311                return newArray;
312            } else {
313                return Array.newInstance(newArrayComponentType, 1);
314            }
315        }

317        public static boolean equals(final char[] a, final char[] a2, final int off, final int len) {
318            if(a == a2) {
319                return true;
320            }
321            if(a == null || a2 == null) {
322                return false;
323            }
324            int length = a.length;
325            if(len != length) {
326                return false;
327            }
328            for(int i = 0; i < len; i++) {
329                if(a[i] != a2[off + i]) {
330                    return false;
331                }
332            }
333            return true;
334        }

380        private static <T, U> T[] copyOfRange(final U[] original, final int from, final int to, final Class<? extends T[]> newType) {

382            if(newLength < 0) {
383                throw new IllegalArgumentException(from + " > " + to);

388            return copy;

391        public static int binarySearch(final Object[] a, final int fromIndex, final int toIndex, final Object key) {
392            int low = fromIndex;
393            int high = toIndex - 1;
394    
395            while(low <= high) {
396                int mid = (low + high) >>> 1;
397                Comparable midVal = (Comparable) a[mid];
398                int cmp = midVal.compareTo(key);
399    
400                if(cmp < 0) {
401                    low = mid + 1;
402                } else if(cmp > 0) {
403                    high = mid - 1;
404                } else {
405                    return mid; // key found
406                }
407            }
408            return -(low + 1); // key not found.
409        }

412         * Reverses the order of the given array.
413         */
414        public static void reverse(final int[] ary) {
415            if(ary == null) {
416                return;
417            }
418            int i = 0;
419            int j = ary.length - 1;
420            int tmp;
421            while(j > i) {
422                tmp = ary[j];
423                ary[j] = ary[i];
424                ary[i] = tmp;
425                j--;
426                i++;
427            }
428        }

431            if(ary == null) {
432                return null;
433            }
434            int[] b = new int[ary.length];
435            for(int i = 0; i < ary.length; i++) {
436                b[ary.length - i - 1] = ary[i];
437            }
438            return b;
439        }

455        public static boolean startsWith(final byte[] target, final byte[] prefix) {
456            final int testlen = prefix.length;
457            if(target.length < testlen) {
458                return false;
459            }
460            for(int i = 0; i < testlen; i++) {
461                if(target[i] != prefix[i]) {
462                    return false;
463                }
464            }
465            return true;
466        }

468        public static int compareTo(final byte lhs[], final byte rhs[]) {
469            if(lhs == rhs) {
470                return 0;
471            }
472            final int len = Math.min(lhs.length, rhs.length);
473            for(int i = 0; i < len; i++) {
474                if(lhs[i] != rhs[i]) {
475                    return (lhs[i] & 0xFF) - (rhs[i] & 0xFF);
476                }
477            }
478            return lhs.length - rhs.length;
479        }

502        public static int compare(final byte[] b1, final int off1, final int len1, final byte[] b2, final int off2, final int len2) {
503            for(int i = 0; i < len1 && i < len2; i++) {
504                final int d = (b1[off1 + i] & 0xFF) - (b2[off2 + i] & 0xFF);
505                if(d != 0) {
506                    return d;
507                }
508            }
509            return len1 - len2;
510        }

532        public static void shuffle(final Object[] array) {
533            final Random r = new Random();
534            final int limit = array.length;
535            for(int i = 0; i < limit; ++i) {
536                swap(array, i, r.nextInt(limit));
537            }
538        }

540        public static void shuffle(final Object[] array, final long seed) {
541            final Random r = new Random(seed);
542            final int limit = array.length;
543            for(int i = 0; i < limit; ++i) {
544                swap(array, i, r.nextInt(limit));
545            }
546        }

548        public static void swap(final Object[] array, final int i, final int j) {
549            Object o = array[i];
550            array[i] = array[j];
551            array[j] = o;
552        }
