034    /**
035     * ChainedHash implementation for int to Object hash.
036     * <DIV lang="en"></DIV>
037     * <DIV lang="ja"></DIV>
038     * 
039     * @author Makoto YUI (yuin405+xbird@gmail.com)
040     */
041    public class IntHash<V> implements Externalizable, Iterable<IntHash.BucketEntry<V>>, Closeable {
042        private static final long serialVersionUID = 1L;
043        private static final float DEFAULT_LOAD_FACTOR = 0.7f;
044    
045        private transient float _loadFactor = DEFAULT_LOAD_FACTOR;
046    
047        protected BucketEntry<V>[] _buckets;
048        protected int _mask;
049        protected int _threshold;
050        protected int _size = 0;
051    
052        private transient PagingProfile _profile = null;
053    
054        /**
055         * Create a hash table that can comfortably hold the specified number of entries.
056         * The actual table created to be is the smallest prime greater than size * 2.
057         */
058        public IntHash(int size, float loadFactor) {
059            int bucketSize = HashUtils.nextPowerOfTwo(size);
060            this._buckets = new BucketEntry[bucketSize];
061            this._mask = bucketSize - 1;
062            this._loadFactor = loadFactor;
063            this._threshold = (int) (size * loadFactor);
064        }
065    
066        public IntHash(int size) {
067            this(size, DEFAULT_LOAD_FACTOR);
068        }
069    
070        public IntHash() {}//for Object Serialization
071    
072        public int size() {
073            return _size;
074        }
075    
076        /**
077         * Put an entry for the given key number.
078         * If one already exists, old value is replaced.
079         * 
080         * @return old value for the given key. if not found, return -1.
081         */
082        public V put(int key, V value) {
083            final int bucket = indexFor(key, _mask);
084            // find an entry
085            final BucketEntry<V> first = _buckets[bucket];
086            for(BucketEntry<V> e = first; e != null; e = e.next) {
087                if(key == e.key) {
088                    V replaced = e.value;
089                    e.value = value;
090                    e.recordAccess(this);
091                    return replaced; // found
092                }
093            }
094            // if not found, create a new entry.
095            addEntry(bucket, key, value, first);
096            return null;
097        }
098    
099        public V putIfAbsent(int key, V value) {
100            final int bucket = indexFor(key, _mask);
101            // find an entry
102            final BucketEntry<V> first = _buckets[bucket];
103            for(BucketEntry<V> e = first; e != null; e = e.next) {
104                if(key == e.key) {
105                    return e.value;
106                }
107            }
108            // if not found, create a new entry.
109            addEntry(bucket, key, value, first);
110            return null;
111        }
112    
113        public boolean contains(int key) {
114            final int bucket = indexFor(key, _mask);
115            for(BucketEntry e = _buckets[bucket]; e != null; e = e.next) {
116                if(key == e.key) {
117                    return true;
118                }
119            }
120            return false;
121        }
122    
123        public final V get(final int key) {
124            final BucketEntry<V>[] entries = _buckets;
125            final int bucket = indexFor(key, _mask);
126            for(BucketEntry<V> e = entries[bucket]; e != null; e = e.next) {
127                if(key == e.key) {
128                    e.recordAccess(this);
129                    return e.value;
130                }
131            }
132            return null;
133        }
134    
135        public synchronized void clear() {
136            BucketEntry tab[] = _buckets;
137            for(int i = tab.length; --i >= 0;) {
138                tab[i] = null;
139            }
140            this._size = 0;
141        }
142    
143        protected final V purge(final int key) {
144            if(_profile != null) {
145                _profile.recordPurgation(key);
146            }
147            return this.remove(key);
148        }
149    
150        public V remove(final int key) {
151            final int bucket = indexFor(key, _mask);
152            // find an entry
153            BucketEntry<V> e, prev = null;
154            for(e = _buckets[bucket]; e != null; prev = e, e = e.next) {
155                if(key == e.key) {
156                    if(prev != null) {
157                        prev.next = e.next;
158                    } else {
159                        _buckets[bucket] = e.next;
160                    }
161                    --_size;
162                    e.recordRemoval(this);
163                    return e.value;
164                }
165            }
166            return null;
167        }
168    
169        public static class BucketEntry<V> implements Externalizable {
170            private static final long serialVersionUID = 1L;
171    
172            int key;
173            V value;
174            BucketEntry<V> next;
175    
176            BucketEntry(int key, V value, BucketEntry<V> next) {
177                this.key = key;
178                this.value = value;
179                this.next = next;
180            }
181    
182            private BucketEntry(int key, V value) {
183                this(key, value, null);
184            }
185    
186            public BucketEntry() {}// for serialization
187    
188            public int getKey() {
189                return key;
190            }
191    
192            public V getValue() {
193                return value;
194            }
195    
196            public void readExternal(ObjectInput in) throws IOException, ClassNotFoundException {
197                this.key = in.readInt();
198                this.value = (V) in.readObject();
199                boolean hasNext = in.readBoolean();
200                BucketEntry<V> cur = this;
201                while(hasNext) {
202                    final int k = in.readInt();
203                    final V v = (V) in.readObject();
204                    BucketEntry<V> n = new BucketEntry<V>(k, v);
205                    cur.next = n;
206                    cur = n;
207                    hasNext = in.readBoolean();
208                }
209            }
210    
211            public void writeExternal(ObjectOutput out) throws IOException {
212                assert (value != null);
213                BucketEntry<V> cur = this;
214                while(true) {
215                    out.writeInt(cur.key);
216                    out.writeObject(cur.value);
217                    if(cur.next != null) {// hasNext
218                        out.writeBoolean(true);
219                        cur = cur.next;
220                    } else {
221                        out.writeBoolean(false);
222                        break;
223                    }
224                }
225            }
226    
227            @Override
228            public String toString() {
229                return new StringBuilder(64).append(key).append('/').append(value).toString();
230            }
231    
232            protected void recordAccess(IntHash m) {}
233    
234            protected void recordRemoval(IntHash m) {}
235    
236        }
237    
238        protected void addEntry(int bucket, int key, V value, BucketEntry<V> next) {
239            final BucketEntry<V> entry = new BucketEntry<V>(key, value, next);
240            this._buckets[bucket] = entry;
241            if(++_size > _threshold) {
242                resize(_buckets.length << 1);
243            }
244        }
245    
246        protected void resize(int newCapacity) {
247            int adequateSize = HashUtils.nextPowerOfTwo(newCapacity);
248            BucketEntry<V>[] newTable = new BucketEntry[adequateSize];
249            this._mask = newCapacity - 1;
250            rehash(newTable, newCapacity);
251            this._buckets = newTable;
252            this._threshold = (int) (newCapacity * _loadFactor);
253        }
254    
255        private void rehash(final BucketEntry<V>[] newTable, final int newsize) {
256            final int oldsize = _buckets.length;
257            for(int i = 0; i < oldsize; i++) {
258                BucketEntry<V> oldEntry = _buckets[i];
259                while(oldEntry != null) {
260                    BucketEntry<V> e = oldEntry;
261                    oldEntry = oldEntry.next;
262                    int bucket = indexFor(e.key, _mask);
263                    e.next = newTable[bucket];
264                    newTable[bucket] = e;
265                }
266            }
267        }
268    
269        private static final int indexFor(final int key, final int mask) {
270            return (key & 0x7fffffff) & mask;
271        }
272    
273        public void readExternal(ObjectInput in) throws IOException, ClassNotFoundException {
274            this._threshold = in.readInt();
275            this._size = in.readInt();
276            final int blen = in.readInt();
277            this._buckets = new BucketEntry[blen];
278            for(int i = 0; i < blen; i++) {
279                _buckets[i] = (BucketEntry<V>) in.readObject();
280            }
281        }
282    
283        public void writeExternal(ObjectOutput out) throws IOException {
284            out.writeInt(_threshold);
285            out.writeInt(_size);
286            out.writeInt(_buckets.length);
287            for(int i = 0; i < _buckets.length; i++) {
288                out.writeObject(_buckets[i]);
289            }
290        }
291    
292        public Iterator<BucketEntry<V>> iterator() {
293            return new IntIterator<V>();
294        }
295    
296        @Override
297        public String toString() {
298            final StringBuilder buf = new StringBuilder(512);
299            for(BucketEntry e : this) {
300                if(buf.length() > 0) {
301                    buf.append(", ");
302                }
303                buf.append('{');
304                buf.append(e.key);
305                buf.append('/');
306                buf.append(e.value);
307                buf.append('}');
308            }
309            return buf.toString();
310        }

312        private final class IntIterator<V> implements Iterator<BucketEntry<V>> {
313    
314            private int cursor = 0;
315            private int curBucketIndex = 0;
316            private BucketEntry<V> curBucket = null;
317    
318            IntIterator() {}
319    
320            public boolean hasNext() {
321                return _size > cursor;
322            }
323    
324            public BucketEntry<V> next() {
325                ++cursor;
326                if(curBucket == null) {
327                    for(int i = curBucketIndex; i < _buckets.length; i++) {
328                        BucketEntry e = _buckets[i];
329                        if(e != null) {
330                            this.curBucket = e;
331                            this.curBucketIndex = i + 1;
332                            break;
333                        }
334                    }
335                }
336                if(curBucket != null) {
337                    BucketEntry<V> e = curBucket;
338                    this.curBucket = e.next;
339                    return e;
340                }
341                throw new NoSuchElementException();
342            }
343    
344            public void remove() {
345                throw new UnsupportedOperationException();
346            }
347        }

349        public static final class IntLRUMap<V> extends IntHash<V> {
350            private static final long serialVersionUID = -9193071605707035207L;
351    
352            private final int maxCapacity;
353    
354            private final transient ChainedEntry<V> entryChainHeader;
355    
356            public IntLRUMap(final int limit) {
357                super(limit, 1.0f);
358                this.maxCapacity = limit;
359                this.entryChainHeader = initEntryChain();
360            }
361    
362            private ChainedEntry<V> initEntryChain() {
363                ChainedEntry<V> header = new ChainedEntry<V>(-1, null, null);
364                header.prev = header.next = header;
365                return header;
366            }
367    
368            @Override
369            protected void addEntry(int bucket, int key, V value, BucketEntry<V> next) {
370                final ChainedEntry<V> newEntry = new ChainedEntry<V>(key, value, next);
371                this._buckets[bucket] = newEntry;
372                newEntry.addBefore(entryChainHeader);
373                ++_size;
374                ChainedEntry eldest = entryChainHeader.next;
375                if(removeEldestEntry(eldest)) {
376                    purge(eldest.key);
377                } else {
378                    if(_size > _threshold) {
379                        resize(2 * _buckets.length);
380                    }
381                }
382            }
383    
384            private final boolean removeEldestEntry(BucketEntry eldest) {
385                return size() > maxCapacity;
386            }
387    
388            private static final class ChainedEntry<V> extends BucketEntry<V> {
389                private static final long serialVersionUID = 1L;
390    
391                private ChainedEntry<V> prev, next;
392    
393                ChainedEntry(int key, V value, BucketEntry<V> next) {
394                    super(key, value, next);
395                }
396    
397                @Override
398                protected void recordAccess(IntHash m) {
399                    remove();
400                    IntLRUMap lm = (IntLRUMap) m;
401                    addBefore(lm.entryChainHeader);
402                }
403    
404                @Override
405                protected void recordRemoval(IntHash m) {
406                    remove();
407                }
408    
409                /**
410                 * Removes this entry from the linked list.
411                 */
412                private void remove() {
413                    prev.next = next;
414                    next.prev = prev;
415                    prev = null;
416                    next = null;
417                }
418    
419                /**
420                 * Inserts this entry before the specified existing entry in the list.
421                 */
422                private void addBefore(ChainedEntry<V> existingEntry) {
423                    next = existingEntry;
424                    prev = existingEntry.prev;
425                    prev.next = this;
426                    next.prev = this;
427                }
428            }
429    
430            public Iterator<BucketEntry<V>> iterator() {
431                return new OrderIterator<V>(entryChainHeader);
432            }
433    
434            private class OrderIterator<V> implements Iterator<BucketEntry<V>> {
435    
436                private ChainedEntry<V> entry;
437    
438                public OrderIterator(ChainedEntry<V> e) {
439                    if(e == null) {
440                        throw new IllegalArgumentException();
441                    }
442                    this.entry = e;
443                }
444    
445                public boolean hasNext() {
446                    return entry.next != entryChainHeader;
447                }
448    
449                public BucketEntry<V> next() {
450                    entry = entry.next;
451                    if(entry == entryChainHeader) {
452                        throw new NoSuchElementException();
453                    }
454                    return entry;
455                }
456    
457                public void remove() {
458                    throw new UnsupportedOperationException();
459                }
460            }
461        }
462    
463        public void setPagingProfile(final PagingProfile profile) {
464            this._profile = profile;
465        }
466    
467        public void close() throws IOException {
468            this._buckets = null;
469        }
