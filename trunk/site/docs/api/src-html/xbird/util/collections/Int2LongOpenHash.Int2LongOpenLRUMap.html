036    public class Int2LongOpenHash implements Externalizable {
037        private static final long serialVersionUID = -8162355845665353513L;
038    
039        protected static final byte FREE = 0;
040        protected static final byte FULL = 1;
041        protected static final byte REMOVED = 2;
042    
043        private static final float DEFAULT_LOAD_FACTOR = 0.7f;
044        private static final float DEFAULT_GROW_FACTOR = 2.0f;
045    
046        protected final transient float _loadFactor;
047        protected final transient float _growFactor;
048    
049        protected int _used = 0;
050        protected int _threshold;
051    
052        protected int[] _keys;
053        protected long[] _values;
054        protected byte[] _states;
055    
056        protected Int2LongOpenHash(int size, float loadFactor, float growFactor, boolean forcePrime) {
057            if(size < 1) {
058                throw new IllegalArgumentException();
059            }
060            this._loadFactor = loadFactor;
061            this._growFactor = growFactor;
062            int actualSize = forcePrime ? Primes.findLeastPrimeNumber(size) : size;
063            this._keys = new int[actualSize];
064            this._values = new long[actualSize];
065            this._states = new byte[actualSize];
066            this._threshold = (int) (actualSize * _loadFactor);
067        }
068    
069        public Int2LongOpenHash(int size, float loadFactor, float growFactor) {
070            this(size, loadFactor, growFactor, true);
071        }
072    
073        public Int2LongOpenHash(int size) {
074            this(size, DEFAULT_LOAD_FACTOR, DEFAULT_GROW_FACTOR, true);
075        }
076    
077        public Int2LongOpenHash() {// required for serialization
078            this._loadFactor = DEFAULT_LOAD_FACTOR;
079            this._growFactor = DEFAULT_GROW_FACTOR;
080        }
081    
082        public boolean containsKey(int key) {
083            return findKey(key) >= 0;
084        }
085    
086        /**
087         * @return -1L if not found
088         */
089        public long get(int key) {
090            int i = findKey(key);
091            if(i < 0) {
092                return -1L;
093            }
094            recordAccess(i);
095            return _values[i];
096        }
097    
098        public long put(int key, long value) {
099            int hash = keyHash(key);
100            int keyLength = _keys.length;
101            int keyIdx = hash % keyLength;
102    
103            boolean expanded = preAddEntry(keyIdx);
104            if(expanded) {
105                keyLength = _keys.length;
106                keyIdx = hash % keyLength;
107            }
108    
109            int[] keys = _keys;
110            long[] values = _values;
111            byte[] states = _states;
112    
113            if(states[keyIdx] == FULL) {// second hashing
114                if(keys[keyIdx] == key) {
115                    long old = values[keyIdx];
116                    values[keyIdx] = value;
117                    recordAccess(keyIdx);
118                    return old;
119                }
120                // try second hash
121                int decr = 1 + (hash % (keyLength - 2));
122                for(;;) {
123                    keyIdx -= decr;
124                    if(keyIdx < 0) {
125                        keyIdx += keyLength;
126                    }
127                    if(isFree(keyIdx, key)) {
128                        break;
129                    }
130                    if(states[keyIdx] == FULL && keys[keyIdx] == key) {
131                        long old = values[keyIdx];
132                        values[keyIdx] = value;
133                        recordAccess(keyIdx);
134                        return old;
135                    }
136                }
137            }
138            keys[keyIdx] = key;
139            values[keyIdx] = value;
140            states[keyIdx] = FULL;
141            postAddEntry(keyIdx);
142            return -1L;
143        }
144    
145        /** Return weather the required slot is free for new entry */
146        protected boolean isFree(int index, int key) {
147            byte stat = _states[index];
148            if(stat == FREE) {
149                return true;
150            }
151            if(stat == REMOVED && _keys[index] == key) {
152                return true;
153            }
154            return false;
155        }
156    
157        /** @return expanded or not */
158        protected boolean preAddEntry(int index) {
159            if(++_used >= _threshold) {// too filled
160                int newCapacity = Math.round(_keys.length * _growFactor);
161                ensureCapacity(newCapacity);
162                return true;
163            }
164            return false;
165        }
166    
167        protected void postAddEntry(int index) {}
168    
169        protected int findKey(int key) {
170            int[] keys = _keys;
171            byte[] states = _states;
172            int keyLength = keys.length;
173    
174            int hash = keyHash(key);
175            int keyIdx = hash % keyLength;
176            if(states[keyIdx] != FREE) {
177                if(states[keyIdx] == FULL && keys[keyIdx] == key) {
178                    return keyIdx;
179                }
180                // try second hash
181                int decr = 1 + (hash % (keyLength - 2));
182                for(;;) {
183                    keyIdx -= decr;
184                    if(keyIdx < 0) {
185                        keyIdx += keyLength;
186                    }
187                    if(isFree(keyIdx, key)) {
188                        return -1;
189                    }
190                    if(states[keyIdx] == FULL && keys[keyIdx] == key) {
191                        return keyIdx;
192                    }
193                }
194            }
195            return -1;
196        }
197    
198        public long remove(int key) {
199            int[] keys = _keys;
200            long[] values = _values;
201            byte[] states = _states;
202            int keyLength = keys.length;
203    
204            int hash = keyHash(key);
205            int keyIdx = hash % keyLength;
206            if(states[keyIdx] != FREE) {
207                if(states[keyIdx] == FULL && keys[keyIdx] == key) {
208                    long old = values[keyIdx];
209                    states[keyIdx] = REMOVED;
210                    --_used;
211                    recordRemoval(keyIdx);
212                    return old;
213                }
214                //  second hash
215                int decr = 1 + (hash % (keyLength - 2));
216                for(;;) {
217                    keyIdx -= decr;
218                    if(keyIdx < 0) {
219                        keyIdx += keyLength;
220                    }
221                    if(states[keyIdx] == FREE) {
222                        return -1L;
223                    }
224                    if(states[keyIdx] == FULL && keys[keyIdx] == key) {
225                        long old = values[keyIdx];
226                        states[keyIdx] = REMOVED;
227                        --_used;
228                        recordRemoval(keyIdx);
229                        return old;
230                    }
231                }
232            }
233            return -1L;
234        }

236        public int size() {
237            return _used;
238        }
239    
240        public void clear() {
241            Arrays.fill(_states, FREE);
242            this._used = 0;
243        }

246            return new MapIterator();

249        @Override
250        public String toString() {
251            int len = size() * 10 + 2;
252            StringBuilder buf = new StringBuilder(len);
253            buf.append('{');
254            IMapIterator i = entries();
255            while(i.next() != -1) {
256                buf.append(i.getKey());
257                buf.append('=');
258                buf.append(i.getValue());
259                if(i.hasNext()) {
260                    buf.append(',');
261                }
262            }
263            buf.append('}');
264            return buf.toString();
265        }

267        protected void ensureCapacity(int newCapacity) {
268            if(_used < _threshold) {
269                throw new IllegalStateException();
270            }
271            int prime = Primes.findLeastPrimeNumber(newCapacity);
272            rehash(prime);
273            this._threshold = Math.round(prime * _loadFactor);
274        }
275    
276        private void rehash(int newCapacity) {
277            int oldCapacity = _keys.length;
278            if(newCapacity <= oldCapacity) {
279                throw new IllegalArgumentException("new: " + newCapacity + ", old: " + oldCapacity);
280            }
281            int[] newkeys = new int[newCapacity];
282            long[] newValues = new long[newCapacity];
283            byte[] newStates = new byte[newCapacity];
284            this._used = 0;
285            for(int i = 0; i < oldCapacity; i++) {
286                if(_states[i] == FULL) {
287                    _used++;
288                    int k = _keys[i];
289                    long v = _values[i];
290                    int hash = keyHash(k);
291                    int keyIdx = hash % newCapacity;
292                    if(newStates[keyIdx] == FULL) {// second hashing
293                        int decr = 1 + (hash % (newCapacity - 2));
294                        while(newStates[keyIdx] != FREE) {
295                            keyIdx -= decr;
296                            if(keyIdx < 0) {
297                                keyIdx += newCapacity;
298                            }
299                        }
300                    }
301                    newkeys[keyIdx] = k;
302                    newValues[keyIdx] = v;
303                    newStates[keyIdx] = FULL;
304                }
305            }
306            this._keys = newkeys;
307            this._values = newValues;
308            this._states = newStates;
309        }
310    
311        private static int keyHash(int key) {
312            return key & 0x7fffffff;
313        }
314    
315        protected void recordAccess(int idx) {};
316    
317        protected void recordRemoval(int idx) {};
318    
319        public void writeExternal(ObjectOutput out) throws IOException {
320            out.writeInt(_threshold);
321            out.writeInt(_used);
322    
323            out.writeInt(_keys.length);
324            IMapIterator i = entries();
325            while(i.next() != -1) {
326                out.writeInt(i.getKey());
327                out.writeLong(i.getValue());
328            }
329        }
330    
331        public void readExternal(ObjectInput in) throws IOException, ClassNotFoundException {
332            this._threshold = in.readInt();
333            this._used = in.readInt();
334    
335            int keylen = in.readInt();
336            int[] keys = new int[keylen];
337            long[] values = new long[keylen];
338            byte[] states = new byte[keylen];
339            for(int i = 0; i < _used; i++) {
340                int k = in.readInt();
341                long v = in.readLong();
342                int hash = keyHash(k);
343                int keyIdx = hash % keylen;
344                if(states[keyIdx] != FREE) {// second hash
345                    int decr = 1 + (hash % (keylen - 2));
346                    for(;;) {
347                        keyIdx -= decr;
348                        if(i < 0) {
349                            keyIdx += keylen;
350                        }
351                        if(states[keyIdx] == FREE) {
352                            break;
353                        }
354                    }
355                }
356                states[keyIdx] = FULL;
357                keys[keyIdx] = k;
358                values[keyIdx] = v;
359            }
360            this._keys = keys;
361            this._values = values;
362            this._states = states;
363        }

365        public interface IMapIterator {
366    
367            public boolean hasNext();
368    
369            /**
370             * @return -1 if not found
371             */
372            public int next();
373    
374            public int getKey();
375    
376            public long getValue();
377    
378        }

380        private final class MapIterator implements IMapIterator {
381    
382            int nextEntry;
383            int lastEntry = -1;
384    
385            MapIterator() {
386                this.nextEntry = nextEntry(0);
387            }
388    
389            /** find the index of next full entry */
390            int nextEntry(int index) {
391                while(index < _keys.length && _states[index] != FULL) {
392                    index++;
393                }
394                return index;
395            }
396    
397            public boolean hasNext() {
398                return nextEntry < _keys.length;
399            }
400    
401            public int next() {
402                if(!hasNext()) {
403                    return -1;
404                }
405                int curEntry = nextEntry;
406                this.lastEntry = curEntry;
407                this.nextEntry = nextEntry(curEntry + 1);
408                return curEntry;
409            }
410    
411            public int getKey() {
412                if(lastEntry == -1) {
413                    throw new IllegalStateException();
414                }
415                return _keys[lastEntry];
416            }
417    
418            public long getValue() {
419                if(lastEntry == -1) {
420                    throw new IllegalStateException();
421                }
422                return _values[lastEntry];
423            }
424        }

426        public static final class Int2LongOpenLRUMap extends Int2LongOpenHash {
427    
428            private final int maxCapacity;
429            private final ChainEntry evictChainHeader;
430            private final IntHash<ChainEntry> entries;
431            private boolean reachedLimit = false;
432    
433            /**
434             * allocates twice more open rooms for hash than limit, and 
435             * the hash table is fixed.
436             */
437            public Int2LongOpenLRUMap(int limit) {
438                super(limit * 2, 1.0f, limit, true);
439                this.maxCapacity = limit;
440                this.entries = new IntHash<ChainEntry>(limit);
441                this.evictChainHeader = initEntryChain();
442            }
443    
444            /**
445             * growFactor is limit / init.
446             * This semantics means expansion occourrs at most once
447             * when more than 75% of rooms are filled.
448             */
449            public Int2LongOpenLRUMap(int init, int limit) {
450                super(init, 0.75f, limit / init, true);
451                if(limit < init) {
452                    throw new IllegalArgumentException("init '" + init + "'must be less than limit '"
453                            + limit + '\'');
454                }
455                this.maxCapacity = limit;
456                this.entries = new IntHash<ChainEntry>(limit);
457                this.evictChainHeader = initEntryChain();
458            }
459    
460            private ChainEntry initEntryChain() {
461                ChainEntry header = new ChainEntry(-1);
462                header.prev = header.next = header;
463                return header;
464            }
465    
466            @Override
467            protected boolean preAddEntry(int index) {
468                ++_used;
469                if(removeEldestEntry()) {
470                    this.reachedLimit = true;
471                    ChainEntry eldest = evictChainHeader.next;
472                    if(eldest == null) {
473                        throw new IllegalStateException();
474                    }
475                    long removed = directRemove(eldest.entryIndex);
476                    if(removed == -1L) {
477                        throw new IllegalStateException();
478                    }
479                } else {
480                    // TODO REVIEWME
481                    // assert (_threshold == maxCapacity) : "threshold: " + _threshold + ", maxCapacity: " + maxCapacity;
482                    if(_used > _threshold) {// too filled
483                        throw new IllegalStateException("expansion is required for elements limited map");
484                    }
485                }
486                return false;
487            }
488    
489            private long directRemove(int index) {
490                long old = _values[index];
491                _states[index] = REMOVED;
492                --_used;
493                recordRemoval(index);
494                return old;
495            }
496    
497            @Override
498            protected void postAddEntry(int idx) {
499                ChainEntry newEntry = new ChainEntry(idx);
500                ChainEntry old = entries.put(idx, newEntry);
501                if(old != null) {
502                    throw new IllegalStateException();
503                }
504                newEntry.addBefore(evictChainHeader);
505            }
506    
507            /**
508             * Move the entry to chain head.
509             */
510            @Override
511            protected void recordAccess(int idx) {
512                ChainEntry entry = entries.get(idx);
513                if(entry == null) {
514                    throw new IllegalStateException();
515                }
516                entry.remove();
517                entry.addBefore(evictChainHeader);
518            }
519    
520            /**
521             * Remove the entry from the chain.
522             */
523            @Override
524            protected void recordRemoval(int idx) {
525                ChainEntry entry = entries.remove(idx);
526                if(entry == null) {
527                    throw new IllegalStateException();
528                }
529                entry.remove();
530            }
531    
532            private boolean removeEldestEntry() {
533                return _used > maxCapacity;
534            }
535    
536            @Override
537            protected boolean isFree(int index, int key) {
538                byte stat = _states[index];
539                return (stat == FREE) || (stat == REMOVED);
540            }
541    
542            /** Do not trust me, consider this as rough estimated size. */
543            @Override
544            public int size() {// TODO REVIEWME
545                if(reachedLimit) {
546                    return maxCapacity;
547                } else {
548                    return _used;
549                }
550            }
551    
552            @Override
553            public IMapIterator entries() {
554                return new OrderedMapIterator(evictChainHeader);
555            }
556    
557            static final class ChainEntry {
558    
559                final int entryIndex;
560                ChainEntry prev = null, next = null;
561    
562                public ChainEntry(int index) {
563                    this.entryIndex = index;
564                }
565    
566                /**
567                 * Removes this entry from the linked list.
568                 */
569                private void remove() {
570                    prev.next = next;
571                    next.prev = prev;
572                    prev = null;
573                    next = null;
574                }
575    
576                /**
577                 * Inserts this entry before the specified existing entry in the list.
578                 */
579                private void addBefore(ChainEntry existingEntry) {
580                    next = existingEntry;
581                    prev = existingEntry.prev;
582                    prev.next = this;
583                    next.prev = this;
584                }
585            }
586    
587            final class OrderedMapIterator implements IMapIterator {
588    
589                private ChainEntry entry;
590    
591                OrderedMapIterator(ChainEntry e) {
592                    assert (e != null);
593                    this.entry = e;
594                }
595    
596                public int getKey() {
597                    return _keys[entry.entryIndex];
598                }
599    
600                public long getValue() {
601                    return _values[entry.entryIndex];
602                }
603    
604                public boolean hasNext() {
605                    return entry.next != evictChainHeader;
606                }
607    
608                public int next() {
609                    ChainEntry e = entry.next;
610                    if(e == evictChainHeader) {
611                        return -1;
612                    }
613                    this.entry = e;
614                    return e.entryIndex;
615                }
616            }
617        }
