042        protected static final byte FREE = 0;
043        protected static final byte FULL = 1;
044        protected static final byte REMOVED = 2;
045    
046        private static final float DEFAULT_LOAD_FACTOR = 0.7f;
047        private static final float DEFAULT_GROW_FACTOR = 2.0f;
048    
049        protected final transient float _loadFactor;
050        protected final transient float _growFactor;
051    
052        protected int _used = 0;
053        protected int _threshold;
054    
055        protected int[] _keys;
056        protected V[] _values;
057        protected byte[] _states;

059        protected IntOpenHash(int size, float loadFactor, float growFactor, boolean forcePrime) {
060            if(size < 1) {
061                throw new IllegalArgumentException();
062            }
063            this._loadFactor = loadFactor;
064            this._growFactor = growFactor;
065            int actualSize = forcePrime ? Primes.findLeastPrimeNumber(size) : size;
066            this._keys = new int[actualSize];
067            this._values = (V[]) new Object[actualSize];
068            this._states = new byte[actualSize];
069            this._threshold = Math.round(actualSize * _loadFactor);
070        }
071    
072        public IntOpenHash(int size, float loadFactor, float growFactor) {
073            this(size, loadFactor, growFactor, true);
074        }

076        public IntOpenHash(int size) {
077            this(size, DEFAULT_LOAD_FACTOR, DEFAULT_GROW_FACTOR, true);
078        }

080        public IntOpenHash() {// required for serialization
081            this._loadFactor = DEFAULT_LOAD_FACTOR;
082            this._growFactor = DEFAULT_GROW_FACTOR;

085        public boolean containsKey(int key) {
086            return findKey(key) >= 0;
087        }

089        public final V get(final int key) {
090            final int i = findKey(key);
091            if(i < 0) {
092                return null;
093            }
094            recordAccess(i);
095            return _values[i];
096        }

098        public V put(final int key, final V value) {
099            final int hash = keyHash(key);
100            int keyLength = _keys.length;
101            int keyIdx = hash % keyLength;
102    
103            final boolean expanded = preAddEntry(keyIdx);
104            if(expanded) {
105                keyLength = _keys.length;
106                keyIdx = hash % keyLength;
107            }
108    
109            final int[] keys = _keys;
110            final V[] values = _values;
111            final byte[] states = _states;
112    
113            if(states[keyIdx] == FULL) {// second hashing
114                if(keys[keyIdx] == key) {
115                    V old = values[keyIdx];
116                    values[keyIdx] = value;
117                    recordAccess(keyIdx);
118                    return old;
119                }
120                // try second hash
121                final int decr = 1 + (hash % (keyLength - 2));
122                for(;;) {
123                    keyIdx -= decr;
124                    if(keyIdx < 0) {
125                        keyIdx += keyLength;
126                    }
127                    if(isFree(keyIdx, key)) {
128                        break;
129                    }
130                    if(states[keyIdx] == FULL && keys[keyIdx] == key) {
131                        V old = values[keyIdx];
132                        values[keyIdx] = value;
133                        recordAccess(keyIdx);
134                        return old;
135                    }
136                }
137            }
138            keys[keyIdx] = key;
139            values[keyIdx] = value;
140            states[keyIdx] = FULL;
141            postAddEntry(keyIdx);
142            return null;
143        }
144    
145        public V putIfAbsent(final int key, final V value) {
146            final int hash = keyHash(key);
147            int keyLength = _keys.length;
148            int keyIdx = hash % keyLength;
149    
150            final boolean expanded = preAddEntry(keyIdx);
151            if(expanded) {
152                keyLength = _keys.length;
153                keyIdx = hash % keyLength;
154            }
155    
156            final int[] keys = _keys;
157            final V[] values = _values;
158            final byte[] states = _states;
159    
160            if(states[keyIdx] == FULL) {// second hashing
161                if(keys[keyIdx] == key) {
162                    return values[keyIdx];
163                }
164                // try second hash
165                final int decr = 1 + (hash % (keyLength - 2));
166                for(;;) {
167                    keyIdx -= decr;
168                    if(keyIdx < 0) {
169                        keyIdx += keyLength;
170                    }
171                    if(isFree(keyIdx, key)) {
172                        break;
173                    }
174                    if(states[keyIdx] == FULL && keys[keyIdx] == key) {
175                        return values[keyIdx];
176                    }
177                }
178            }
179            keys[keyIdx] = key;
180            values[keyIdx] = value;
181            states[keyIdx] = FULL;
182            postAddEntry(keyIdx);
183            return null;
184        }

186        /** Return weather the required slot is free for new entry */
187        protected boolean isFree(int index, int key) {
188            byte stat = _states[index];
189            if(stat == FREE) {
190                return true;
191            }
192            if(stat == REMOVED && _keys[index] == key) {
193                return true;
194            }
195            return false;
196        }
197    
198        /** @return expanded or not */
199        protected boolean preAddEntry(int index) {
200            if(++_used >= _threshold) {// too filled
201                int newCapacity = Math.round(_keys.length * _growFactor);
202                ensureCapacity(newCapacity);
203                return true;
204            }
205            return false;
206        }
207    
208        protected void postAddEntry(int index) {}
209    
210        private int findKey(int key) {
211            int[] keys = _keys;
212            byte[] states = _states;
213            int keyLength = keys.length;
214    
215            int hash = keyHash(key);
216            int keyIdx = hash % keyLength;
217            if(states[keyIdx] != FREE) {
218                if(states[keyIdx] == FULL && keys[keyIdx] == key) {
219                    return keyIdx;
220                }
221                // try second hash
222                int decr = 1 + (hash % (keyLength - 2));
223                for(;;) {
224                    keyIdx -= decr;
225                    if(keyIdx < 0) {
226                        keyIdx += keyLength;
227                    }
228                    if(isFree(keyIdx, key)) {
229                        return -1;
230                    }
231                    if(states[keyIdx] == FULL && keys[keyIdx] == key) {
232                        return keyIdx;
233                    }
234                }
235            }
236            return -1;
237        }
238    
239        public V remove(int key) {
240            int[] keys = _keys;
241            V[] values = _values;
242            byte[] states = _states;
243            int keyLength = keys.length;
244    
245            int hash = keyHash(key);
246            int keyIdx = hash % keyLength;
247            if(states[keyIdx] != FREE) {
248                if(states[keyIdx] == FULL && keys[keyIdx] == key) {
249                    V old = values[keyIdx];
250                    states[keyIdx] = REMOVED;
251                    --_used;
252                    recordRemoval(keyIdx);
253                    return old;
254                }
255                //  second hash
256                int decr = 1 + (hash % (keyLength - 2));
257                for(;;) {
258                    keyIdx -= decr;
259                    if(keyIdx < 0) {
260                        keyIdx += keyLength;
261                    }
262                    if(states[keyIdx] == FREE) {
263                        return null;
264                    }
265                    if(states[keyIdx] == FULL && keys[keyIdx] == key) {
266                        V old = values[keyIdx];
267                        states[keyIdx] = REMOVED;
268                        --_used;
269                        recordRemoval(keyIdx);
270                        return old;
271                    }
272                }
273            }
274            return null;
275        }

290        @Override
291        public String toString() {
292            int len = size() * 10 + 2;
293            StringBuilder buf = new StringBuilder(len);
294            buf.append('{');
295            IMapIterator i = entries();
296            while(i.next() != -1) {
297                buf.append(i.getKey());
298                buf.append('=');
299                buf.append(i.getValue());
300                if(i.hasNext()) {
301                    buf.append(',');
302                }
303            }
304            buf.append('}');
305            return buf.toString();
306        }

308        private void ensureCapacity(int newCapacity) {
309            if(_used < _threshold) {
310                throw new IllegalStateException();
311            }
312            int prime = Primes.findLeastPrimeNumber(newCapacity);
313            rehash(prime);
314            this._threshold = Math.round(prime * _loadFactor);
315        }
316    
317        protected void rehash(int newCapacity) {
318            int oldCapacity = _keys.length;
319            if(newCapacity <= oldCapacity) {
320                throw new IllegalArgumentException("new: " + newCapacity + ", old: " + oldCapacity);
321            }
322            final int[] oldKeys = _keys;
323            final V[] oldValues = _values;
324            final byte[] oldStates = _states;
325            int[] newkeys = new int[newCapacity];
326            V[] newValues = (V[]) new Object[newCapacity];
327            byte[] newStates = new byte[newCapacity];
328            this._used = 0;
329            for(int i = 0; i < oldCapacity; i++) {
330                if(oldStates[i] == FULL) {
331                    _used++;
332                    int k = oldKeys[i];
333                    V v = oldValues[i];
334                    int hash = keyHash(k);
335                    int keyIdx = hash % newCapacity;
336                    if(newStates[keyIdx] == FULL) {// second hashing
337                        int decr = 1 + (hash % (newCapacity - 2));
338                        while(newStates[keyIdx] != FREE) {
339                            keyIdx -= decr;
340                            if(keyIdx < 0) {
341                                keyIdx += newCapacity;
342                            }
343                        }
344                    }
345                    newkeys[keyIdx] = k;
346                    newValues[keyIdx] = v;
347                    newStates[keyIdx] = FULL;
348                }
349            }
350            this._keys = newkeys;
351            this._values = newValues;
352            this._states = newStates;
353        }
354    
355        private static int keyHash(int key) {
356            return key & 0x7fffffff;
357        }

363        public void writeExternal(ObjectOutput out) throws IOException {
364            out.writeInt(_threshold);
365            out.writeInt(_used);
366    
367            out.writeInt(_keys.length);
368            IMapIterator i = entries();
369            while(i.next() != -1) {
370                out.writeInt(i.getKey());
371                out.writeObject(i.getValue());
372            }
373        }
374    
375        public void readExternal(ObjectInput in) throws IOException, ClassNotFoundException {
376            this._threshold = in.readInt();
377            this._used = in.readInt();
378    
379            int keylen = in.readInt();
380            int[] keys = new int[keylen];
381            V[] values = (V[]) new Object[keylen];
382            byte[] states = new byte[keylen];
383            for(int i = 0; i < _used; i++) {
384                int k = in.readInt();
385                V v = (V) in.readObject();
386                int hash = keyHash(k);
387                int keyIdx = hash % keylen;
388                if(states[keyIdx] != FREE) {// second hash
389                    int decr = 1 + (hash % (keylen - 2));
390                    for(;;) {
391                        keyIdx -= decr;
392                        if(i < 0) {
393                            keyIdx += keylen;
394                        }
395                        if(states[keyIdx] == FREE) {
396                            break;
397                        }
398                    }
399                }
400                states[keyIdx] = FULL;
401                keys[keyIdx] = k;
402                values[keyIdx] = v;
403            }
404            this._keys = keys;
405            this._values = values;
406            this._states = states;
407        }

409        public interface IMapIterator<V> {
410    
411            public boolean hasNext();
412    
413            public int next();
414    
415            public int getKey();
416    
417            public V getValue();
418    
419        }

421        private final class MapIterator implements IMapIterator {
422    
423            int nextEntry;
424            int lastEntry = -1;
425    
426            MapIterator() {
427                this.nextEntry = nextEntry(0);
428            }
429    
430            /** find the index of next full entry */
431            int nextEntry(int index) {
432                while(index < _keys.length && _states[index] != FULL) {
433                    index++;
434                }
435                return index;
436            }
437    
438            public boolean hasNext() {
439                return nextEntry < _keys.length;
440            }
441    
442            public int next() {
443                if(!hasNext()) {
444                    return -1;
445                }
446                int curEntry = nextEntry;
447                this.lastEntry = curEntry;
448                this.nextEntry = nextEntry(curEntry + 1);
449                return curEntry;
450            }
451    
452            public int getKey() {
453                if(lastEntry == -1) {
454                    throw new IllegalStateException();
455                }
456                return _keys[lastEntry];
457            }
458    
459            public V getValue() {
460                if(lastEntry == -1) {
461                    throw new IllegalStateException();
462                }
463                return _values[lastEntry];
464            }
465        }

467        public static final class IntOpenLRUMap<V> extends IntOpenHash<V> {
468            private static final long serialVersionUID = -1884240205310136433L;
469    
470            private final int maxCapacity;
471            private final ChainEntry evictChainHeader;
472            private boolean reachedLimit = false;
473            private ChainEntry[] _entries;
474    
475            /**
476             * allocates twice more open rooms for hash than limit, and 
477             * the hash table is fixed.
478             */
479            public IntOpenLRUMap(int limit) {
480                super(limit * 2, 1.0f, limit, true);
481                this.maxCapacity = limit;
482                this._entries = new ChainEntry[_keys.length];
483                this.evictChainHeader = initEntryChain();
484            }
485    
486            /**
487             * growFactor is limit / init.
488             * This semantics means expansion occourrs at most once
489             * when more than 75% of rooms are filled.
490             */
491            public IntOpenLRUMap(int init, int limit) {
492                super(init, 0.75f, limit / init, true);
493                if(limit < init) {
494                    throw new IllegalArgumentException("init '" + init + "'must be less than limit '"
495                            + limit + '\'');
496                }
497                this.maxCapacity = limit;
498                this._entries = new ChainEntry[_keys.length];
499                this.evictChainHeader = initEntryChain();
500            }
501    
502            private ChainEntry initEntryChain() {
503                ChainEntry header = new ChainEntry(-1);
504                header.prev = header.next = header;
505                return header;
506            }
507    
508            @Override
509            protected boolean preAddEntry(int index) {
510                ++_used;
511                if(removeEldestEntry()) {
512                    this.reachedLimit = true;
513                    ChainEntry eldest = evictChainHeader.next;
514                    if(eldest == null) {
515                        throw new IllegalStateException();
516                    }
517                    V removed = directRemove(eldest.entryIndex);
518                    if(removed == null) {
519                        throw new IllegalStateException();
520                    }
521                } else {
522                    // TODO REVIEWME
523                    // assert (_threshold == maxCapacity) : "threshold: " + _threshold + ", maxCapacity: " + maxCapacity;
524                    if(_used > _threshold) {// too filled
525                        throw new IllegalStateException("expansion is required for elements limited map");
526                    }
527                }
528                return false;
529            }
530    
531            private V directRemove(int index) {
532                V old = _values[index];
533                _states[index] = REMOVED;
534                --_used;
535                recordRemoval(index);
536                return old;
537            }
538    
539            @Override
540            protected void postAddEntry(int idx) {
541                ChainEntry newEntry = new ChainEntry(idx);
542                if(_entries[idx] != null) {
543                    throw new IllegalStateException();
544                }
545                _entries[idx] = newEntry;
546                newEntry.addBefore(evictChainHeader);
547            }
548    
549            /**
550             * Move the entry to chain head.
551             */
552            @Override
553            protected void recordAccess(int idx) {
554                ChainEntry entry = _entries[idx];
555                if(entry == null) {
556                    throw new IllegalStateException();
557                }
558                entry.remove();
559                entry.addBefore(evictChainHeader);
560            }
561    
562            /**
563             * Remove the entry from the chain.
564             */
565            @Override
566            protected void recordRemoval(int idx) {
567                ChainEntry entry = _entries[idx];
568                if(entry == null) {
569                    throw new IllegalStateException();
570                }
571                entry.remove();
572                _entries[idx] = null;
573            }
574    
575            private boolean removeEldestEntry() {
576                return _used > maxCapacity;
577            }
578    
579            @Override
580            protected boolean isFree(int index, int key) {
581                byte stat = _states[index];
582                return (stat == FREE) || (stat == REMOVED);
583            }
584    
585            /** Do not trust me, consider this as rough estimated size. */
586            @Override
587            public int size() {// TODO REVIEWME
588                if(reachedLimit) {
589                    return maxCapacity;
590                } else {
591                    return _used;
592                }
593            }
594    
595            @Override
596            public IMapIterator entries() {
597                return new OrderedMapIterator(evictChainHeader);
598            }
599    
600            @Override
601            protected final void rehash(int newCapacity) {
602                int oldCapacity = _keys.length;
603                if(newCapacity <= oldCapacity) {
604                    throw new IllegalArgumentException("new: " + newCapacity + ", old: " + oldCapacity);
605                }
606                final int[] oldKeys = _keys;
607                final V[] oldValues = _values;
608                final byte[] oldStates = _states;
609                final ChainEntry[] oldEntries = _entries;
610                final int[] newkeys = new int[newCapacity];
611                final V[] newValues = (V[]) new Object[newCapacity];
612                final byte[] newStates = new byte[newCapacity];
613                ChainEntry[] newEntries = new ChainEntry[newCapacity];
614                this._used = 0;
615                for(int i = 0; i < oldCapacity; i++) {
616                    if(oldStates[i] == FULL) {
617                        _used++;
618                        int k = oldKeys[i];
619                        V v = oldValues[i];
620                        ChainEntry e = oldEntries[i];
621                        int hash = keyHash(k);
622                        int keyIdx = hash % newCapacity;
623                        if(newStates[keyIdx] == FULL) {// second hashing
624                            int decr = 1 + (hash % (newCapacity - 2));
625                            while(newStates[keyIdx] != FREE) {
626                                keyIdx -= decr;
627                                if(keyIdx < 0) {
628                                    keyIdx += newCapacity;
629                                }
630                            }
631                        }
632                        newkeys[keyIdx] = k;
633                        newValues[keyIdx] = v;
634                        newStates[keyIdx] = FULL;
635                        newEntries[keyIdx] = e;
636                    }
637                }
638                this._keys = newkeys;
639                this._values = newValues;
640                this._states = newStates;
641                this._entries = newEntries;
642            }
643    
644            static final class ChainEntry {
645    
646                final int entryIndex;
647                ChainEntry prev = null, next = null;
648    
649                public ChainEntry(int index) {
650                    this.entryIndex = index;
651                }
652    
653                /**
654                 * Removes this entry from the linked list.
655                 */
656                private void remove() {
657                    prev.next = next;
658                    next.prev = prev;
659                    prev = null;
660                    next = null;
661                }
662    
663                /**
664                 * Inserts this entry before the specified existing entry in the list.
665                 */
666                private void addBefore(ChainEntry existingEntry) {
667                    next = existingEntry;
668                    prev = existingEntry.prev;
669                    prev.next = this;
670                    next.prev = this;
671                }
672            }
673    
674            final class OrderedMapIterator implements IMapIterator {
675    
676                private ChainEntry entry;
677    
678                OrderedMapIterator(ChainEntry e) {
679                    if(e == null) {
680                        throw new IllegalArgumentException();
681                    }
682                    this.entry = e;
683                }
684    
685                public int getKey() {
686                    return _keys[entry.entryIndex];
687                }
688    
689                public V getValue() {
690                    return _values[entry.entryIndex];
691                }
692    
693                public boolean hasNext() {
694                    return entry.next != evictChainHeader;
695                }
696    
697                public int next() {
698                    ChainEntry e = entry.next;
699                    if(e == evictChainHeader) {
700                        return -1;
701                    }
702                    this.entry = e;
703                    return e.entryIndex;
704                }
705            }
706        }
