038    package xbird.util.concurrent.reference;
039    
040    /**
041     * Reference type. Used to specify what type of reference to keep to a
042     * referent.
043     *
044     * @see java.lang.ref.Reference
045     * @author crazybob@google.com (Bob Lee)
046     * @author Makoto YUI (yuin405+xbird@gmail.com)
047     */
048    public enum ReferenceType {
049    
050        /**
051         * Prevents referent from being reclaimed by the garbage collector.
052         */
053        STRONG,
054    
055        /**
056         * Referent reclaimed in an LRU fashion when the VM runs low on memory and
057         * no strong references exist.
058         *
059         * @see java.lang.ref.SoftReference
060         */
061        SOFT,
062    
063        /**
064         * Referent reclaimed when no strong or soft references exist.
065         *
066         * @see java.lang.ref.WeakReference
067         */
068        WEAK,
069    
070        /**
071         * Similar to weak references except the garbage collector doesn't actually
072         * reclaim the referent. More flexible alternative to finalization.
073         *
074         * @see java.lang.ref.PhantomReference
075         */
076        PHANTOM;
077    }
