021    package xbird.util.concurrent.counter;
022    
023    import java.util.concurrent.atomic.AtomicInteger;
024    
025    import xbird.util.lang.HashUtils;
026    
027    /**
028     * 
029     * <DIV lang="en"></DIV>
030     * <DIV lang="ja"></DIV>
031     * 
032     * @author Makoto YUI (yuin405+xbird@gmail.com)
033     */
034    public final class StripeAtomicIntCounter implements ICounter {
035        private static final long serialVersionUID = -1886936728047307997L;
036    
037        private final AtomicInteger[] _cnts;
038        private final int _size;
039        private final int _mask;
040    
041        private volatile int _fuzzy_sum_cache;
042        private volatile long _fuzzy_time;
043    
044        public StripeAtomicIntCounter(int nstripe) {
045            assert (nstripe > 0) : nstripe;
046            final int power2 = HashUtils.nextPowerOfTwo(nstripe);
047            this._size = power2;
048            this._mask = power2 - 1;
049    
050            this._cnts = new AtomicInteger[power2];
051            for(int i = 0; i < power2; i++) {
052                _cnts[i] = new AtomicInteger();
053            }
054        }
055        
056        public StripeAtomicIntCounter(int nstripe, int initValue) {
057            assert (nstripe > 0) : nstripe;
058            final int power2 = HashUtils.nextPowerOfTwo(nstripe);
059            this._size = power2;
060            this._mask = power2 - 1;
061    
062            this._cnts = new AtomicInteger[power2];
063            _cnts[0] = new AtomicInteger(initValue);
064            for(int i = 1; i < power2; i++) {
065                _cnts[i] = new AtomicInteger();
066            }
067        }
068    
069        public int get() {
070            final AtomicInteger[] cnts = _cnts;
071            int sum = 0;
072            for(int i = 0; i < _size; i++) {
073                sum += cnts[i].get();
074            }
075            return sum;
076        }
077    
078        public int estimateGet() {
079            if(_size < 64) {
080                return get();
081            }
082            final long millis = System.currentTimeMillis();
083            if(_fuzzy_time != millis) {
084                _fuzzy_sum_cache = get();
085                _fuzzy_time = millis; // Indicate freshness of cached value
086            }
087            return _fuzzy_sum_cache;
088        }
089    
090        public void add(final int x) {
091            int idx = HashUtils.hash(Thread.currentThread(), _size - 1);
092            _cnts[idx].addAndGet(x);
093        }
094    
095        public void add2(final int x) {
096            int hash = HashUtils.hash(Thread.currentThread());
097            int idx = hash & _mask;
098            _cnts[idx].addAndGet(x);
099        }
100    
101        public int getAndAdd(final int x) {
102            int l = get();
103            add(x);
104            return l;
105        }
106    
107        public int addAndGet(final int x) {
108            add(x);
109            return get();
110        }
111    
112        public int incrementAndGet() {
113            return addAndGet(1);
114        }
115    
116        public int decrementAndGet() {
117            return addAndGet(-1);
118        }
119    
120        public int getAndIncrement() {
121            return getAndAdd(1);
122        }
123    
124        public int getAndDecrement() {
125            return getAndAdd(-1);
126        }
127    
128    }
