018    /**
019     * An unbounded {@linkplain TransferQueue} based on linked nodes.
020     * This queue orders elements FIFO (first-in-first-out) with respect
021     * to any given producer.  The <em>head</em> of the queue is that
022     * element that has been on the queue the longest time for some
023     * producer.  The <em>tail</em> of the queue is that element that has
024     * been on the queue the shortest time for some producer.
025     *
026     * <p>Beware that, unlike in most collections, the <tt>size</tt>
027     * method is <em>NOT</em> a constant-time operation. Because of the
028     * asynchronous nature of these queues, determining the current number
029     * of elements requires a traversal of the elements.
030     *
031     * <p>This class and its iterator implement all of the
032     * <em>optional</em> methods of the {@link Collection} and {@link
033     * Iterator} interfaces.
034     *
035     * <p>Memory consistency effects: As with other concurrent
036     * collections, actions in a thread prior to placing an object into a
037     * {@code LinkedTransferQueue}
038     * <a href="package-summary.html#MemoryVisibility"><i>happen-before</i></a>
039     * actions subsequent to the access or removal of that element from
040     * the {@code LinkedTransferQueue} in another thread.
041     *
042     * <p>This class is a member of the
043     * <a href="{@docRoot}/../technotes/guides/collections/index.html">
044     * Java Collections Framework</a>.
045     *
046     * @since 1.7
047     * @author Doug Lea
048     * @param <E> the type of elements held in this collection
049     *
050     */
051    public class LinkedTransferQueue<E> extends AbstractQueue<E>
052            implements TransferQueue<E>, java.io.Serializable {
053        private static final long serialVersionUID = -3223113410248163686L;
054    
055        /*
056         * This is still a work in prgress...
057         *
058         * This class extends the approach used in FIFO-mode
059         * SynchronousQueues. See the internal documentation, as well as
060         * the PPoPP 2006 paper "Scalable Synchronous Queues" by Scherer,
061         * Lea & Scott
062         * (http://www.cs.rice.edu/~wns1/papers/2006-PPoPP-SQ.pdf)
063         *
064         * The main extension is to provide different Wait modes
065         * for the main "xfer" method that puts or takes items.
066         * These don't impact the basic dual-queue logic, but instead
067         * control whether or how threads block upon insertion
068         * of request or data nodes into the dual queue.
069         */
070    
071        // Wait modes for xfer method
072        static final int NOWAIT = 0;
073        static final int TIMEOUT = 1;
074        static final int WAIT = 2;
075    
076        /** The number of CPUs, for spin control */
077        static final int NCPUS = Runtime.getRuntime().availableProcessors();
078    
079        /**
080         * The number of times to spin before blocking in timed waits.
081         * The value is empirically derived -- it works well across a
082         * variety of processors and OSes. Empirically, the best value
083         * seems not to vary with number of CPUs (beyond 2) so is just
084         * a constant.
085         */
086        static final int maxTimedSpins = (NCPUS < 2) ? 0 : 32;
087    
088        /**
089         * The number of times to spin before blocking in untimed waits.
090         * This is greater than timed value because untimed waits spin
091         * faster since they don't need to check times on each spin.
092         */
093        static final int maxUntimedSpins = maxTimedSpins * 16;
094    
095        /**
096         * The number of nanoseconds for which it is faster to spin
097         * rather than to use timed park. A rough estimate suffices.
098         */
099        static final long spinForTimeoutThreshold = 1000L;
100    
101        /** 
102         * Node class for LinkedTransferQueue. Opportunistically subclasses from
103         * AtomicReference to represent item. Uses Object, not E, to allow
104         * setting item to "this" after use, to avoid garbage
105         * retention. Similarly, setting the next field to this is used as
106         * sentinel that node is off list.
107         */
108        static final class QNode extends AtomicReference<Object> {
109            volatile QNode next;
110            volatile Thread waiter; // to control park/unpark
111            final boolean isData;
112    
113            QNode(Object item, boolean isData) {
114                super(item);
115                this.isData = isData;
116            }
117    
118            static final AtomicReferenceFieldUpdater<QNode, QNode> nextUpdater = AtomicReferenceFieldUpdater.newUpdater(QNode.class, QNode.class, "next");
119    
120            boolean casNext(QNode cmp, QNode val) {
121                return nextUpdater.compareAndSet(this, cmp, val);
122            }
123        }
124    
125        /**
126         * Padded version of AtomicReference used for head, tail and
127         * cleanMe, to alleviate contention across threads CASing one vs
128         * the other.
129         */
130        static final class PaddedAtomicReference<T> extends AtomicReference<T> {
131            // enough padding for 64bytes with 4byte refs
132            Object p0, p1, p2, p3, p4, p5, p6, p7, p8, p9, pa, pb, pc, pd, pe;
133    
134            PaddedAtomicReference(T r) {
135                super(r);
136            }
137        }

139        private final QNode dummy = new QNode(null, false);
140        private final PaddedAtomicReference<QNode> head = new PaddedAtomicReference<QNode>(dummy);
141        private final PaddedAtomicReference<QNode> tail = new PaddedAtomicReference<QNode>(dummy);
142    
143        /**
144         * Reference to a cancelled node that might not yet have been
145         * unlinked from queue because it was the last inserted node
146         * when it cancelled.
147         */
148        private final PaddedAtomicReference<QNode> cleanMe = new PaddedAtomicReference<QNode>(null);
149    
150        /**
151         * Tries to cas nh as new head; if successful, unlink
152         * old head's next node to avoid garbage retention.
153         */
154        private boolean advanceHead(QNode h, QNode nh) {
155            if(h == head.get() && head.compareAndSet(h, nh)) {
156                h.next = h; // forget old next
157                return true;
158            }
159            return false;
160        }

162        /**
163         * Puts or takes an item. Used for most queue operations (except
164         * poll() and tryTransfer())
165         * @param e the item or if null, signfies that this is a take
166         * @param mode the wait mode: NOWAIT, TIMEOUT, WAIT
167         * @param nanos timeout in nanosecs, used only if mode is TIMEOUT
168         * @return an item, or null on failure
169         */
170        private Object xfer(Object e, int mode, long nanos) {
171            boolean isData = (e != null);
172            QNode s = null;
173            final PaddedAtomicReference<QNode> head = this.head;
174            final PaddedAtomicReference<QNode> tail = this.tail;
175    
176            for(;;) {
177                QNode t = tail.get();
178                QNode h = head.get();
179    
180                if(t != null && (t == h || t.isData == isData)) {
181                    if(s == null)
182                        s = new QNode(e, isData);
183                    QNode last = t.next;
184                    if(last != null) {
185                        if(t == tail.get())
186                            tail.compareAndSet(t, last);
187                    } else if(t.casNext(null, s)) {
188                        tail.compareAndSet(t, s);
189                        return awaitFulfill(t, s, e, mode, nanos);
190                    }
191                }
192    
193                else if(h != null) {
194                    QNode first = h.next;
195                    if(t == tail.get() && first != null && advanceHead(h, first)) {
196                        Object x = first.get();
197                        if(x != first && first.compareAndSet(x, e)) {
198                            LockSupport.unpark(first.waiter);
199                            return isData ? e : x;
200                        }
201                    }
202                }
203            }
204        }
205    
206        /**
207         * Version of xfer for poll() and tryTransfer, which
208         * simpifies control paths both here and in xfer
209         */
210        private Object fulfill(Object e) {
211            boolean isData = (e != null);
212            final PaddedAtomicReference<QNode> head = this.head;
213            final PaddedAtomicReference<QNode> tail = this.tail;
214    
215            for(;;) {
216                QNode t = tail.get();
217                QNode h = head.get();
218    
219                if(t != null && (t == h || t.isData == isData)) {
220                    QNode last = t.next;
221                    if(t == tail.get()) {
222                        if(last != null)
223                            tail.compareAndSet(t, last);
224                        else
225                            return null;
226                    }
227                } else if(h != null) {
228                    QNode first = h.next;
229                    if(t == tail.get() && first != null && advanceHead(h, first)) {
230                        Object x = first.get();
231                        if(x != first && first.compareAndSet(x, e)) {
232                            LockSupport.unpark(first.waiter);
233                            return isData ? e : x;
234                        }
235                    }
236                }
237            }
238        }

240        /**
241         * Spins/blocks until node s is fulfilled or caller gives up,
242         * depending on wait mode.
243         *
244         * @param pred the predecessor of waiting node
245         * @param s the waiting node
246         * @param e the comparison value for checking match
247         * @param mode mode
248         * @param nanos timeout value
249         * @return matched item, or s if cancelled
250         */
251        private Object awaitFulfill(QNode pred, QNode s, Object e, int mode, long nanos) {
252            if(mode == NOWAIT)
253                return null;
254    
255            long lastTime = (mode == TIMEOUT) ? System.nanoTime() : 0;
256            Thread w = Thread.currentThread();
257            int spins = -1; // set to desired spin count below
258            for(;;) {
259                if(w.isInterrupted())
260                    s.compareAndSet(e, s);
261                Object x = s.get();
262                if(x != e) { // Node was matched or cancelled
263                    advanceHead(pred, s); // unlink if head
264                    if(x == s) // was cancelled
265                        return clean(pred, s);
266                    else if(x != null) {
267                        s.set(s); // avoid garbage retention
268                        return x;
269                    } else
270                        return e;
271                }
272    
273                if(mode == TIMEOUT) {
274                    long now = System.nanoTime();
275                    nanos -= now - lastTime;
276                    lastTime = now;
277                    if(nanos <= 0) {
278                        s.compareAndSet(e, s); // try to cancel
279                        continue;
280                    }
281                }
282                if(spins < 0) {
283                    QNode h = head.get(); // only spin if at head
284                    spins = ((h != null && h.next == s) ? (mode == TIMEOUT ? maxTimedSpins
285                            : maxUntimedSpins) : 0);
286                }
287                if(spins > 0)
288                    --spins;
289                else if(s.waiter == null)
290                    s.waiter = w;
291                else if(mode != TIMEOUT) {
292                    //                LockSupport.park(this);
293                    LockSupport.park(); // allows run on java5
294                    s.waiter = null;
295                    spins = -1;
296                } else if(nanos > spinForTimeoutThreshold) {
297                    //                LockSupport.parkNanos(this, nanos);
298                    LockSupport.parkNanos(nanos);
299                    s.waiter = null;
300                    spins = -1;
301                }
302            }
303        }

305        /**
306         * Gets rid of cancelled node s with original predecessor pred.
307         * @return null (to simplify use by callers)
308         */
309        private Object clean(QNode pred, QNode s) {
310            Thread w = s.waiter;
311            if(w != null) { // Wake up thread
312                s.waiter = null;
313                if(w != Thread.currentThread())
314                    LockSupport.unpark(w);
315            }
316    
317            for(;;) {
318                if(pred.next != s) // already cleaned
319                    return null;
320                QNode h = head.get();
321                QNode hn = h.next; // Absorb cancelled first node as head
322                if(hn != null && hn.next == hn) {
323                    advanceHead(h, hn);
324                    continue;
325                }
326                QNode t = tail.get(); // Ensure consistent read for tail
327                if(t == h)
328                    return null;
329                QNode tn = t.next;
330                if(t != tail.get())
331                    continue;
332                if(tn != null) { // Help advance tail
333                    tail.compareAndSet(t, tn);
334                    continue;
335                }
336                if(s != t) { // If not tail, try to unsplice
337                    QNode sn = s.next;
338                    if(sn == s || pred.casNext(s, sn))
339                        return null;
340                }
341                QNode dp = cleanMe.get();
342                if(dp != null) { // Try unlinking previous cancelled node
343                    QNode d = dp.next;
344                    QNode dn;
345                    if(d == null || // d is gone or
346                            d == dp || // d is off list or
347                            d.get() != d || // d not cancelled or
348                            (d != t && // d not tail and
349                                    (dn = d.next) != null && //   has successor
350                                    dn != d && //   that is on list
351                            dp.casNext(d, dn))) // d unspliced
352                        cleanMe.compareAndSet(dp, null);
353                    if(dp == pred)
354                        return null; // s is already saved node
355                } else if(cleanMe.compareAndSet(null, pred))
356                    return null; // Postpone cleaning s
357            }
358        }

377        public void put(E e) throws InterruptedException {
378            if(e == null)
379                throw new NullPointerException();
380            if(Thread.interrupted())
381                throw new InterruptedException();
382            xfer(e, NOWAIT, 0);
383        }

385        public boolean offer(E e, long timeout, TimeUnit unit) throws InterruptedException {
386            if(e == null)
387                throw new NullPointerException();
388            if(Thread.interrupted())
389                throw new InterruptedException();
390            xfer(e, NOWAIT, 0);
391            return true;
392        }

401        public void transfer(E e) throws InterruptedException {
402            if(e == null)
403                throw new NullPointerException();
404            if(xfer(e, WAIT, 0) == null) {
405                Thread.interrupted();
406                throw new InterruptedException();
407            }
408        }
409    
410        public boolean tryTransfer(E e, long timeout, TimeUnit unit) throws InterruptedException {
411            if(e == null)
412                throw new NullPointerException();
413            if(xfer(e, TIMEOUT, unit.toNanos(timeout)) != null)
414                return true;
415            if(!Thread.interrupted())
416                return false;
417            throw new InterruptedException();
418        }
419    
420        public boolean tryTransfer(E e) {
421            if(e == null)
422                throw new NullPointerException();
423            return fulfill(e) != null;
424        }

426        public E take() throws InterruptedException {
427            Object e = xfer(null, WAIT, 0);
428            if(e != null)
429                return (E) e;
430            Thread.interrupted();
431            throw new InterruptedException();
432        }

434        public E poll(long timeout, TimeUnit unit) throws InterruptedException {
435            Object e = xfer(null, TIMEOUT, unit.toNanos(timeout));
436            if(e != null || !Thread.interrupted())
437                return (E) e;
438            throw new InterruptedException();
439        }

441        public E poll() {
442            return (E) fulfill(null);

445        public int drainTo(Collection<? super E> c) {
446            if(c == null)
447                throw new NullPointerException();
448            if(c == this)
449                throw new IllegalArgumentException();
450            int n = 0;
451            E e;
452            while((e = poll()) != null) {
453                c.add(e);
454                ++n;
455            }
456            return n;
457        }
458    
459        public int drainTo(Collection<? super E> c, int maxElements) {
460            if(c == null)
461                throw new NullPointerException();
462            if(c == this)
463                throw new IllegalArgumentException();
464            int n = 0;
465            E e;
466            while(n < maxElements && (e = poll()) != null) {
467                c.add(e);
468                ++n;
469            }
470            return n;
471        }

475        /**
476         * Return head after performing any outstanding helping steps
477         */
478        private QNode traversalHead() {
479            for(;;) {
480                QNode t = tail.get();
481                QNode h = head.get();
482                if(h != null && t != null) {
483                    QNode last = t.next;
484                    QNode first = h.next;
485                    if(t == tail.get()) {
486                        if(last != null)
487                            tail.compareAndSet(t, last);
488                        else if(first != null) {
489                            Object x = first.get();
490                            if(x == first)
491                                advanceHead(h, first);
492                            else
493                                return h;
494                        } else
495                            return h;
496                    }
497                }
498            }
499        }

501        public Iterator<E> iterator() {
502            return new Itr();
503        }

505        /**
506         * Iterators. Basic strategy os to travers list, treating
507         * non-data (i.e., request) nodes as terminating list.
508         * Once a valid data node is found, the item is cached
509         * so that the next call to next() will return it even
510         * if subsequently removed.
511         */
512        class Itr implements Iterator<E> {
513            QNode nextNode; // Next node to return next
514            QNode currentNode; // last returned node, for remove()
515            QNode prevNode; // predecessor of last returned node
516            E nextItem; // Cache of next item, once commited to in next
517    
518            Itr() {
519                nextNode = traversalHead();
520                advance();
521            }
522    
523            E advance() {
524                prevNode = currentNode;
525                currentNode = nextNode;
526                E x = nextItem;
527    
528                QNode p = nextNode.next;
529                for(;;) {
530                    if(p == null || !p.isData) {
531                        nextNode = null;
532                        nextItem = null;
533                        return x;
534                    }
535                    Object item = p.get();
536                    if(item != p && item != null) {
537                        nextNode = p;
538                        nextItem = (E) item;
539                        return x;
540                    }
541                    prevNode = p;
542                    p = p.next;
543                }
544            }
545    
546            public boolean hasNext() {
547                return nextNode != null;
548            }
549    
550            public E next() {
551                if(nextNode == null)
552                    throw new NoSuchElementException();
553                return advance();
554            }
555    
556            public void remove() {
557                QNode p = currentNode;
558                QNode prev = prevNode;
559                if(prev == null || p == null)
560                    throw new IllegalStateException();
561                Object x = p.get();
562                if(x != null && x != p && p.compareAndSet(x, p))
563                    clean(prev, p);
564            }
565        }
566    
567        public E peek() {
568            for(;;) {
569                QNode h = traversalHead();
570                QNode p = h.next;
571                if(p == null)
572                    return null;
573                Object x = p.get();
574                if(p != x) {
575                    if(!p.isData)
576                        return null;
577                    if(x != null)
578                        return (E) x;
579                }
580            }
581        }
582    
583        public boolean isEmpty() {
584            for(;;) {
585                QNode h = traversalHead();
586                QNode p = h.next;
587                if(p == null)
588                    return true;
589                Object x = p.get();
590                if(p != x) {
591                    if(!p.isData)
592                        return true;
593                    if(x != null)
594                        return false;
595                }
596            }
597        }
598    
599        public boolean hasWaitingConsumer() {
600            for(;;) {
601                QNode h = traversalHead();
602                QNode p = h.next;
603                if(p == null)
604                    return false;
605                Object x = p.get();
606                if(p != x)
607                    return !p.isData;
608            }
609        }

611        /**
612         * Returns the number of elements in this queue.  If this queue
613         * contains more than <tt>Integer.MAX_VALUE</tt> elements, returns
614         * <tt>Integer.MAX_VALUE</tt>.
615         *
616         * <p>Beware that, unlike in most collections, this method is
617         * <em>NOT</em> a constant-time operation. Because of the
618         * asynchronous nature of these queues, determining the current
619         * number of elements requires an O(n) traversal.
620         *
621         * @return the number of elements in this queue
622         */
623        public int size() {
624            int count = 0;
625            QNode h = traversalHead();
626            for(QNode p = h.next; p != null && p.isData; p = p.next) {
627                Object x = p.get();
628                if(x != null && x != p) {
629                    if(++count == Integer.MAX_VALUE) // saturated
630                        break;
631                }
632            }
633            return count;
634        }
635    
636        public int getWaitingConsumerCount() {
637            int count = 0;
638            QNode h = traversalHead();
639            for(QNode p = h.next; p != null && !p.isData; p = p.next) {
640                if(p.get() == null) {
641                    if(++count == Integer.MAX_VALUE)
642                        break;
643                }
644            }
645            return count;
646        }
647    
648        public int remainingCapacity() {
649            return Integer.MAX_VALUE;
650        }

652        /**
653         * Save the state to a stream (that is, serialize it).
654         *
655         * @serialData All of the elements (each an <tt>E</tt>) in
656         * the proper order, followed by a null
657         * @param s the stream
658         */
659        private void writeObject(java.io.ObjectOutputStream s) throws java.io.IOException {
660            s.defaultWriteObject();
661            for(Iterator<E> it = iterator(); it.hasNext();)
662                s.writeObject(it.next());
663            // Use trailing null as sentinel
664            s.writeObject(null);
665        }

667        /**
668         * Reconstitute the Queue instance from a stream (that is,
669         * deserialize it).
670         * @param s the stream
671         */
672        private void readObject(java.io.ObjectInputStream s) throws java.io.IOException,
673                ClassNotFoundException {
674            s.defaultReadObject();
675            for(;;) {
676                E item = (E) s.readObject();
677                if(item == null)
678                    break;
679                else
680                    offer(item);
681            }
682        }
