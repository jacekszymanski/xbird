039    /**
040     * This extends the public domain version of java.util.concurrent.ConcurrentHashMap
041     * to use the system identity hash code, instead of the class' hash code.
042     *
043     * @author Doug Lea
044     * @param <K> the type of keys maintained by this map
045     * @param <V> the type of mapped values
046     * @see java.util.concurrent.ConcurrentHashMap
047     */
048    public class ConcurrentIdentityHashMap<K, V> extends AbstractMap<K, V>
049            implements ConcurrentMap<K, V>, Serializable {
050        private static final long serialVersionUID = 7249069246763182397L;
051    
052        /*
053         * The basic strategy is to subdivide the table among Segments,
054         * each of which itself is a concurrently readable hash table.
055         */
056    
057        /* ---------------- Constants -------------- */
058    
059        /**
060         * The default initial capacity for this table,
061         * used when not otherwise specified in a constructor.
062         */
063        static final int DEFAULT_INITIAL_CAPACITY = 16;
064    
065        /**
066         * The default load factor for this table, used when not
067         * otherwise specified in a constructor.
068         */
069        static final float DEFAULT_LOAD_FACTOR = 0.75f;
070    
071        /**
072         * The default concurrency level for this table, used when not
073         * otherwise specified in a constructor.
074         */
075        static final int DEFAULT_CONCURRENCY_LEVEL = 16;
076    
077        /**
078         * The maximum capacity, used if a higher value is implicitly
079         * specified by either of the constructors with arguments.  MUST
080         * be a power of two <= 1<<30 to ensure that entries are indexable
081         * using ints.
082         */
083        static final int MAXIMUM_CAPACITY = 1 << 30;
084    
085        /**
086         * The maximum number of segments to allow; used to bound
087         * constructor arguments.
088         */
089        static final int MAX_SEGMENTS = 1 << 16; // slightly conservative
090    
091        /**
092         * Number of unsynchronized retries in size and containsValue
093         * methods before resorting to locking. This is used to avoid
094         * unbounded retries if tables undergo continuous modification
095         * which would make it impossible to obtain an accurate result.
096         */
097        static final int RETRIES_BEFORE_LOCK = 2;
098    
099        /* ---------------- Fields -------------- */
100    
101        /**
102         * Mask value for indexing into segments. The upper bits of a
103         * key's hash code are used to choose the segment.
104         */
105        final int segmentMask;
106    
107        /**
108         * Shift value for indexing within segments.
109         */
110        final int segmentShift;
111    
112        /**
113         * The segments, each of which is a specialized hash table
114         */
115        final Segment<K, V>[] segments;
116    
117        transient Set<K> keySet;
118        transient Set<Map.Entry<K, V>> entrySet;
119        transient Collection<V> values;

121        /* ---------------- Small Utilities -------------- */

123        /**
124         * Applies a supplemental hash function to a given hashCode, which
125         * defends against poor quality hash functions.  This is critical
126         * because ConcurrentHashMap uses power-of-two length hash tables,
127         * that otherwise encounter collisions for hashCodes that do not
128         * differ in lower bits.
129         */
130        private static int hash(Object x) {
131            int h = System.identityHashCode(x);
132            // This function ensures that hashCodes that differ only by
133            // constant multiples at each bit position have a bounded
134            // number of collisions (approximately 8 at default load factor).
135            h ^= (h >>> 20) ^ (h >>> 12);
136            return h ^ (h >>> 7) ^ (h >>> 4);
137        }
138    
139        /**
140         * Returns the segment that should be used for key with given hash
141         * @param hash the hash code for the key
142         * @return the segment
143         */
144        final Segment<K, V> segmentFor(int hash) {
145            return segments[(hash >>> segmentShift) & segmentMask];
146        }

150        /**
151         * ConcurrentHashMap list entry. Note that this is never exported
152         * out as a user-visible Map.Entry.
153         *
154         * Because the value field is volatile, not final, it is legal wrt
155         * the Java Memory Model for an unsynchronized reader to see null
156         * instead of initial value when read via a data race.  Although a
157         * reordering leading to this is not likely to ever actually
158         * occur, the Segment.readValueUnderLock method is used as a
159         * backup in case a null (pre-initialized) value is ever seen in
160         * an unsynchronized access method.
161         */
162        static final class HashEntry<K, V> {
163            final K key;
164            final int hash;
165            volatile V value;
166            final HashEntry<K, V> next;
167    
168            HashEntry(K key, int hash, HashEntry<K, V> next, V value) {
169                this.key = key;
170                this.hash = hash;
171                this.next = next;
172                this.value = value;
173            }
174    
175            @SuppressWarnings("unchecked")
176            static final <K, V> HashEntry<K, V>[] newArray(int i) {
177                return new HashEntry[i];
178            }
179        }

181        /**
182         * Segments are specialized versions of hash tables.  This
183         * subclasses from ReentrantLock opportunistically, just to
184         * simplify some locking and avoid separate construction.
185         */
186        static final class Segment<K, V> extends ReentrantLock implements Serializable {
187            /*
188             * Segments maintain a table of entry lists that are ALWAYS
189             * kept in a consistent state, so can be read without locking.
190             * Next fields of nodes are immutable (final).  All list
191             * additions are performed at the front of each bin. This
192             * makes it easy to check changes, and also fast to traverse.
193             * When nodes would otherwise be changed, new nodes are
194             * created to replace them. This works well for hash tables
195             * since the bin lists tend to be short. (The average length
196             * is less than two for the default load factor threshold.)
197             *
198             * Read operations can thus proceed without locking, but rely
199             * on selected uses of volatiles to ensure that completed
200             * write operations performed by other threads are
201             * noticed. For most purposes, the "count" field, tracking the
202             * number of elements, serves as that volatile variable
203             * ensuring visibility.  This is convenient because this field
204             * needs to be read in many read operations anyway:
205             *
206             *   - All (unsynchronized) read operations must first read the
207             *     "count" field, and should not look at table entries if
208             *     it is 0.
209             *
210             *   - All (synchronized) write operations should write to
211             *     the "count" field after structurally changing any bin.
212             *     The operations must not take any action that could even
213             *     momentarily cause a concurrent read operation to see
214             *     inconsistent data. This is made easier by the nature of
215             *     the read operations in Map. For example, no operation
216             *     can reveal that the table has grown but the threshold
217             *     has not yet been updated, so there are no atomicity
218             *     requirements for this with respect to reads.
219             *
220             * As a guide, all critical volatile reads and writes to the
221             * count field are marked in code comments.
222             */
223    
224            private static final long serialVersionUID = 2249069246763182397L;
225    
226            /**
227             * The number of elements in this segment's region.
228             */
229            transient volatile int count;
230    
231            /**
232             * Number of updates that alter the size of the table. This is
233             * used during bulk-read methods to make sure they see a
234             * consistent snapshot: If modCounts change during a traversal
235             * of segments computing size or checking containsValue, then
236             * we might have an inconsistent view of state so (usually)
237             * must retry.
238             */
239            transient int modCount;
240    
241            /**
242             * The table is rehashed when its size exceeds this threshold.
243             * (The value of this field is always <tt>(int)(capacity *
244             * loadFactor)</tt>.)
245             */
246            transient int threshold;
247    
248            /**
249             * The per-segment table.
250             */
251            transient volatile HashEntry<K, V>[] table;
252    
253            /**
254             * The load factor for the hash table.  Even though this value
255             * is same for all segments, it is replicated to avoid needing
256             * links to outer object.
257             * @serial
258             */
259            final float loadFactor;
260    
261            Segment(int initialCapacity, float lf) {
262                loadFactor = lf;
263                setTable(HashEntry.<K, V> newArray(initialCapacity));
264            }
265    
266            @SuppressWarnings("unchecked")
267            static final <K, V> Segment<K, V>[] newArray(int i) {
268                return new Segment[i];
269            }
270    
271            /**
272             * Sets table to new HashEntry array.
273             * Call only while holding lock or in constructor.
274             */
275            void setTable(HashEntry<K, V>[] newTable) {
276                threshold = (int) (newTable.length * loadFactor);
277                table = newTable;
278            }
279    
280            /**
281             * Returns properly casted first entry of bin for given hash.
282             */
283            HashEntry<K, V> getFirst(int hash) {
284                HashEntry<K, V>[] tab = table;
285                return tab[hash & (tab.length - 1)];
286            }
287    
288            /**
289             * Reads value field of an entry under lock. Called if value
290             * field ever appears to be null. This is possible only if a
291             * compiler happens to reorder a HashEntry initialization with
292             * its table assignment, which is legal under memory model
293             * but is not known to ever occur.
294             */
295            V readValueUnderLock(HashEntry<K, V> e) {
296                lock();
297                try {
298                    return e.value;
299                } finally {
300                    unlock();
301                }
302            }
303    
304            /* Specialized implementations of map methods */
305    
306            V get(Object key, int hash) {
307                if(count != 0) { // read-volatile
308                    HashEntry<K, V> e = getFirst(hash);
309                    while(e != null) {
310                        if(e.hash == hash && key.equals(e.key)) {
311                            V v = e.value;
312                            if(v != null)
313                                return v;
314                            return readValueUnderLock(e); // recheck
315                        }
316                        e = e.next;
317                    }
318                }
319                return null;
320            }
321    
322            boolean containsKey(Object key, int hash) {
323                if(count != 0) { // read-volatile
324                    HashEntry<K, V> e = getFirst(hash);
325                    while(e != null) {
326                        if(e.hash == hash && key.equals(e.key))
327                            return true;
328                        e = e.next;
329                    }
330                }
331                return false;
332            }
333    
334            boolean containsValue(Object value) {
335                if(count != 0) { // read-volatile
336                    HashEntry<K, V>[] tab = table;
337                    int len = tab.length;
338                    for(int i = 0; i < len; i++) {
339                        for(HashEntry<K, V> e = tab[i]; e != null; e = e.next) {
340                            V v = e.value;
341                            if(v == null) // recheck
342                                v = readValueUnderLock(e);
343                            if(value.equals(v))
344                                return true;
345                        }
346                    }
347                }
348                return false;
349            }
350    
351            boolean replace(K key, int hash, V oldValue, V newValue) {
352                lock();
353                try {
354                    HashEntry<K, V> e = getFirst(hash);
355                    while(e != null && (e.hash != hash || !key.equals(e.key)))
356                        e = e.next;
357    
358                    boolean replaced = false;
359                    if(e != null && oldValue.equals(e.value)) {
360                        replaced = true;
361                        e.value = newValue;
362                    }
363                    return replaced;
364                } finally {
365                    unlock();
366                }
367            }
368    
369            V replace(K key, int hash, V newValue) {
370                lock();
371                try {
372                    HashEntry<K, V> e = getFirst(hash);
373                    while(e != null && (e.hash != hash || !key.equals(e.key)))
374                        e = e.next;
375    
376                    V oldValue = null;
377                    if(e != null) {
378                        oldValue = e.value;
379                        e.value = newValue;
380                    }
381                    return oldValue;
382                } finally {
383                    unlock();
384                }
385            }
386    
387            V put(K key, int hash, V value, boolean onlyIfAbsent) {
388                lock();
389                try {
390                    int c = count;
391                    if(c++ > threshold) // ensure capacity
392                        rehash();
393                    HashEntry<K, V>[] tab = table;
394                    int index = hash & (tab.length - 1);
395                    HashEntry<K, V> first = tab[index];
396                    HashEntry<K, V> e = first;
397                    while(e != null && (e.hash != hash || !key.equals(e.key)))
398                        e = e.next;
399    
400                    V oldValue;
401                    if(e != null) {
402                        oldValue = e.value;
403                        if(!onlyIfAbsent)
404                            e.value = value;
405                    } else {
406                        oldValue = null;
407                        ++modCount;
408                        tab[index] = new HashEntry<K, V>(key, hash, first, value);
409                        count = c; // write-volatile
410                    }
411                    return oldValue;
412                } finally {
413                    unlock();
414                }
415            }
416    
417            void rehash() {
418                HashEntry<K, V>[] oldTable = table;
419                int oldCapacity = oldTable.length;
420                if(oldCapacity >= MAXIMUM_CAPACITY)
421                    return;
422    
423                /*
424                 * Reclassify nodes in each list to new Map.  Because we are
425                 * using power-of-two expansion, the elements from each bin
426                 * must either stay at same index, or move with a power of two
427                 * offset. We eliminate unnecessary node creation by catching
428                 * cases where old nodes can be reused because their next
429                 * fields won't change. Statistically, at the default
430                 * threshold, only about one-sixth of them need cloning when
431                 * a table doubles. The nodes they replace will be garbage
432                 * collectable as soon as they are no longer referenced by any
433                 * reader thread that may be in the midst of traversing table
434                 * right now.
435                 */
436    
437                HashEntry<K, V>[] newTable = HashEntry.newArray(oldCapacity << 1);
438                threshold = (int) (newTable.length * loadFactor);
439                int sizeMask = newTable.length - 1;
440                for(int i = 0; i < oldCapacity; i++) {
441                    // We need to guarantee that any existing reads of old Map can
442                    //  proceed. So we cannot yet null out each bin.
443                    HashEntry<K, V> e = oldTable[i];
444    
445                    if(e != null) {
446                        HashEntry<K, V> next = e.next;
447                        int idx = e.hash & sizeMask;
448    
449                        //  Single node on list
450                        if(next == null)
451                            newTable[idx] = e;
452    
453                        else {
454                            // Reuse trailing consecutive sequence at same slot
455                            HashEntry<K, V> lastRun = e;
456                            int lastIdx = idx;
457                            for(HashEntry<K, V> last = next; last != null; last = last.next) {
458                                int k = last.hash & sizeMask;
459                                if(k != lastIdx) {
460                                    lastIdx = k;
461                                    lastRun = last;
462                                }
463                            }
464                            newTable[lastIdx] = lastRun;
465    
466                            // Clone all remaining nodes
467                            for(HashEntry<K, V> p = e; p != lastRun; p = p.next) {
468                                int k = p.hash & sizeMask;
469                                HashEntry<K, V> n = newTable[k];
470                                newTable[k] = new HashEntry<K, V>(p.key, p.hash, n, p.value);
471                            }
472                        }
473                    }
474                }
475                table = newTable;
476            }
477    
478            /**
479             * Remove; match on key only if value null, else match both.
480             */
481            V remove(Object key, int hash, Object value) {
482                lock();
483                try {
484                    int c = count - 1;
485                    HashEntry<K, V>[] tab = table;
486                    int index = hash & (tab.length - 1);
487                    HashEntry<K, V> first = tab[index];
488                    HashEntry<K, V> e = first;
489                    while(e != null && (e.hash != hash || !key.equals(e.key)))
490                        e = e.next;
491    
492                    V oldValue = null;
493                    if(e != null) {
494                        V v = e.value;
495                        if(value == null || value.equals(v)) {
496                            oldValue = v;
497                            // All entries following removed node can stay
498                            // in list, but all preceding ones need to be
499                            // cloned.
500                            ++modCount;
501                            HashEntry<K, V> newFirst = e.next;
502                            for(HashEntry<K, V> p = first; p != e; p = p.next)
503                                newFirst = new HashEntry<K, V>(p.key, p.hash, newFirst, p.value);
504                            tab[index] = newFirst;
505                            count = c; // write-volatile
506                        }
507                    }
508                    return oldValue;
509                } finally {
510                    unlock();
511                }
512            }
513    
514            void clear() {
515                if(count != 0) {
516                    lock();
517                    try {
518                        HashEntry<K, V>[] tab = table;
519                        for(int i = 0; i < tab.length; i++)
520                            tab[i] = null;
521                        ++modCount;
522                        count = 0; // write-volatile
523                    } finally {
524                        unlock();
525                    }
526                }
527            }
528        }

532        /**
533         * Creates a new, empty map with the specified initial
534         * capacity, load factor and concurrency level.
535         *
536         * @param initialCapacity the initial capacity. The implementation
537         * performs internal sizing to accommodate this many elements.
538         * @param loadFactor  the load factor threshold, used to control resizing.
539         * Resizing may be performed when the average number of elements per
540         * bin exceeds this threshold.
541         * @param concurrencyLevel the estimated number of concurrently
542         * updating threads. The implementation performs internal sizing
543         * to try to accommodate this many threads.
544         * @throws IllegalArgumentException if the initial capacity is
545         * negative or the load factor or concurrencyLevel are
546         * nonpositive.
547         */
548        public ConcurrentIdentityHashMap(int initialCapacity, float loadFactor, int concurrencyLevel) {
549            if(!(loadFactor > 0) || initialCapacity < 0 || concurrencyLevel <= 0)
550                throw new IllegalArgumentException();
551    
552            if(concurrencyLevel > MAX_SEGMENTS)
553                concurrencyLevel = MAX_SEGMENTS;
554    
555            // Find power-of-two sizes best matching arguments
556            int sshift = 0;
557            int ssize = 1;
558            while(ssize < concurrencyLevel) {
559                ++sshift;
560                ssize <<= 1;
561            }
562            segmentShift = 32 - sshift;
563            segmentMask = ssize - 1;
564            this.segments = Segment.newArray(ssize);
565    
566            if(initialCapacity > MAXIMUM_CAPACITY)
567                initialCapacity = MAXIMUM_CAPACITY;
568            int c = initialCapacity / ssize;
569            if(c * ssize < initialCapacity)
570                ++c;
571            int cap = 1;
572            while(cap < c)
573                cap <<= 1;
574    
575            for(int i = 0; i < this.segments.length; ++i)
576                this.segments[i] = new Segment<K, V>(cap, loadFactor);
577        }

627            this(Math.max((int) (m.size() / DEFAULT_LOAD_FACTOR) + 1, DEFAULT_INITIAL_CAPACITY), DEFAULT_LOAD_FACTOR, DEFAULT_CONCURRENCY_LEVEL);
628            putAll(m);
629        }

631        /**
632         * Returns <tt>true</tt> if this map contains no key-value mappings.
633         *
634         * @return <tt>true</tt> if this map contains no key-value mappings
635         */
636        public boolean isEmpty() {
637            final Segment<K, V>[] segments = this.segments;
638            /*
639             * We keep track of per-segment modCounts to avoid ABA
640             * problems in which an element in one segment was added and
641             * in another removed during traversal, in which case the
642             * table was never actually empty at any point. Note the
643             * similar use of modCounts in the size() and containsValue()
644             * methods, which are the only other methods also susceptible
645             * to ABA problems.
646             */
647            int[] mc = new int[segments.length];
648            int mcsum = 0;
649            for(int i = 0; i < segments.length; ++i) {
650                if(segments[i].count != 0)
651                    return false;
652                else
653                    mcsum += mc[i] = segments[i].modCount;
654            }
655            // If mcsum happens to be zero, then we know we got a snapshot
656            // before any modifications at all were made.  This is
657            // probably common enough to bother tracking.
658            if(mcsum != 0) {
659                for(int i = 0; i < segments.length; ++i) {
660                    if(segments[i].count != 0 || mc[i] != segments[i].modCount)
661                        return false;
662                }
663            }
664            return true;
665        }

667        /**
668         * Returns the number of key-value mappings in this map.  If the
669         * map contains more than <tt>Integer.MAX_VALUE</tt> elements, returns
670         * <tt>Integer.MAX_VALUE</tt>.
671         *
672         * @return the number of key-value mappings in this map
673         */
674        public int size() {
675            final Segment<K, V>[] segments = this.segments;
676            long sum = 0;
677            long check = 0;
678            int[] mc = new int[segments.length];
679            // Try a few times to get accurate count. On failure due to
680            // continuous async changes in table, resort to locking.
681            for(int k = 0; k < RETRIES_BEFORE_LOCK; ++k) {
682                check = 0;
683                sum = 0;
684                int mcsum = 0;
685                for(int i = 0; i < segments.length; ++i) {
686                    sum += segments[i].count;
687                    mcsum += mc[i] = segments[i].modCount;
688                }
689                if(mcsum != 0) {
690                    for(int i = 0; i < segments.length; ++i) {
691                        check += segments[i].count;
692                        if(mc[i] != segments[i].modCount) {
693                            check = -1; // force retry
694                            break;
695                        }
696                    }
697                }
698                if(check == sum)
699                    break;
700            }
701            if(check != sum) { // Resort to locking all segments
702                sum = 0;
703                for(int i = 0; i < segments.length; ++i)
704                    segments[i].lock();
705                for(int i = 0; i < segments.length; ++i)
706                    sum += segments[i].count;
707                for(int i = 0; i < segments.length; ++i)
708                    segments[i].unlock();
709            }
710            if(sum > Integer.MAX_VALUE)
711                return Integer.MAX_VALUE;
712            else
713                return (int) sum;
714        }

746        /**
747         * Returns <tt>true</tt> if this map maps one or more keys to the
748         * specified value. Note: This method requires a full internal
749         * traversal of the hash table, and so is much slower than
750         * method <tt>containsKey</tt>.
751         *
752         * @param value value whose presence in this map is to be tested
753         * @return <tt>true</tt> if this map maps one or more keys to the
754         *         specified value
755         * @throws NullPointerException if the specified value is null
756         */
757        public boolean containsValue(Object value) {
758            if(value == null)
759                throw new NullPointerException();
760    
761            // See explanation of modCount use above
762    
763            final Segment<K, V>[] segments = this.segments;
764            int[] mc = new int[segments.length];
765    
766            // Try a few times without locking
767            for(int k = 0; k < RETRIES_BEFORE_LOCK; ++k) {
768                int mcsum = 0;
769                for(int i = 0; i < segments.length; ++i) {
770                    mcsum += mc[i] = segments[i].modCount;
771                    if(segments[i].containsValue(value))
772                        return true;
773                }
774                boolean cleanSweep = true;
775                if(mcsum != 0) {
776                    for(int i = 0; i < segments.length; ++i) {
777                        if(mc[i] != segments[i].modCount) {
778                            cleanSweep = false;
779                            break;
780                        }
781                    }
782                }
783                if(cleanSweep)
784                    return false;
785            }
786            // Resort to locking all segments
787            for(int i = 0; i < segments.length; ++i)
788                segments[i].lock();
789            boolean found = false;
790            try {
791                for(int i = 0; i < segments.length; ++i) {
792                    if(segments[i].containsValue(value)) {
793                        found = true;
794                        break;
795                    }
796                }
797            } finally {
798                for(int i = 0; i < segments.length; ++i)
799                    segments[i].unlock();
800            }
801            return found;
802        }

1014        abstract class HashIterator {
1015            int nextSegmentIndex;
1016            int nextTableIndex;
1017            HashEntry<K, V>[] currentTable;
1018            HashEntry<K, V> nextEntry;
1019            HashEntry<K, V> lastReturned;
1020    
1021            HashIterator() {
1022                nextSegmentIndex = segments.length - 1;
1023                nextTableIndex = -1;
1024                advance();
1025            }
1026    
1027            public boolean hasMoreElements() {
1028                return hasNext();
1029            }
1030    
1031            final void advance() {
1032                if(nextEntry != null && (nextEntry = nextEntry.next) != null)
1033                    return;
1034    
1035                while(nextTableIndex >= 0) {
1036                    if((nextEntry = currentTable[nextTableIndex--]) != null)
1037                        return;
1038                }
1039    
1040                while(nextSegmentIndex >= 0) {
1041                    Segment<K, V> seg = segments[nextSegmentIndex--];
1042                    if(seg.count != 0) {
1043                        currentTable = seg.table;
1044                        for(int j = currentTable.length - 1; j >= 0; --j) {
1045                            if((nextEntry = currentTable[j]) != null) {
1046                                nextTableIndex = j - 1;
1047                                return;
1048                            }
1049                        }
1050                    }
1051                }
1052            }
1053    
1054            public boolean hasNext() {
1055                return nextEntry != null;
1056            }
1057    
1058            HashEntry<K, V> nextEntry() {
1059                if(nextEntry == null)
1060                    throw new NoSuchElementException();
1061                lastReturned = nextEntry;
1062                advance();
1063                return lastReturned;
1064            }
1065    
1066            public void remove() {
1067                if(lastReturned == null)
1068                    throw new IllegalStateException();
1069                ConcurrentIdentityHashMap.this.remove(lastReturned.key);
1070                lastReturned = null;
1071            }
1072        }

1199        /**
1200         * Save the state of the <tt>ConcurrentHashMap</tt> instance to a
1201         * stream (i.e., serialize it).
1202         * @param s the stream
1203         * @serialData
1204         * the key (Object) and value (Object)
1205         * for each key-value mapping, followed by a null pair.
1206         * The key-value mappings are emitted in no particular order.
1207         */
1208        private void writeObject(java.io.ObjectOutputStream s) throws IOException {
1209            s.defaultWriteObject();
1210    
1211            for(int k = 0; k < segments.length; ++k) {
1212                Segment<K, V> seg = segments[k];
1213                seg.lock();
1214                try {
1215                    HashEntry<K, V>[] tab = seg.table;
1216                    for(int i = 0; i < tab.length; ++i) {
1217                        for(HashEntry<K, V> e = tab[i]; e != null; e = e.next) {
1218                            s.writeObject(e.key);
1219                            s.writeObject(e.value);
1220                        }
1221                    }
1222                } finally {
1223                    seg.unlock();
1224                }
1225            }
1226            s.writeObject(null);
1227            s.writeObject(null);
1228        }

1230        /**
1231         * Reconstitute the <tt>ConcurrentHashMap</tt> instance from a
1232         * stream (i.e., deserialize it).
1233         * @param s the stream
1234         */
1235        private void readObject(java.io.ObjectInputStream s) throws IOException, ClassNotFoundException {
1236            s.defaultReadObject();
1237    
1238            // Initialize each segment to be minimally sized, and let grow.
1239            for(int i = 0; i < segments.length; ++i) {
1240                segments[i].setTable(new HashEntry[1]);
1241            }
1242    
1243            // Read the keys and values, and put the mappings in the table
1244            for(;;) {
1245                K key = (K) s.readObject();
1246                V value = (V) s.readObject();
1247                if(key == null)
1248                    break;
1249                put(key, value);
1250            }
1251        }
