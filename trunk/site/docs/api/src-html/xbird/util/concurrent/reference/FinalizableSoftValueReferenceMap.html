021    package xbird.util.concurrent.reference;
022    
023    import java.util.concurrent.ConcurrentMap;
024    
025    import xbird.util.concurrent.collections.ConcurrentCollectionProvider;
026    
027    /**
028     * 
029     * <DIV lang="en"></DIV>
030     * <DIV lang="ja"></DIV>
031     * 
032     * @author Makoto YUI (yuin405+xbird@gmail.com)
033     */
034    public final class FinalizableSoftValueReferenceMap<K, V> extends ReferenceMap<K, V> {
035        private static final long serialVersionUID = 7200623989171305154L;
036    
037        private final ReferentFinalizer<K, V> _finalizer;
038    
039        public FinalizableSoftValueReferenceMap(ConcurrentMap<K, V> delegate, ReferentFinalizer<K, V> finalizer, ReferenceType keyReferenceType) {
040            super(delegate, keyReferenceType, ReferenceType.SOFT);
041            this._finalizer = finalizer;
042        }
043    
044        /**
045         * Instantiates with using <code>ReferenceType.STRONG</code> for keys.
046         */
047        public FinalizableSoftValueReferenceMap(ReferentFinalizer<K, V> finalizer) {
048            this(ConcurrentCollectionProvider.<K, V> createConcurrentMap(16), finalizer, ReferenceType.STRONG);
049        }
050    
051        @Override
052        Object referenceValue(Object keyReference, V value) {
053            return new FinalizableSoftValueReference(keyReference, value);
054        }
055    
056        final class FinalizableSoftValueReference extends FinalizableSoftReference<V>
057                implements InternalReference {
058            final Object keyReference;
059            final V value;
060    
061            public FinalizableSoftValueReference(Object keyReference, V value) {
062                super(value);
063                this.keyReference = keyReference;
064                this.value = value;
065            }
066    
067            public void finalizeReferent() {
068                delegate.remove(keyReference, this);
069                _finalizer.finalize(dereferenceKey(keyReference), value);
070            }
071    
072            @Override
073            public boolean equals(Object obj) {
074                return referenceEquals(this, obj);
075            }
076        }
077    }
