021    package xbird.util;
022    
023    /**
024     * StopWatch provides a API for timings.
025     * 
026     * @author Makoto YUI (yuin405+xbird@gmail.com)
027     */
028    public final class StopWatch {
029    
030        private final String label;
031        private long begin = 0;
032        private long end = 0;
033    
034        public StopWatch() {
035            this(null);
036        }
037    
038        public StopWatch(String label) {
039            this.label = label;
040            start();
041        }
042    
043        public void start() {
044            begin = System.currentTimeMillis();
045        }
046    
047        public long stop() {
048            end = System.currentTimeMillis();
049            return end - begin;
050        }
051    
052        public void suspend() {
053            end = System.currentTimeMillis();
054        }
055    
056        public void resume() {
057            begin += (System.currentTimeMillis() - end);
058        }
059    
060        public void reset() {
061            begin = 0;
062            end = 0;
063        }
064    
065        public long elapsed() {
066            if (end != 0) {
067                return end - begin;
068            } else {
069                return System.currentTimeMillis() - begin;
070            }
071        }
072    
073        @Override
074        public String toString() {
075            final StringBuilder buf = new StringBuilder();
076            if (label != null) {
077                buf.append(label + ": ");
078            }
079            long t = elapsed();
080            long hour = t / 3600000;
081            if (hour > 0) {
082                buf.append(hour + "h ");
083                t = t % 3600000;
084            }
085            long min = t / 60000;
086            if (min > 0) {
087                buf.append(min + "m ");
088                t = t % 60000;
089            }
090            long sec = t / 1000;
091            if (sec > 0) {
092                buf.append(sec + "s ");
093                t = t % 1000;
094            }
095            if (t > 0) {
096                buf.append(t + "ms");
097            }
098            return buf.toString();
099        }
100        
101        public static String elapsedTime(long t) {
102            final StringBuilder buf = new StringBuilder();
103            long hour = t / 3600000;
104            if (hour > 0) {
105                buf.append(hour + "h ");
106                t = t % 3600000;
107            }
108            long min = t / 60000;
109            if (min > 0) {
110                buf.append(min + "m ");
111                t = t % 60000;
112            }
113            long sec = t / 1000;
114            if (sec > 0) {
115                buf.append(sec + "s ");
116                t = t % 1000;
117            }
118            if (t > 0) {
119                buf.append(t + "ms");
120            }
121            return buf.toString();
122        }
123    }
