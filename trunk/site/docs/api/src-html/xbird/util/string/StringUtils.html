034        public static String toBitString(final double d) {
035            final char[] bit = new char[64];
036            final long dd = Double.doubleToLongBits(d);
037            long mask = 1L;
038            for(int i = 0; i < 64; i++) {
039                final long bitval = dd & mask;
040                if(bitval == 0) {
041                    bit[63 - i] = '0';
042                } else {
043                    bit[63 - i] = '1';
044                }
045                mask <<= 1;
046            }
047            return String.valueOf(bit);
048        }

096        public static String toBitString(final byte[] b) {
097            final char[] bits = new char[8 * b.length];
098            for(int i = 0; i < b.length; i++) {
099                final byte byteval = b[i];
100                int bytei = i << 3;
101                int mask = 0x1;
102                for(int j = 7; j >= 0; j--) {
103                    final int bitval = byteval & mask;
104                    if(bitval == 0) {
105                        bits[bytei + j] = '0';
106                    } else {
107                        bits[bytei + j] = '1';
108                    }
109                    mask <<= 1;
110                }
111            }
112            return String.valueOf(bits);
113        }

119        /**
120         * @param stripChars if null, remove leading unicode whitespaces.
121         */
122        public static CharSequence strip(CharSequence src, String stripChars) {
123            if(src == null || src.length() == 0) {
124                return src;
125            }
126            final CharSequence striped = stripStart(src, stripChars);
127            return stripEnd(striped, stripChars);
128        }
129    
130        /**
131         * @param stripChars if null, remove leading unicode whitespaces.
132         */
133        public static CharSequence stripStart(CharSequence src, String stripChars) {
134            int srclen;
135            if(src == null || (srclen = src.length()) == 0) {
136                return src;
137            }
138            int start = 0;
139            if(stripChars == null) {
140                while((start != srclen) && Character.isWhitespace(src.charAt(start))) {
141                    start++;
142                }
143            } else if(stripChars.length() == 0) {
144                return src;
145            } else {
146                while((start != srclen) && (stripChars.indexOf(src.charAt(start)) != -1)) {
147                    start++;
148                }
149            }
150            return src.subSequence(start, srclen);
151        }
152    
153        /**
154         * @param stripChars if null, remove leading unicode whitespaces.
155         */
156        public static CharSequence stripEnd(CharSequence src, String stripChars) {
157            int end;
158            if(src == null || (end = src.length()) == 0) {
159                return src;
160            }
161            if(stripChars == null) {
162                while((end != 0) && Character.isWhitespace(src.charAt(end - 1))) {
163                    end--;
164                }
165            } else if(stripChars.length() == 0) {
166                return src;
167            } else {
168                while((end != 0) && (stripChars.indexOf(src.charAt(end - 1)) != -1)) {
169                    end--;
170                }
171            }
172            return src.subSequence(0, end);
173        }

175        /**
176         * Checks whether the String a valid Java number.
177         * this code is ported from jakarta commons lang.
178         * 
179         * @link http://jakarta.apache.org/commons/lang/apidocs/org/apache/commons/lang/math/NumberUtils.html
180         */
181        public static boolean isNumber(String str) {
182            if(str == null || str.length() == 0) {
183                return false;
184            }
185            char[] chars = str.toCharArray();
186            int sz = chars.length;
187            boolean hasExp = false;
188            boolean hasDecPoint = false;
189            boolean allowSigns = false;
190            boolean foundDigit = false;
191            // deal with any possible sign up front
192            int start = (chars[0] == '-') ? 1 : 0;
193            if(sz > start + 1) {
194                if(chars[start] == '0' && chars[start + 1] == 'x') {
195                    int i = start + 2;
196                    if(i == sz) {
197                        return false; // str == "0x"
198                    }
199                    // checking hex (it can't be anything else)
200                    for(; i < chars.length; i++) {
201                        if((chars[i] < '0' || chars[i] > '9') && (chars[i] < 'a' || chars[i] > 'f')
202                                && (chars[i] < 'A' || chars[i] > 'F')) {
203                            return false;
204                        }
205                    }
206                    return true;
207                }
208            }
209            sz--; // don't want to loop to the last char, check it afterwords
210            // for type qualifiers
211            int i = start;
212            // loop to the next to last char or to the last char if we need another digit to
213            // make a valid number (e.g. chars[0..5] = "1234E")
214            while(i < sz || (i < sz + 1 && allowSigns && !foundDigit)) {
215                if(chars[i] >= '0' && chars[i] <= '9') {
216                    foundDigit = true;
217                    allowSigns = false;
218    
219                } else if(chars[i] == '.') {
220                    if(hasDecPoint || hasExp) {
221                        // two decimal points or dec in exponent   
222                        return false;
223                    }
224                    hasDecPoint = true;
225                } else if(chars[i] == 'e' || chars[i] == 'E') {
226                    // we've already taken care of hex.
227                    if(hasExp) {
228                        // two E's
229                        return false;
230                    }
231                    if(!foundDigit) {
232                        return false;
233                    }
234                    hasExp = true;
235                    allowSigns = true;
236                } else if(chars[i] == '+' || chars[i] == '-') {
237                    if(!allowSigns) {
238                        return false;
239                    }
240                    allowSigns = false;
241                    foundDigit = false; // we need a digit after the E
242                } else {
243                    return false;
244                }
245                i++;
246            }
247            if(i < chars.length) {
248                if(chars[i] >= '0' && chars[i] <= '9') {
249                    // no type qualifier, OK
250                    return true;
251                }
252                if(chars[i] == 'e' || chars[i] == 'E') {
253                    // can't have an E at the last byte
254                    return false;
255                }
256                if(!allowSigns
257                        && (chars[i] == 'd' || chars[i] == 'D' || chars[i] == 'f' || chars[i] == 'F')) {
258                    return foundDigit;
259                }
260                if(chars[i] == 'l' || chars[i] == 'L') {
261                    // not allowing L with an exponent
262                    return foundDigit && !hasExp;
263                }
264                // last character is illegal
265                return false;
266            }
267            // allowSigns is true iff the val ends in 'E'
268            // found digit it to make sure weird stuff like '.' and '1E-' doesn't pass
269            return !allowSigns && foundDigit;
270        }

272        public static boolean equals(final String trg, final char[] ch, final int offset, final int length) {
273            assert (trg != null && ch != null);
274            final int trglen = trg.length();
275            if(trglen != length) {
276                return false;
277            }
278            for(int i = offset; i < length; i++) {
279                final char tc = trg.charAt(0);
280                if(tc != ch[offset + 1]) {
281                    return false;
282                }
283            }
284            return true;
285        }

294        public static byte[] getBytes(final String s) {
295            assert (s != null);
296            final int len = s.length();
297            final byte[] b = new byte[len * 2];
298            for(int i = 0; i < len; i++) {
299                Primitives.putChar(b, i * 2, s.charAt(i));
300            }
301            return b;
302        }

308        public static String toString(byte[] b, int off, int len) {
309            final int clen = len >>> 1;
310            final char[] c = new char[clen];
311            for(int i = 0; i < clen; i++) {
312                final int j = off + (i << 1);
313                c[i] = (char) ((b[j + 1] & 0xFF) + ((b[j + 0]) << 8));
314            }
315            return new String(c);
316        }

318        private static final float KBYTES = 1024f;
319        private static final float MBYTES = KBYTES * KBYTES;
320        private static final float GBYTES = MBYTES * KBYTES;
321    
322        public static String displayBytesSize(final long n) {
323            final String size;
324            final long abs = Math.abs(n);
325            if(abs < KBYTES) {
326                size = n + " bytes";
327            } else if(abs < MBYTES) {
328                size = String.format("%.2f", n / KBYTES) + " kB";
329            } else if(abs < GBYTES) {
330                size = String.format("%.2f", n / MBYTES) + " MB";
331            } else {
332                return String.format("%.2f", n / GBYTES) + " GB";
333            }
334            return size;
335        }

337        private static final char[] HEX_DIGITS = "0123456789ABCDEF".toCharArray();
338        private static final int UPPER_NIBBLE_MASK = 0xF0;
339        private static final int LOWER_NIBBLE_MASK = 0x0F;
340    
341        public static void encodeHex(final byte b, final StringBuilder buf) {
342            final int upper = (b & UPPER_NIBBLE_MASK) >> 4;
343            final int lower = b & LOWER_NIBBLE_MASK;
344            buf.append(HEX_DIGITS[upper]);
345            buf.append(HEX_DIGITS[lower]);
346        }
347    
348        public static String encodeHex(final byte buf[]) {
349            final int buflen = buf.length;
350            final char[] ch = new char[buflen * 2];
351            for(int i = 0, j = 0; i < buf.length; i++, j += 2) {
352                final byte b = buf[i];
353                final int upper = (b & UPPER_NIBBLE_MASK) >> 4;
354                final int lower = b & LOWER_NIBBLE_MASK;
355                ch[j] = HEX_DIGITS[upper];
356                ch[j + 1] = HEX_DIGITS[lower];
357            }
358            return new String(ch);
359        }
