021    package xbird.util.net;
022    
023    import java.io.IOException;
024    import java.net.Socket;
025    import java.net.SocketAddress;
026    import java.net.SocketException;
027    import java.nio.channels.ByteChannel;
028    import java.nio.channels.DatagramChannel;
029    import java.nio.channels.SocketChannel;
030    
031    import xbird.config.Settings;
032    import xbird.util.pool.PoolableObjectFactory;
033    
034    /**
035     * 
036     * <DIV lang="en"></DIV>
037     * <DIV lang="ja"></DIV>
038     * 
039     * @author Makoto YUI (yuin405+xbird@gmail.com)
040     */
041    public class PoolableSocketChannelFactory
042            implements PoolableObjectFactory<SocketAddress, ByteChannel> {
043    
044        private static final int SWEEP_INTERVAL;
045        private static final int TIME_TO_LIVE;
046        private static final int SO_RCVBUF_SIZE;
047        static {
048            String sweep = Settings.get("xbird.remote.paging_serv.connection.sweep_interval", "60000");
049            String ttl = Settings.get("xbird.remote.paging_serv.connection.ttl", "30000");
050            String rcvbuf = Settings.get("xbird.remote.paging_serv.connection.so_rcvbufsize", "4096");
051            SWEEP_INTERVAL = Integer.parseInt(sweep);
052            TIME_TO_LIVE = Integer.parseInt(ttl);
053            SO_RCVBUF_SIZE = Integer.parseInt(rcvbuf);
054        }
055    
056        private final boolean datagram;
057        private final boolean blocking;
058    
059        public PoolableSocketChannelFactory(boolean datagram, boolean blocking) {
060            this.datagram = datagram;
061            this.blocking = blocking;
062        }
063    
064        public ByteChannel makeObject(SocketAddress sockAddr) {
065            if(datagram) {
066                return createDatagramChannel(sockAddr, blocking);
067            } else {
068                return createSocketChannel(sockAddr, blocking);
069            }
070        }
071    
072        private static SocketChannel createSocketChannel(final SocketAddress sockAddr, final boolean blocking) {
073            final SocketChannel ch;
074            try {
075                ch = SocketChannel.open();
076                ch.configureBlocking(blocking);
077            } catch (IOException e) {
078                throw new IllegalStateException(e);
079            }
080            final Socket sock = ch.socket();
081            try {
082                sock.setReceiveBufferSize(SO_RCVBUF_SIZE);
083                //sock.setTcpNoDelay(true);
084            } catch (SocketException e) {
085                throw new IllegalStateException(e);
086            }
087            try {
088                ch.connect(sockAddr);
089            } catch (IOException e) {
090                throw new IllegalStateException(e);
091            }
092            return ch;
093        }
094    
095        private static DatagramChannel createDatagramChannel(final SocketAddress sockAddr, final boolean blocking) {
096            final DatagramChannel ch;
097            try {
098                ch = DatagramChannel.open();
099                ch.configureBlocking(blocking);
100            } catch (IOException e) {
101                throw new IllegalStateException(e);
102            }
103            try {
104                ch.socket().setBroadcast(false);
105            } catch (SocketException e) {
106                throw new IllegalStateException(e);
107            }
108            try {
109                ch.connect(sockAddr);
110            } catch (IOException e) {
111                throw new IllegalStateException(e);
112            }
113            return ch;
114        }
115    
116        public boolean validateObject(ByteChannel sock) {
117            if(sock == null) {
118                return false;
119            }
120            return sock.isOpen();
121        }
122    
123        public int getSweepInterval() {
124            return SWEEP_INTERVAL;
125        }
126    
127        public int geTimeToLive() {
128            return TIME_TO_LIVE;
129        }
130    
131    }
